package com.nextzy.myais.promotion.module.currentpackage.fragment;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.promotion.helper.currentpackage.model.MyPackageInfoData;
import com.nextzy.myais.promotion.helper.currentpackage.network.PackageInfoManager;

/**
 * Created by trusttanapruk on 9/10/2016.
 */
public class CurrentPackagePresenter extends NextzyPresenter<CurrentPackageContractor.View> implements CurrentPackageContractor.Presenter {
    private MyPackageInfoData mMyPackageInfoData;

    protected CurrentPackagePresenter(CurrentPackageContractor.View view) {
        super(view);
    }


    public static CurrentPackagePresenter createPresenter(CurrentPackageContractor.View view) {
        return new CurrentPackagePresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {
    }

    @Override
    public void callService() {
        PackageInfoManager.getInstance().getMyPackageInfoImpl();
        Mobile.getInstance().logEvent("APP_VIEW_CURRENT_PACKAGE", "5001", "2010");
    }

    @Override
    public MyPackageInfoData getMyPackageInfoData() {
        return mMyPackageInfoData;
    }

    @Override
    public void setMyPackageInfoData(MyPackageInfoData myPackageInfoData) {
        mMyPackageInfoData = myPackageInfoData;
    }

    @Override
    public void initMyPackageInfoData() {
        getView().attachAdapter(mMyPackageInfoData);
    }

    @Subscribe
    public void onMyPackageInfoDataSuccess(MyPackageInfoData data) {
        setMyPackageInfoData(data);
        initMyPackageInfoData();
    }

}
