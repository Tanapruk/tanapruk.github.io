package com.ais.mimo.eservice.config.menu.tab;

import android.support.v4.app.Fragment;

import com.ais.mimo.eservice.utility.TabMenuUtility;
import com.nextzy.myais.balanceusage.config.TabBalanceUsageKey;
import com.nextzy.myais.home.config.TabHomeMenuKey;
import com.nextzy.myais.payment.config.TabPaymentKey;
import com.nextzy.myais.payment.config.TabTopUpKey;
import com.nextzy.myais.promotion.config.TabPromotionMenuKey;
import com.nextzy.myais.service.config.TabServiceMenuKey;

/**
 * Created by Akexorcist on 9/8/2016 AD.
 */

public class TabMenuConfig {
    private static TabMenuConfig tabMenuFragmentUtility;

    public static TabMenuConfig getInstance() {
        if (tabMenuFragmentUtility == null) {
            tabMenuFragmentUtility = new TabMenuConfig();
        }
        return tabMenuFragmentUtility;
    }

    public Fragment getFragment(String key) {
        TabMenuUtility utility = TabMenuUtility.getInstance();
        if (utility.isAvailable(TabHomeMenuKey.class, key)) {
            return utility.getFragment(TabHomeMenuKey.class, key);

        } else if (utility.isAvailable(TabPromotionMenuKey.class, key)) {
            return utility.getFragment(TabPromotionMenuKey.class, key);

        } else if (utility.isAvailable(TabServiceMenuKey.class, key)) {
            return utility.getFragment(TabServiceMenuKey.class, key);

//        } else if (utility.isAvailable(TabSampleBasicMenuKey.class, key)) {
//            return utility.getFragment(TabSampleBasicMenuKey.class, key);
//
//        } else if (utility.isAvailable(TabSampleListMenuKey.class, key)) {
//            return utility.getFragment(TabSampleListMenuKey.class, key);

        } else if (utility.isAvailable(TabBalanceUsageKey.class, key)) {
            return utility.getFragment(TabBalanceUsageKey.class, key);

        } else if (utility.isAvailable(TabPaymentKey.class, key)) {
            return utility.getFragment(TabPaymentKey.class, key);
        } else if (utility.isAvailable(TabTopUpKey.class, key)) {
            return utility.getFragment(TabTopUpKey.class, key);
        }
//        else if (utility.isAvailable(TabHelpMenuKey.class, key)) {
//            return utility.getFragment(TabHelpMenuKey.class, key);
//        }  else if (utility.isAvailable(TabLive360MenuKey.class, key)) {
//            return utility.getFragment(TabLive360MenuKey.class, key);
//        }
        throw new NullPointerException("Please create tab fragment for \"" + key + "\" key");
    }


}
