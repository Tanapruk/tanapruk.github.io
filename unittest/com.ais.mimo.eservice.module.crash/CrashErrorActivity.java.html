package com.ais.mimo.eservice.module.crash;

import android.content.Intent;
import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.widget.Button;

import com.ais.mimo.eservice.module.login.activity.LoginActivity;
import com.nextzy.myais.common.config.mobile.EmptyMobileException;
import com.nextzy.myais.common.mvp.NextzyActivity;
import com.nextzy.myais.myais.R;

import cat.ereza.customactivityoncrash.CustomActivityOnCrash;

public class CrashErrorActivity extends NextzyActivity {
    private Button btnRestartApp;

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_crash_error;
    }

    @Override
    public void bindView() {
        btnRestartApp = (Button) findViewById(R.id.my_ais_crash_error_btn_restart_app);
    }

    @Override
    public void setupView() {
        btnRestartApp.setOnClickListener(onRestartAppClick());
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        checkEmptyMobileExceptionOccur();
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    private View.OnClickListener onRestartAppClick() {
        return v -> {
            Intent intent = new Intent(CrashErrorActivity.this, LoginActivity.class);
            CustomActivityOnCrash.restartApplicationWithIntent(CrashErrorActivity.this, intent, null);
        };
    }

    private void checkEmptyMobileExceptionOccur() {
        String stackTrace = CustomActivityOnCrash.getStackTraceFromIntent(getIntent());
        if (stackTrace != null && stackTrace.contains(EmptyMobileException.class.getName())) {
            Intent intent = new Intent(this, LoginActivity.class);
            startActivity(intent);
        }
        Log.e(this.getClass().getSimpleName(), stackTrace);
    }
}
