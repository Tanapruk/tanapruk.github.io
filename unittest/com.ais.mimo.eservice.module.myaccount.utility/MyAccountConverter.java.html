package com.ais.mimo.eservice.module.myaccount.utility;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.utility.model.MobileTypeGroup;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.config.mobile.MobileContact;
import com.nextzy.myais.common.config.mobile.MobileType;
import com.nextzy.myais.common.config.mobile.MobileUtility;
import com.nextzy.myais.common.utility.NextzyUtility;

import org.jetbrains.annotations.NotNull;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by TheKhaeng on 11/14/2016.
 */

public class MyAccountConverter {
    public final static String STATUS_ELIGIBLE = "eligible";
    public final static String STATUS_ACTIVE = "active";
    private static MyAccountConverter util;

    public static MyAccountConverter getInstance() {
        if (util == null) {
            util = new MyAccountConverter();
        }
        return util;
    }

    public ArrayList<MobileData> convertToRegisterMyNumberList(ArrayList<MobileContact> contacts) {
        ArrayList<MobileData> myNumberList = new ArrayList<>();
        for (MobileContact contact : contacts) {
            MobileData mobile = new MobileData();
            mobile.setNumber(contact.getMobileNo());
            myNumberList.add(mobile);
        }
        return myNumberList;
    }

    public String getMasterEmailFromMobileList(List<MobileData> mobileDataList) {
        for (MobileData mobileData : mobileDataList) {
            if (isMobileDataHasEmail(mobileData)) {
                return mobileData.getEmailMyAis();
            }
        }
        return null;
    }


    public MobileTypeGroup groupMobileDataListByStatus(List<MobileData> mobileDataList,
                                                       @NotNull String status,
                                                       boolean isFilterMainNumber) {
        MobileTypeGroup model = new MobileTypeGroup();
        String currentMobileNumber = Mobile.getInstance().getMasterMobileNumber();
        if (mobileDataList != null) {
            for (MobileData mobile : mobileDataList) {
                if (status.equalsIgnoreCase(mobile.getStatusCode())) {
                    MobileType type = MobileUtility.getInstance().getMobileTypeByMobileData(
                            mobile.getChargeType(),
                            mobile.getMobileType());
                    if (isFilterMainNumber
                            && NextzyUtility.getInstance().isStringEqual(currentMobileNumber, mobile.getNumber())) {
                        continue;
                    }
                    if (MobileType.FBB.equals(type)) {
                        model.addFbbMobile(mobile);
                    } else if (MobileType.PREPAID.equals(type)) {
                        model.addPrepaidMobile(mobile);
                    } else if (MobileType.POSTPAID.equals(type)) {
                        model.addPostpaidMobile(mobile);
                    }
                }
            }
        }
        return model;
    }

    private boolean isMobileDataHasEmail(MobileData mobileData) {
        return mobileData.getNumber() != null
                && mobileData.getNumber().equals(Mobile.getInstance().getMasterMobileNumber())
                && !NextzyUtility.getInstance().isEmptyString(mobileData.getEmailMyAis());
    }
}
