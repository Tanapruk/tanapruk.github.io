package com.ais.mimo.eservice.widget;

import android.content.Context;
import android.os.Parcel;
import android.os.Parcelable;
import android.support.v7.app.AlertDialog;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.view.ViewTreeObserver;
import android.widget.BaseAdapter;
import android.widget.Button;
import android.widget.FrameLayout;
import android.widget.ListView;
import android.widget.TextView;

import com.nextzy.myais.common.delegate.FormatUtility;
import com.nextzy.myais.common.utility.NextzyUtility;
import com.nextzy.myais.myais.R;

import java.util.ArrayList;
import java.util.List;

import static com.nextzy.myais.common.delegate.FormatUtility.FORMAT_MOBILE_NUMBER;

/**
 * Created by RoyleKoonlert on 3/3/2017 AD.
 */

public class MyNumberDropDownDialog {
    private AlertDialog alertDialog;
    private PopupWindowAdapter adapter;
    private MyNumberListener myNumberListener;
    private ListView listView;

    public MyNumberDropDownDialog(String mobileNumberSelected) {
        adapter = new PopupWindowAdapter(mobileNumberSelected);
        adapter.setAdapterController(new AdapterController());
    }

    public void setItemList(List<ContentItem> contentItemList) {
        adapter.setMenuItemList(contentItemList);
    }

    public void showPopupMenu(Context context) {
        View view = LayoutInflater.from(context)
                .inflate(R.layout.my_ais_account_widget_my_number_view_popup_window, null);
        int maximumHeight = (int) context.getResources().getDimension(R.dimen.my_ais_account_drop_down_height);
        int height = adapter.getCount() <= 3 ? ViewGroup.LayoutParams.WRAP_CONTENT : maximumHeight;
        listView = (ListView) view.findViewById(R.id.my_ais_account_widget_my_number_lv);
        listView.setAdapter(adapter);

        if (adapter.getCount() > 3) {
            listView.setScrollbarFadingEnabled(false);
        }

        AlertDialog.Builder builder = new AlertDialog.Builder(context);
        builder.setView(view);
        alertDialog = builder.create();
        alertDialog.show();
        alertDialog.getWindow().setLayout(ViewGroup.LayoutParams.WRAP_CONTENT, height);
        setScrollListViewToMobileNumberSelectedPosition();
    }

    private void setScrollListViewToMobileNumberSelectedPosition() {
        listView.getViewTreeObserver().addOnGlobalLayoutListener(
                new ViewTreeObserver.OnGlobalLayoutListener() {
                    @Override
                    public void onGlobalLayout() {
                        listView.getViewTreeObserver().removeGlobalOnLayoutListener(this);
                        listView.setSelection(getMobileNumberSelectedPosition());
                    }
                });
    }

    private int getMobileNumberSelectedPosition() {
        List<ContentItem> contentItems = adapter.adapterController.getContentItemList();
        String mobileSelected = adapter.getMobileNumberSelected();
        for (int i = 0; i < contentItems.size(); i++) {
            if (contentItems.get(i).getMobileNumber().equals(mobileSelected)) {
                if (i != 0)
                    i -= 1;
                return i;
            }
        }
        return 0;
    }

    public void closePopupMenu() {
        if (alertDialog != null) {
            alertDialog.dismiss();
            alertDialog = null;
            if (myNumberListener != null) {
                myNumberListener.onDismiss();
            }
        }
    }

    public void setMyNumberListener(MyNumberListener myNumberListener) {
        this.myNumberListener = myNumberListener;
    }

    public void onDismiss() {
        closePopupMenu();
    }

    public class PopupWindowAdapter extends BaseAdapter {
        private String mobileNumberSelected;
        private AdapterController adapterController;
        private OnMyNumberClickListener onMyNumberClickListener;

        public PopupWindowAdapter(String mobileNumberSelected) {
            this.mobileNumberSelected = mobileNumberSelected;
        }

        public String getMobileNumberSelected() {
            return mobileNumberSelected;
        }

        public void setAdapterController(AdapterController adapterController) {
            this.adapterController = adapterController;
            setOnMyNumberClickListener(adapterController);
        }

        public void setOnMyNumberClickListener(OnMyNumberClickListener onMyNumberClickListener) {
            this.onMyNumberClickListener = onMyNumberClickListener;
        }

        public void setMenuItemList(List<ContentItem> contentItemList) {
            if (adapterController != null) {
                adapterController.setContentItemList(contentItemList);
            }
        }

        @Override
        public int getCount() {
            return adapterController != null ? adapterController.getAllItemCount() : 0;
        }

        @Override
        public ContentItem getItem(int i) {
            return adapterController != null ? adapterController.getBaseItemList(i) : null;
        }

        @Override
        public long getItemId(int i) {
            return i;
        }

        @Override
        public View getView(int i, View view, ViewGroup viewGroup) {
            if (view == null) {
                view = LayoutInflater.from(viewGroup.getContext())
                        .inflate(R.layout.my_ais_account_widget_my_number_view_item, viewGroup, false);
            }

            ContentItem contentItem = getItem(i);
            contentItem.setSelected(contentItem.getMobileNumber().equals(mobileNumberSelected));

            FrameLayout frameLayoutContainer = (FrameLayout) view.findViewById(R.id.my_ais_account_widget_my_number_view_item_container);
            final Button btnSelectionIndicator = (Button) view.findViewById(R.id.my_ais_account_widget_my_number_view_btn_selection_indicator);
            TextView tvMobileNumber = (TextView) view.findViewById(R.id.my_ais_account_widget_my_number_view_tv);
            tvMobileNumber.setText(NextzyUtility.getInstance().applyAisNumberPattern(contentItem.getMobileNumber()));
            btnSelectionIndicator.setSelected(contentItem.isSelected());
            view.setOnClickListener(view1 -> {
                onMyMobileNumberItemClick(i);
                btnSelectionIndicator.setSelected(true);
            });
            btnSelectionIndicator.setOnClickListener(view1 -> {
                onMyMobileNumberItemClick(i);
                btnSelectionIndicator.setSelected(true);
            });

            return view;
        }

        private void onMyMobileNumberItemClick(int position) {
            if (!adapter.getItem(position).isSelected()) {
                if (onMyNumberClickListener != null) {
                    onMyNumberClickListener.onMyNumberSelected(position);
                }
                if (myNumberListener != null) {
                    ContentItem contentItem = adapterController.getContentItem(position);
                    String mobileNumber = contentItem.getMobileNumber();
                    myNumberListener.onMyNumberSelected(FormatUtility.getInstance().getStringWithFormat(mobileNumber, FORMAT_MOBILE_NUMBER), position);
                }
                notifyDataSetChanged();
            } else {
                onDismiss();
            }

        }
    }

    public class AdapterController implements OnMyNumberClickListener {
        List<ContentItem> contentItemList;

        public AdapterController() {
            this.contentItemList = new ArrayList<>();
        }

        public void setContentItemList(List<ContentItem> contentItemList) {
            this.contentItemList = contentItemList;
        }

        public List<ContentItem> getContentItemList() {
            return contentItemList;
        }

        private void clearAllContentItemSelection() {
            if (contentItemList != null) {
                for (ContentItem contentItem : contentItemList) {
                    contentItem.setSelected(false);
                }
            }
        }

        private boolean setContentItemSelected(int position) {
            if (contentItemList != null && contentItemList.size() > position) {
                ContentItem contentItem = contentItemList.get(position);
                contentItem.setSelected(true);
                return true;
            }
            return false;
        }

        @Override
        public void onMyNumberSelected(int position) {
            clearAllContentItemSelection();
            setContentItemSelected(position);
        }

        public ContentItem getBaseItemList(int position) {
            if (contentItemList != null && contentItemList.size() > position) {
                return contentItemList.get(position);
            }
            return null;
        }

        public ContentItem getContentItem(int position) {
            if (contentItemList != null && contentItemList.size() > position) {
                return contentItemList.get(position);
            }
            return null;
        }

        public int getAllItemCount() {
            return contentItemList != null ? contentItemList.size() : 0;
        }
    }

    public static class ContentItem implements Parcelable {
        private boolean isSelected;
        private String mobileNumber;

        public ContentItem(String mobileNumber) {
            this.isSelected = false;
            this.mobileNumber = mobileNumber;
        }

        public boolean isSelected() {
            return isSelected;
        }

        public void setSelected(boolean selected) {
            isSelected = selected;
        }

        public String getMobileNumber() {
            return mobileNumber;
        }

        public void setMobileNumber(String mobileNumber) {
            this.mobileNumber = mobileNumber;
        }

        @Override
        public int describeContents() {
            return 0;
        }

        @Override
        public void writeToParcel(Parcel dest, int flags) {
            dest.writeByte(this.isSelected ? (byte) 1 : (byte) 0);
            dest.writeString(this.mobileNumber);
        }

        public ContentItem() {
        }

        protected ContentItem(Parcel in) {
            this.isSelected = in.readByte() != 0;
            this.mobileNumber = in.readString();
        }

        public static final Parcelable.Creator<ContentItem> CREATOR = new Parcelable.Creator<ContentItem>() {
            @Override
            public ContentItem createFromParcel(Parcel source) {
                return new ContentItem(source);
            }

            @Override
            public ContentItem[] newArray(int size) {
                return new ContentItem[size];
            }
        };
    }

    private interface OnMyNumberClickListener {
        void onMyNumberSelected(int position);
    }

    public interface MyNumberListener {
        void onMyNumberSelected(String mobileNumber, int position);

        void onDismiss();
    }
}
