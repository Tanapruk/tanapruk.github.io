package com.ais.mimo.eservice.widget;

import android.content.Context;
import android.content.res.TypedArray;
import android.os.Build;
import android.os.Parcel;
import android.os.Parcelable;
import android.support.annotation.RequiresApi;
import android.util.AttributeSet;
import android.widget.ImageView;
import android.widget.TextView;

import com.nextzy.myais.common.widget.NextzyViewGroup;
import com.nextzy.myais.myais.R;

/**
 * Created by Akexorcist on 11/21/2016 AD.
 */

public class ProfileUserDetailView extends NextzyViewGroup {
    private ImageView ivIcon;
    private TextView tvText;

    private int iconResId;
    private String text;

    public ProfileUserDetailView(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    public ProfileUserDetailView(Context context, AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
    }

    @RequiresApi(api = Build.VERSION_CODES.LOLLIPOP)
    public ProfileUserDetailView(Context context, AttributeSet attrs, int defStyleAttr, int defStyleRes) {
        super(context, attrs, defStyleAttr, defStyleRes);
    }

    @Override
    protected int getLayoutRes() {
        return R.layout.my_ais_profile_view_user_detail;
    }

    @Override
    protected void setupStyleables(AttributeSet attrs, int defStyleAttr, int defStyleRes) {
        TypedArray typedArray = getContext().obtainStyledAttributes(attrs, R.styleable.ProfileUserDetailView);
        text = typedArray.getString(R.styleable.ProfileUserDetailView_pud_text);
        iconResId = typedArray.getResourceId(R.styleable.ProfileUserDetailView_pud_icon, -1);
        typedArray.recycle();
    }

    @Override
    protected void bindView() {
        ivIcon = (ImageView) findViewById(R.id.my_ais_profile_iv_icon);
        tvText = (TextView) findViewById(R.id.my_ais_profile_tv_text);
    }

    @Override
    protected void setupView() {
        updateView();
    }

    public void setText(String text) {
        this.text = text;
        updateText(text);
    }

    public void setIcon(int resId) {
        this.iconResId = resId;
        updateIcon(resId);
    }

    private void updateText(String text) {
        tvText.setText(text);
    }

    private void updateIcon(int resId) {
        if (resId != -1) {
            ivIcon.setImageResource(resId);
        }
    }

    private void updateView() {
        updateText(text);
        updateIcon(iconResId);
    }

    @Override
    protected Parcelable onSaveInstanceState() {
        Parcelable superState = super.onSaveInstanceState();
        SavedState ss = (SavedState) onSaveChildInstanceState(new SavedState(superState));
        ss.text = this.text;
        ss.iconResId = this.iconResId;
        return ss;
    }

    @Override
    protected void onRestoreInstanceState(Parcelable state) {
        if (!(state instanceof SavedState)) {
            super.onRestoreInstanceState(state);
            return;
        }
        SavedState ss = (SavedState) state;
        super.onRestoreInstanceState(ss.getSuperState());
        this.text = ss.text;
        this.iconResId = ss.iconResId;
        updateView();
    }

    private static class SavedState extends ChildSavedState {
        String text;
        int iconResId;

        SavedState(Parcelable superState) {
            super(superState);
        }

        private SavedState(Parcel in, ClassLoader classLoader) {
            super(in, classLoader);
            this.text = in.readString();
            this.iconResId = in.readInt();
        }

        @Override
        public void writeToParcel(Parcel out, int flags) {
            super.writeToParcel(out, flags);
            out.writeString(this.text);
            out.writeInt(this.iconResId);
        }

        public static final ClassLoaderCreator<SavedState> CREATOR = new ClassLoaderCreator<SavedState>() {
            @Override
            public SavedState createFromParcel(Parcel source, ClassLoader loader) {
                return new SavedState(source, loader);
            }

            public SavedState createFromParcel(Parcel in) {
                return null;
            }


            public SavedState[] newArray(int size) {
                return new SavedState[size];
            }
        };
    }
}
