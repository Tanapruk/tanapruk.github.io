package com.ais.mimo.eservice.widget;

import android.content.Context;
import android.os.Parcel;
import android.os.Parcelable;
import android.util.AttributeSet;
import android.view.View;
import android.widget.TextView;

import com.nextzy.myais.common.utility.NextzyLanguage;
import com.nextzy.myais.common.widget.NextzyViewGroup;
import com.nextzy.myais.myais.R;

/**
 * Created by trusttanapruk on 10/14/2016.
 */

public class LoginLanguageBar extends NextzyViewGroup implements View.OnClickListener {
    private TextView tvSwitchLanguage;
    private LoginLanguageBarListener listener;
    private boolean isThai;


    public LoginLanguageBar(Context context) {
        super(context);
    }

    public LoginLanguageBar(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    @Override
    protected int getLayoutRes() {
        return R.layout.my_ais_login_language_bar;
    }

    @Override
    protected void setupStyleables(AttributeSet attrs, int defStyleAttr, int defStyleRes) {

    }

    @Override
    protected void bindView() {
        tvSwitchLanguage = (TextView) findViewById(R.id.my_ais_login_language_bar_tv_switch_language);
    }

    @Override
    protected void setupView() {
        tvSwitchLanguage.setOnClickListener(this);
        setCurrentLanguage();
        setLanguageSwitch(isThai);
    }

    private void setCurrentLanguage() {
        isThai = NextzyLanguage.getInstance().isThai();
    }

    public void setLoginLanguageBarListener(LoginLanguageBarListener listener) {
        this.listener = listener;
    }


    @Override
    public void onClick(View view) {
        if (view == tvSwitchLanguage) {
            toggleLanguage();
        }
    }

    private void toggleLanguage() {
        if (listener != null) {
            listener.setLanguage(isThai ? NextzyLanguage.ENGLISH : NextzyLanguage.THAI);
        }
        setLanguageSwitch(isThai);
        isThai = !isThai;
    }

    private void setLanguageSwitch(boolean isThai) {
        tvSwitchLanguage.setText(isThai ? String.valueOf(NextzyLanguage.ENGLISH).toUpperCase() : String.valueOf(NextzyLanguage.THAI).toUpperCase());
    }

    private static final String KEY_ISTHAI = "ISTHAI";

    @Override
    protected Parcelable onSaveInstanceState() {
        Parcelable superState = super.onSaveInstanceState();
        SavedState ss = (SavedState) onSaveChildInstanceState(new SavedState(superState));
        ss.isThai = this.isThai;
        return ss;
    }

    @Override
    protected void onRestoreInstanceState(Parcelable state) {
        if (!(state instanceof SavedState)) {
            super.onRestoreInstanceState(state);
            return;
        }
        SavedState ss = (SavedState) state;
        super.onRestoreInstanceState(ss.getSuperState());
        this.isThai = ss.isThai;
        setLanguageSwitch(isThai);
    }

    private static class SavedState extends ChildSavedState {
        boolean isThai;

        SavedState(Parcelable superState) {
            super(superState);
        }

        private SavedState(Parcel in, ClassLoader classLoader) {
            super(in, classLoader);
            this.isThai = (in.readInt() == 1);
        }

        @Override
        public void writeToParcel(Parcel out, int flags) {
            super.writeToParcel(out, flags);
            out.writeInt((this.isThai) ? 1 : 0);
        }

        public static final ClassLoaderCreator<SavedState> CREATOR = new ClassLoaderCreator<SavedState>() {
            @Override
            public SavedState createFromParcel(Parcel source, ClassLoader loader) {
                return new SavedState(source, loader);
            }

            public SavedState createFromParcel(Parcel in) {
                return null;
            }


            public SavedState[] newArray(int size) {
                return new SavedState[size];
            }
        };
    }
}
