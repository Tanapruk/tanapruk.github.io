package com.ais.mimo.eservice.module.setting.fragment.main;

import com.ais.mimo.eservice.helper.pushnotification.network.NotificationManager;
import com.ais.mimo.eservice.module.setting.event.GoToFirstNewPasscodeFragmentEvent;
import com.ais.mimo.eservice.module.setting.event.GoToPasscodeSettingFragmentEvent;
import com.ais.mimo.eservice.module.setting.event.SetLanguageEvent;
import com.hwangjr.rxbus.RxBus;
import com.nextzy.myais.common.mvp.NextzyPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class MainSettingPresenter extends NextzyPresenter<MainSettingContractor.View> implements MainSettingContractor.Presenter{


    protected MainSettingPresenter( MainSettingContractor.View view ){
        super( view );
    }

    public static MainSettingPresenter createPresenter( MainSettingContractor.View view ){
        return new MainSettingPresenter( view );
    }

    @Override
    public void onViewCreate(){
        RxBus.get().register( this );
    }

    @Override
    public void onViewDestroy(){
        RxBus.get().unregister( this );
    }

    @Override
    public void onViewStart(){
        // Do nothing.
    }

    @Override
    public void onViewStop(){
        // Do nothing.
    }

    /*******************/
    /** For view call **/
    /*******************/
    //<editor-fold desc="For view folding">
    @Override
    public void pushSubscriber(boolean isChecked) {
        getView().showLoadingDialog();
        if (isChecked) {
            NotificationManager.getInstance().pushNotificationSubscribers();
        } else {
            NotificationManager.getInstance().pushNotificationUnsubscribers();
        }
    }
    @Override
    public void setLanguage(String language){
        RxBus.get().post(new SetLanguageEvent(language));
    }

    @Override
    public void openPasscodeSettingFragment(){
        RxBus.get().post(new GoToPasscodeSettingFragmentEvent());
    }

    @Override
    public void openFirstNewPasscodeActivity() {
        RxBus.get().post(new GoToFirstNewPasscodeFragmentEvent());
    }
    //</editor-fold>

}
