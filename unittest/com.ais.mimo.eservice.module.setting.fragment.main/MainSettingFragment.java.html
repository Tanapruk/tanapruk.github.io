    private void setupPasscodeSettingView() {
        if (fingerManager.isFingerprintSupported(getContext())
                && fingerManager.isFingerprintEnrolled(getContext())) {
            msPasscode.setText(R.string.my_ais_setting_passcode_and_touch_id);
        } else {
            msPasscode.setText(R.string.my_ais_setting_passcode);
        }
    }

        if (getActivity() instanceof SettingActivity) {
            ((SettingActivity) getActivity()).setToolbarTitle(getString(R.string.my_ais_setting_title));
        }
    }

        SharedPreferences sharedPreferences = PreferenceManager.getDefaultSharedPreferences(getContext());

    }

    @Override
    public void onActivityResult(int requestCode, int resultCode, Intent data) {
        switch (requestCode) {
            case REQUEST_PASSCODE:
                if (resultCode == RESULT_MATCH_PASSCODE) {
                    getPresenter().openPasscodeSettingFragment();
                }
                break;
        }

    @Override
    public void onClick(View view) {
        if (view == msAccount) {
            onAccountClick();
        } else if (view == msFavourite) {
            onFavouriteClick();
        } else if (view == msPasscode) {
            onPasscodeSettingClick();
        } else if (view == btnTermCondition) {
            onTermConditionClick();
        }
    }


    @Override
    public void onCheckedChanged(View view, boolean isChecked) {
        if (view == msLanguage) {
            onLanguageChanged(isChecked);
        } else if (view == msNotification) {
            onNotificationChanged(isChecked);
        }
    }

    private void onLanguageChanged(boolean isChecked) {
        if (isChecked) {
            getPresenter().setLanguage(NextzyLanguage.THAI);
        } else {
            getPresenter().setLanguage(NextzyLanguage.ENGLISH);
        }
    }

    private void onPasscodeSettingClick() {
        if (preference.loadPasscodeSwitchSetting()) {
            Map<String, String> result = preference.loadPasscodeMasterNumberAndPrivateIdAndToken();
            if (isNotEmptyAll(result)) {
                Bundle bundle = new Bundle();
                bundle.putString(KEY_MASTER_NUMBER, result.get(KEY_MASTER_NUMBER));
                bundle.putString(KEY_PASSCODE_FOR_VALIDATE, result.get(KEY_PASSCODE));
                bundle.putString(KEY_PRIVATE_ID, result.get(KEY_PRIVATE_ID));
                bundle.putString(KEY_TOKEN, result.get(KEY_TOKEN));
                openActivityOnResult(ValidatePasscodeActivity.class, REQUEST_PASSCODE, bundle);
            } else if (isPasscodeEmpty(result)) {
                getPresenter().openFirstNewPasscodeActivity();
            } else {
                getPresenter().openPasscodeSettingFragment();
            }
        } else {
            getPresenter().openPasscodeSettingFragment();
        }
    }

    private boolean isPasscodeEmpty(Map<String, String> result) {
        return result != null && result.get(KEY_PASSCODE) != null;
    }

    private boolean isNotEmptyAll(Map<String, String> result) {
        return result != null
                && result.get(KEY_MASTER_NUMBER) != null
                && result.get(KEY_PASSCODE) != null
                && result.get(KEY_PRIVATE_ID) != null
                && result.get(KEY_TOKEN) != null;
    }
