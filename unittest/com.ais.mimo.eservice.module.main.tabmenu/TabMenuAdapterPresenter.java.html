package com.ais.mimo.eservice.module.main.tabmenu;

import com.nextzy.myais.common.menu.MyAisMenu;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.common.utility.NextzyLanguage;
import com.nextzy.myais.common.utility.NextzyUtility;

import java.util.List;

/**
 * Created by nextzy on 8/23/2016 AD.
 */
public class TabMenuAdapterPresenter extends NextzyPresenter<TabMenuAdapterContractor.View> implements TabMenuAdapterContractor.Presenter {
    private List<MyAisMenu.Menu> menuList;

    public static TabMenuAdapterPresenter createPresenter(TabMenuAdapterContractor.View view) {
        return new TabMenuAdapterPresenter(view);
    }

    public TabMenuAdapterPresenter(TabMenuAdapterContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {

    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {

    }

    @Override
    public String getPageTitle(int position) {
        MyAisMenu.Menu menu = menuList.get(position);
        return NextzyUtility.getInstance().getLocalMessage(menu.getNameEn(), menu.getNameTh(), NextzyLanguage.getInstance().getCurrentLanguage());
    }

    @Override
    public void setMenuList(List<MyAisMenu.Menu> menuList) {
        this.menuList = menuList;
    }

    @Override
    public List<MyAisMenu.Menu> getMenuList() {
        return menuList;
    }

    @Override
    public int getMenuSize() {
        return menuList == null ? 0 : menuList.size();
    }

    @Override
    public MyAisMenu.Menu getMenu(int position) {
        return menuList != null ? menuList.get(position) : null;
    }

    @Override
    public int getMenuIndex(String key) {
        if (menuList != null) {
            for (int index = 0; index < menuList.size(); index++) {
                if (menuList.get(index).getKey().equals(key)) {
                    return index;
                }
            }
        }
        return -1;
    }
}
