package com.ais.mimo.eservice.module.main.tabmenu;

import android.support.v4.app.Fragment;
import android.support.v4.app.FragmentManager;
import android.util.SparseArray;

import com.ais.mimo.eservice.config.menu.tab.TabMenuConfig;
import com.akexorcist.localizationactivity.LanguageSetting;
import com.nextzy.myais.common.menu.MyAisMenu;
import com.nextzy.myais.common.mvp.NextzyPagerAdapter;

import java.util.List;

/**
 * Created by nextzy on 8/23/2016 AD.
 */
public class TabMenuAdapter extends NextzyPagerAdapter<TabMenuAdapterContractor.Presenter> implements TabMenuAdapterContractor.View {

    public TabMenuAdapter(FragmentManager fm) {
        super(fm);
    }

    /**
     * This hold all instatiated fragments so we can reuse them
     */
    private SparseArray<Fragment> mSparseArray = new SparseArray<>();
    @Override
    public void createPresenter() {
        TabMenuAdapterPresenter.createPresenter(this);
    }


    @Override
    public Fragment getItem(int position) {
        String key = getMenu(position).getKey();
        final Fragment fragment = TabMenuConfig.getInstance().getFragment(key);
        mSparseArray.append(position, fragment);
        return fragment;
    }

    public Fragment getFragment(int position) {
        return mSparseArray.get(position);
    }

    public MyAisMenu.Menu getMenu(int position) {
        return getPresenter().getMenu(position);
    }

    @Override
    public String getCurrentLanguage() {
        return LanguageSetting.getLanguage();
    }

    @Override
    public int getCount() {
        return getPresenter().getMenuSize();
    }

    @Override
    public CharSequence getPageTitle(int position) {
        return getPresenter().getPageTitle(position);
    }

    public void setMenuList(List<MyAisMenu.Menu> menuList) {
        getPresenter().setMenuList(menuList);
        notifyDataSetChanged();
    }

    public int getTabIndexByKey(String key) {
        return getPresenter().getMenuIndex(key);
    }
}
