package com.ais.mimo.eservice.helper.login.network.response;

import com.google.gson.annotations.SerializedName;
import com.nextzy.myais.common.network.model.GenericResult;

/**
 * Created by Akexorcist on 2/17/2016 AD.
 */
public class AppSettingsResult extends GenericResult {
    String mobileNumber;
    String apiKey;
    @SerializedName("lang")
    String language;
    @SerializedName("auditLogID")
    String auditLogId;
    String nType;

    public void setApiKey(String apiKey) {
        this.apiKey = apiKey;
    }

    public String getMobileNumber() {
        return mobileNumber;
    }

    public String getApiKey() {
        return apiKey;
    }

    public String getLanguage() {
        return language;
    }

    public String getAuditLogId() {
        return auditLogId;
    }

    public String getnType() {
        return nType;
    }

    public void setMobileNumber(String mobileNumber) {
        this.mobileNumber = mobileNumber;
    }

    public void setLanguage(String language) {
        this.language = language;
    }

    public void setAuditLogId(String auditLogId) {
        this.auditLogId = auditLogId;
    }

    public void setnType(String nType) {
        this.nType = nType;
    }
}
