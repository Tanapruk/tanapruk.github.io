package com.ais.mimo.eservice.helper.login.network.response;

import com.nextzy.myais.common.network.model.GenericData;

/**
 * Created by trusttanapruk on 8/11/2016.
 */
public class ISwopData extends GenericData {
    public boolean isInternetAvailable() {
        return getInternetRemaining() != null;
    }

    public boolean isVoiceAvailable() {
        return getVoiceRemaining() != null;
    }

    public boolean hasISwop() {
        return isInternetAvailable() && isVoiceAvailable();
    }
    Remaining internetRemaining;
    Remaining voiceRemaining;
    boolean isSuspended;

    public void setInternetRemaining(Remaining internetRemaining) {
        this.internetRemaining = internetRemaining;
    }

    public void setVoiceRemaining(Remaining voiceRemaining) {
        this.voiceRemaining = voiceRemaining;
    }

    public void setSuspended(boolean suspended) {
        isSuspended = suspended;
    }

    public Remaining getInternetRemaining() {
        return internetRemaining;
    }

    public Remaining getVoiceRemaining() {
        return voiceRemaining;
    }

    public boolean isSuspended() {
        return isSuspended;
    }

    public static class Remaining {
        String remaining;
        String unit;

        public void setRemaining(String remaining) {
            this.remaining = remaining;
        }

        public void setUnit(String unit) {
            this.unit = unit;
        }

        public String getRemaining() {
            return remaining;
        }

        public String getUnit() {
            return unit;
        }
    }
}
