package com.ais.mimo.eservice.helper.login.network.response;

import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.config.mobile.MobileContact;
import com.nextzy.myais.common.config.mobile.MobileType;
import com.nextzy.myais.common.config.mobile.MobileUtility;
import com.nextzy.myais.common.network.model.GenericResult;

import java.util.ArrayList;
import java.util.List;

public class VerifyMyAISEditAccountResult extends GenericResult<VerifyMyAISData> {

    private boolean isMobileNumber() {
        if (Mobile.getInstance().getMobileNumber() == null) {
            throw new NullPointerException("You must logged in before coming to this method ");
        }
        return Mobile.getInstance().getMobileNumber().startsWith("0");
    }

    private boolean isFibre() {
        if (Mobile.getInstance().getMobileNumber() == null) {
            throw new NullPointerException("You must logged in before coming to this method ");
        }
        return Mobile.getInstance().getMobileNumber().startsWith("88");
    }

    private boolean isMyAis(String yOrN) {
        return hasData() && yOrN.equalsIgnoreCase(getData().getIsMyAIS());
    }

    private boolean isMaster(String yOrN) {
        return hasData() && yOrN.equalsIgnoreCase(getData().getIsMaster());
    }

    private boolean isMyAisYes() {
        return isMyAis("Y");
    }

    private boolean isMyAisNo() {
        return isMyAis("N");
    }

    private boolean isMasterYes() {
        return isMaster("Y");
    }

    private boolean isMasterNo() {
        return isMaster("N");
    }

    public boolean shouldShowVerifyChoice() {
        return isMobileNumber() && isMyAisYes() && isMasterYes();
    }

    public boolean shouldGotoPassword() {
        return isFibre() && isMyAisYes() && isMasterYes();
    }

    public boolean isFibreButCannotLogin(){
        return isFibre() && isMyAisYes() && isMasterNo();
    }

    public boolean shouldGotoOtp() {
        return isMobileNumber() && (!isMyAisYes() || !isMasterYes());
    }

    public boolean shouldGotoRegister() {
        return isFibre() && isMyAisNo() && isMasterNo();
    }

    public boolean shouldAlertToLoginWithMaster() {
        return isFibre() && isMyAisYes() && isMasterNo();
    }

    public boolean isCanRequestVerifyIdentity(){
        return isMasterNo() && isMyAisNo();
    }

    public List<MobileContact> toMobileContactList() {
        if (data != null && data.getDataList() != null && !data.getDataList().isEmpty()) {
            List<MobileContact> mobileContacts = new ArrayList<>();
            for (MobileData mobileData : data.getDataList()) {
                if (!"active".equalsIgnoreCase(mobileData.getStatusCode())) {
                    continue;
                }

                MobileType mobileType = MobileUtility.getInstance().getMobileTypeByMobileData(mobileData.getChargeType(), mobileData.getMobileType());
                mobileContacts.add(new MobileContact(mobileData.getNumber(), mobileType));
            }
            return mobileContacts;
        }
        return null;
    }

}
