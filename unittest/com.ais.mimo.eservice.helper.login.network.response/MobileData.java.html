package com.ais.mimo.eservice.helper.login.network.response;

import android.os.Parcel;
import android.os.Parcelable;

import com.nextzy.myais.common.network.model.GenericData;

import lombok.EqualsAndHashCode;

/**
 * Created by Oat on 10/13/2016 AD.
 */

@EqualsAndHashCode(callSuper = false, exclude = {"masterFlg", "chargeType", "mobileType", "statusCode"})
public class MobileData extends GenericData implements Parcelable {
    public static final String CHARGE_TYPE_PREPAID = "Pre-Paid";
    public static final String CHARGE_TYPE_POSTPAID = "Post-Paid";

    private String number;
    private String masterFlg;
    private String chargeType;
    private String mobileType;
    private String statusCode;
    private String emailMyAis;

    public String getNumber() {
        return number;
    }

    public void setNumber(String number) {
        this.number = number;
    }

    public String getMasterFlg() {
        return masterFlg;
    }

    public void setMasterFlg(String masterFlg) {
        this.masterFlg = masterFlg;
    }

    public String getChargeType() {
        return chargeType;
    }

    public void setChargeType(String chargeType) {
        this.chargeType = chargeType;
    }

    public String getMobileType() {
        return mobileType;
    }

    public void setMobileType(String mobileType) {
        this.mobileType = mobileType;
    }

    public String getEmailMyAis() {
        return emailMyAis;
    }

    public void setEmailMyAis(String emailMyAis) {
        this.emailMyAis = emailMyAis;
    }

    public String getStatusCode() {
        return statusCode;
    }

    public void setStatusCode(String statusCode) {
        this.statusCode = statusCode;
    }

    public boolean isMember(String mobileNumber) {
        if (mobileNumber.equals(number)) {
            return true;
        }
        return false;
    }

    public MobileData() {
    }

    @Override
    public int describeContents() {
        return 0;
    }

    @Override
    public void writeToParcel(Parcel dest, int flags) {
        dest.writeString(this.number);
        dest.writeString(this.masterFlg);
        dest.writeString(this.chargeType);
        dest.writeString(this.mobileType);
        dest.writeString(this.statusCode);
        dest.writeString(this.emailMyAis);
    }

    protected MobileData(Parcel in) {
        this.number = in.readString();
        this.masterFlg = in.readString();
        this.chargeType = in.readString();
        this.mobileType = in.readString();
        this.statusCode = in.readString();
        this.emailMyAis = in.readString();
    }

    public static final Parcelable.Creator<MobileData> CREATOR = new Parcelable.Creator<MobileData>() {
        @Override
        public MobileData createFromParcel(Parcel source) {
            return new MobileData(source);
        }

        @Override
        public MobileData[] newArray(int size) {
            return new MobileData[size];
        }
    };
}
