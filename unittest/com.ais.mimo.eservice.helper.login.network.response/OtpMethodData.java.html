package com.ais.mimo.eservice.helper.login.network.response;

import com.nextzy.myais.common.network.model.GenericData;

 * Created by trusttanapruk on 8/16/2016.
 */
public class OtpMethodData extends GenericData {
    boolean primaryMobile;
    boolean secondaryMobile;
    boolean email;
    String redirectUrl;

    public boolean isPrimaryMobile() {
        return primaryMobile;
    }

    public void setPrimaryMobile(boolean primaryMobile) {
        this.primaryMobile = primaryMobile;
    }

    public boolean isSecondaryMobile() {
        return secondaryMobile;
    }

    public void setSecondaryMobile(boolean secondaryMobile) {
        this.secondaryMobile = secondaryMobile;
    }

    public boolean isEmail() {
        return email;
    }

    public void setEmail(boolean email) {
        this.email = email;
    }

    public String getRedirectUrl() {
        return redirectUrl;
    }

    public void setRedirectUrl(String redirectUrl) {
        this.redirectUrl = redirectUrl;
    }

}
