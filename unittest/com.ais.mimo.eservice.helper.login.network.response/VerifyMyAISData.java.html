package com.ais.mimo.eservice.helper.login.network.response;

import com.nextzy.myais.common.network.model.GenericData;

import java.util.List;

/**
 * Created by Oat on 10/13/2016 AD.
 * Update Change Master by Royle on 02/03/2017.
 */

public class VerifyMyAISData extends GenericData {

    String isMyAIS;
    String isMaster;
    List<MobileData> dataList;

    public String getIsMyAIS() {
        return isMyAIS;
    }

    public boolean getIsMyAisBoolean() {
        return "Y".equalsIgnoreCase(isMyAIS);
    }

    public boolean getIsMyMasterBoolean() {
        return "Y".equalsIgnoreCase(isMaster);
    }

    public void setIsMyAIS(String isMyAIS) {
        this.isMyAIS = isMyAIS;
    }

    public String getIsMaster() {
        return isMaster;
    }

    public void setIsMaster(String isMaster) {
        this.isMaster = isMaster;
    }

    public List<MobileData> getDataList() {
        return dataList;
    }

    public void setDataList(List<MobileData> dataList) {
        this.dataList = dataList;
    }

    public boolean isMember(String mobileNumber) {
        for (MobileData mobileData : getDataList()) {
            if (mobileData.isMember(mobileNumber)) {
                return true;
            }
        }
        return false;
    }
}
