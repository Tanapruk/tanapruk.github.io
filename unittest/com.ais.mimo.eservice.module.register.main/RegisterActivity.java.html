package com.ais.mimo.eservice.module.register.main;

import android.os.Bundle;
import android.support.v4.app.Fragment;
import android.view.View;
import android.widget.LinearLayout;

import com.ais.mimo.eservice.module.register.event.data.SelectPostpaidEventData;
import com.ais.mimo.eservice.module.register.event.data.Step2EventData;
import com.ais.mimo.eservice.module.register.event.data.Step3EventData;
import com.ais.mimo.eservice.module.register.step1.RegisterStep1Fragment;
import com.ais.mimo.eservice.module.register.step2.RegisterStep2Fragment;
import com.ais.mimo.eservice.module.register.step2.changeMaster.RegisterChangeMainNumberFragment;
import com.ais.mimo.eservice.module.register.step3.RegisterStep3Fragment;
import com.ais.mimo.eservice.module.register.step4.RegisterStep4Fragment;
import com.ais.mimo.eservice.module.register.welcome.RegisterWelcomeFragment;
import com.akexorcist.library.flexystepindicator.FlexyStepIndicator;
import com.nextzy.myais.common.delegate.FragmentHelperDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.NextzyAndroidUtility;
import com.nextzy.myais.common.widget.toolbar.simple.MyAisSimpleToolbar;
import com.nextzy.myais.myais.R;

import java.util.Arrays;
import java.util.List;

import static com.ais.mimo.eservice.module.register.base.RegisterBaseFragment.KEY_EMAIL;
import static com.ais.mimo.eservice.module.register.base.RegisterBaseFragment.KEY_ID_CARD;
import static com.ais.mimo.eservice.module.register.base.RegisterBaseFragment.KEY_MOBILE_NUMBER;
import static com.ais.mimo.eservice.module.register.base.RegisterBaseFragment.KEY_PASS;
import static com.ais.mimo.eservice.module.register.base.RegisterBaseFragment.KEY_REPASS;
import static com.ais.mimo.eservice.module.register.step2.RegisterStep2Fragment.KEY_FBB_GROUP;
import static com.ais.mimo.eservice.module.register.step2.RegisterStep2Fragment.KEY_POSTPAID_GROUP;

public class RegisterActivity extends NextzyMvpActivity<RegisterContractor.Presenter> implements RegisterContractor.View {
    private final static String TAG = RegisterActivity.class.getSimpleName();
    public static final String BUNDLE_DATA = "bundle_data";
    private MyAisSimpleToolbar tbHeader;
    private FlexyStepIndicator indicator;
    private LinearLayout background;

    private FragmentHelperDelegate fragmentHelperDelegate = new FragmentHelperDelegate(this);
    private Bundle bundle;

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_register;
    }

    @Override
    protected void createPresenter() {
        RegisterPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        fragmentHelperDelegate.bind(R.id.contentContainer);
        tbHeader = (MyAisSimpleToolbar) findViewById(R.id.my_ais_library_simple_toolbar_tb_header);
        indicator = (FlexyStepIndicator) findViewById(R.id.my_ais_regis_indicator);
        background = (LinearLayout) findViewById(R.id.my_ais_regis_background_layout);
    }

    @Override
    public void setupView() {
        tbHeader.setTitle(R.string.my_ais_register);
        tbHeader.setNavigationOnClickListener(v -> {
            onBackPressed();
        });
        setupIndicator(indicator);
    }


    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        getSupportFragmentManager().beginTransaction()
                .add(R.id.contentContainer,
                        RegisterStep1Fragment.newInstance(getIntent().getExtras()),
                        RegisterStep1Fragment.TAG)
                .commit();
        bundle = new Bundle();
    }

    @Override
    public void restoreView() {
    }

    @Override
    public void saveInstanceState(Bundle outState) {
        outState.putBundle(BUNDLE_DATA, bundle);
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        bundle = savedInstanceState.getBundle(BUNDLE_DATA);
    }
    //</editor-fold>

    private void setupIndicator(FlexyStepIndicator indicator) {
        if (indicator != null) {
            List<String> descList = Arrays.asList(null, null, null, null);

            indicator.setStepDescriptionList(descList);
            indicator.setStepClickable(false);
            setupClickHideKeyboard(indicator);
        }

    }

    public void setIndexIndicator(int index) {
        if (indicator != null) {
            indicator.setVisibility(View.VISIBLE);
            indicator.setCurrentIndex(index);
        }
    }

    protected void setupClickHideKeyboard(View v) {
        if (v != null) {
            v.setOnClickListener(
                    view -> hindKeyboard());
        }
    }

    public void hindIndicator() {
        if (indicator != null) {
            indicator.setVisibility(View.GONE);
        }
    }

    public void setBackground(int colorId) {
        background.setBackgroundResource(colorId);
    }

    private void hindKeyboard() {
        NextzyAndroidUtility.getInstance().hideKeyboard(this);
    }

    @Override
    public void onBackPressed() {
        Fragment fragment = getSupportFragmentManager()
                .findFragmentByTag(RegisterWelcomeFragment.TAG);
        if (fragment instanceof RegisterWelcomeFragment && fragment.isVisible()) {
            finish();
        } else if (getSupportFragmentManager().getBackStackEntryCount() > 0) {
            fragmentHelperDelegate.clearAllBackStacks();
        } else {
            super.onBackPressed();
        }
    }


    @Override
    public void openFragmentStep2(Step2EventData data) {
        bundle.putString(KEY_MOBILE_NUMBER, data.getMobileNumber());
        bundle.putString(KEY_ID_CARD, data.getIdCard());
        bundle.putString(KEY_EMAIL, data.getEmail());
        fragmentHelperDelegate.openFragmentSlideInOut(
                RegisterStep2Fragment.newInstance(bundle),
                null);
    }

    @Override
    public void openFragmentSelectPostMobileNumber(SelectPostpaidEventData data) {
        bundle.putString(KEY_PASS, data.getPassword());
        bundle.putString(KEY_REPASS, data.getRePassword());
        bundle.putParcelableArrayList(KEY_POSTPAID_GROUP, data.getPostpaidGroup());
        bundle.putParcelableArrayList(KEY_FBB_GROUP, data.getFbbGroup());
        fragmentHelperDelegate.openFragmentSlideOutOnly(
                RegisterChangeMainNumberFragment.newInstance(bundle),
                null);

    }

    @Override
    public void openFragmentStep3WithData(Step3EventData data) {
        bundle.putString(KEY_PASS, data.getPassword());
        bundle.putString(KEY_REPASS, data.getRePassword());
        fragmentHelperDelegate.openFragmentSlideInOut(
                RegisterStep3Fragment.newInstance(bundle),
                null);
    }

    @Override
    public void openFragmentStep3NoData() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                RegisterStep3Fragment.newInstance(bundle),
                null);
    }

    @Override
    public void openFragmentStep4() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                RegisterStep4Fragment.newInstance(bundle),
                null);
    }

    @Override
    public void openFragmentWelcome() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                RegisterWelcomeFragment.newInstance(bundle),
                null);
    }
}
