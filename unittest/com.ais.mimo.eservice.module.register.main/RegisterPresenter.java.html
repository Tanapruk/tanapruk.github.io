package com.ais.mimo.eservice.module.register.main;

import com.ais.mimo.eservice.module.register.event.GoToSelectMasterFragmentEvent;
import com.ais.mimo.eservice.module.register.event.GoToStep2FragmentEvent;
import com.ais.mimo.eservice.module.register.event.GoToStep3FragmentEvent;
import com.ais.mimo.eservice.module.register.event.GoToStep3FragmentNoDataEvent;
import com.ais.mimo.eservice.module.register.event.GoToStep4FragmentEvent;
import com.ais.mimo.eservice.module.register.event.GoToWelcomeFragmentEvent;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class RegisterPresenter extends NextzyActivityPresenter<RegisterContractor.View> implements RegisterContractor.Presenter {

    protected RegisterPresenter(RegisterContractor.View view ){
        super( view );
    }

    public static RegisterPresenter createPresenter(RegisterContractor.View view) {
        return new RegisterPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart(){
        // for handle network error
        super.onViewStart();
    }

    @Override
    public void onViewStop(){
        super.onViewStop();
    }


    @Subscribe
    public void onGoToFragmentStep2Event(GoToStep2FragmentEvent event) {
        getView().openFragmentStep2(event.getData());
    }

    @Subscribe
    public void onGoToFragmentStep3Event(GoToStep3FragmentEvent event) {
        getView().openFragmentStep3WithData(event.getData());
    }

    @Subscribe
    public void onGoToFragmentStep3NoDataEvent(GoToStep3FragmentNoDataEvent event) {
        getView().openFragmentStep3NoData();
    }

    @Subscribe
    public void onGoToFragmentSelectPostpaidEvent(GoToSelectMasterFragmentEvent event) {
        getView().openFragmentSelectPostMobileNumber(event.getData());
    }

    @Subscribe
    public void onGoToFragmentStep4NoDataEvent(GoToStep4FragmentEvent event) {
        getView().openFragmentStep4();
    }

    @Subscribe
    public void onGoToFragmentWelcome(GoToWelcomeFragmentEvent event) {
        getView().openFragmentWelcome();
    }
}
