package com.nextzy.myais.common.delegate;

import com.nextzy.myais.common.utility.NextzyUtility;

public class FormatUtility {
    public static final String SEPARATE = "-";
    public static final String FORMAT_MOBILE_NUMBER = "###-###-####";
    public static final String FORMAT_ID_CARD = "#-####-#####-##-#";

    private static FormatUtility instance;

    public static FormatUtility getInstance(){
        if( instance == null )
            instance = new FormatUtility();
        return instance;
    }



    /**
     * @deprecated use {@link NextzyUtility#applyStringPattern(String, String)} instead
     */
    @Deprecated
    public String getStringWithFormat(String text, String format){
        // ex. pattern "(\\d{3})(\\d{3})(\\d+)"
        String pattern = "";
        // ex. replacement "$1-$2-$3"
        String replacement = "";
        String[] formats = format.split(SEPARATE);
        for (int i = 0; i < formats.length; i++) {
            pattern += "(\\d{" + formats[i].length() + "})";
            if (i == 0) {
                replacement += "$" + (i + 1);
            } else {
                replacement += SEPARATE + "$" + (i + 1);
            }
        }
        text = text.replaceFirst(pattern, replacement);
        return text;
    }

    /**
     * @deprecated use {@link NextzyUtility#getTextNoDash(String)} instead
     */
    @Deprecated
    public String getTextNoDash( String text ){
        if( text.contains(SEPARATE)){
            return text.replace(SEPARATE, "");
        }
        return text;
    }

}
