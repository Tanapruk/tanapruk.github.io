public class CdrContentPresenter extends NextzyPresenter<CdrContentContractor.View> implements CdrContentContractor.Presenter {
    private List<BaseCdrItem> cdrItemList;

    public static CdrContentPresenter createPresenter(CdrContentContractor.View view) {
        return new CdrContentPresenter(view);
    }

    public List<BaseCdrItem> getCdrItemList() {
        return cdrItemList;
    }

    public int getCdrItemCount() {
