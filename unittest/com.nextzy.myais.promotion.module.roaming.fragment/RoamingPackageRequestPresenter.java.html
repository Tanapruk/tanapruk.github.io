package com.nextzy.myais.promotion.module.roaming.fragment;

import com.hwangjr.rxbus.RxBus;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.promotion.R;
import com.nextzy.myais.promotion.helper.roaming.delegate.RoamingAutoSpinner;
import com.nextzy.myais.promotion.helper.roaming.delegate.RoamingApplyPackageBlob;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageListEvent;

public class RoamingPackageRequestPresenter extends NextzyPresenter<RoamingPackageRequestContractor.View> implements RoamingPackageRequestContractor.Presenter {
    public RoamingPackageRequestPresenter(RoamingPackageRequestContractor.View view) {
        super(view);
    }

    public static RoamingPackageRequestPresenter createPresenter(RoamingPackageRequestContractor.View view) {
        return new RoamingPackageRequestPresenter(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }

    @Override
    public void onViewStart() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewStop() {
        RxBus.get().unregister(this);
    }

    @Override
    public void autoSelectCustomerType() {
        RoamingAutoSpinner.select(getView());
    }

    @Override
    public void validate(int countryPosition, int customerTypePosition, int packageTypePosition) {
        if (countryPosition == 0) {
            getView().showAlertDialogDismissOnClick(R.string.my_ais_tab_promotion_roaming_alert_package_no_country);
            return;
        }

        if (customerTypePosition == 0) {
            getView().showAlertDialogDismissOnClick(R.string.my_ais_tab_promotion_roaming_alert_package_select_customer_type);
            return;
        }

        if (packageTypePosition == 0) {
            getView().showAlertDialogDismissOnClick(R.string.my_ais_tab_promotion_roaming_alert_package_select_package_type);
            return;
        }

        saveDataAndGotoPackageList();
    }

    private void saveDataAndGotoPackageList() {
        RoamingApplyPackageBlob sharedData = RoamingApplyPackageBlob.getInstance();
        sharedData.setCountryNameParam(getView().getCountryParam());
        sharedData.setIsPrepaidParam(getView().getCustomerTypeParam());
        sharedData.setPackageTypeParam(getView().getPackageTypeParam());

        sharedData.setCountryName(getView().getCountryName());
        sharedData.setCustomerType(getView().getCustomerType());
        sharedData.setPackageType(getView().getPackageType());


        RxBus.get().post(new GotoRoamingPackageListEvent());
    }
}
