package com.ais.mimo.eservice.module.register.intro;

import android.os.Bundle;

import com.ais.mimo.eservice.module.register.main.RegisterActivity;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.widget.AISButton;
import com.nextzy.myais.common.widget.toolbar.simple.MyAisSimpleToolbar;
import com.nextzy.myais.myais.R;

public class RegisterIntroActivity extends NextzyMvpActivity<RegisterIntroContractor.Presenter> implements RegisterIntroContractor.View {
    private final static String TAG = RegisterIntroActivity.class.getSimpleName();
    private MyAisSimpleToolbar tbHeader;
    private AISButton btnCreateMyAis;

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_register_activity_intro;
    }

    @Override
    protected void createPresenter() {
        RegisterIntroPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        tbHeader = (MyAisSimpleToolbar) findViewById(R.id.my_ais_library_simple_toolbar_tb_header);
        btnCreateMyAis = (AISButton) findViewById(R.id.my_ais_register_intro_btn_create_my_ais);
    }

    @Override
    public void setupView() {
        tbHeader.setTitle(R.string.my_ais_register);
        tbHeader.setNavigationOnClickListener(v -> finish());
        btnCreateMyAis.setOnClickListener(view -> openRegisterActivity());
    }


    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
    }

    @Override
    public void restoreView() {
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
    }


    @Override
    public void saveInstanceState(Bundle outState) {
    }

    //</editor-fold>

    @Override
    public void openRegisterActivity() {
        openActivity(RegisterActivity.class, getIntent().getExtras(), true);
    }
}
