package com.nextzy.myais.common.utility.remark;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by trusttanapruk on 9/28/2016.
 */

public class RemarkUtility {
    public static RemarkUtility instance;

    public static RemarkUtility getInstance() {
        if (instance == null) {
            instance = new RemarkUtility();
        }
        return instance;
    }

    public <RemarkBlockData extends RemarkBlock> String sum(List<RemarkBlockData> rawRemarkBlockList) {
        return sum(null, rawRemarkBlockList);
    }

    public <RemarkBlockData extends RemarkBlock> String sum(String[] keys, List<RemarkBlockData> rawRemarkBlockList) {
        if (keys == null) {
            return addAllRemarks(rawRemarkBlockList);
        }

        return addAllRemarks(getDataListByKey(keys, rawRemarkBlockList));
    }

    private <RemarkBlockData extends RemarkBlock> String addAllRemarks(List<RemarkBlockData> filterRemarkBlockList) {
        String remark = "";
        for (RemarkBlock data : filterRemarkBlockList) {
            remark += addBrTag(data.getRemarkHeader());
            remark += addBrTag(data.getRemarkDisplay1());
            remark += addBrTag(data.getRemarkDisplay2());
            remark += addBrTag(data.getRemarkDisplay3());
        }
        return remark;
    }

    private String addBrTag(String remarkDisplay) {
        if (remarkDisplay == null || remarkDisplay.isEmpty()) {
            return "";
        }

        return remarkDisplay + "<br />";
    }

    private <RemarkBlockData extends RemarkBlock> List<RemarkBlockData> getDataListByKey(String[] keys, List<RemarkBlockData> rawRemarkBlockList) {
        if (rawRemarkBlockList == null) {
            return null;
        }
        List<RemarkBlockData> filteredDataList = new ArrayList<>();
        for (String key : keys) {
            for (RemarkBlockData data : rawRemarkBlockList) {
                if (data != null &&
                        data.getKey().equalsIgnoreCase(key) &&
                        !isRemarkDataContainByKey(filteredDataList, data)) {
                    filteredDataList.add(data);
                }
            }
        }
        return filteredDataList;
    }

    private <RemarkBlockData extends RemarkBlock> boolean isRemarkDataContainByKey(List<RemarkBlockData> dataList, RemarkBlockData remarkData) {
        if (dataList == null || remarkData == null) {
            return false;
        }
        for (RemarkBlockData data : dataList) {
            if (data == null || !data.getKey().equalsIgnoreCase(remarkData.getKey())) {
                continue;
            }
            return true;
        }
        return false;
    }
}
