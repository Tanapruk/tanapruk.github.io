package com.nextzy.myais.home.module.summaryinfo;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.home.module.summaryinfo.event.DetachCurrentPackageEvent;
import com.nextzy.myais.home.module.summaryinfo.event.DetachIswopEvent;
import com.nextzy.myais.home.module.summaryinfo.event.DetachSummaryPackageEvent;
import com.nextzy.myais.home.module.summaryinfo.event.DisableViewSwipeableEvent;
import com.nextzy.myais.home.module.summaryinfo.event.EnableViewSwipeableEvent;

/**
 * Created by Akexorcist on 9/13/2016 AD.
 */

public class SummaryInfoPresenter extends NextzyPresenter<SummaryInfoContractor.View> implements SummaryInfoContractor.Presenter {

    public static SummaryInfoPresenter createPresenter(SummaryInfoContractor.View view) {
        return new SummaryInfoPresenter(view);
    }

    public SummaryInfoPresenter(SummaryInfoContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {

    }

    @Subscribe
    public void onDetachIswopEvent(DetachIswopEvent event) {
        getView().detachIswopFragment();
    }

    @Subscribe
    public void onDetachSummaryPackageEvent(DetachSummaryPackageEvent event) {
        getView().detachSummaryPackageFragment();
    }

    @Subscribe
    public void onDetachCurrentPackageEvent(DetachCurrentPackageEvent event) {
        getView().detachCurrentPackageFragment();
    }

    @Subscribe
    public void onEnableViewSwipeableEvent(EnableViewSwipeableEvent event) {
        getView().enablePagerSwipeable();
    }

    @Subscribe
    public void onDisableViewSwipeableEvent(DisableViewSwipeableEvent event) {
        getView().disablePagerSwipeable();
    }
}
