public abstract class BaseMvpPresenter<CV extends BaseMvpContractorView> implements BaseMvpContractorPresenter {
    private CV view;

    protected BaseMvpPresenter(CV view) {
        this.view = view;
        view.setPresenter(this);
    }

    public CV getView() {
        return view;
    }

    /**
     * put the register here e.g.
     *
     * @see com.hwangjr.rxbus.Bus#register(Object)
     * make sure that you call getPresenter().onViewCreate() inside your {@link BaseMvpContractorView}
     */
    @Override
    public abstract void onViewCreate();

    /**
     * put the register here e.g.
     *
     * @see com.hwangjr.rxbus.Bus#unregister(Object)
     * make sure that you call getPresenter().onViewDestroy() inside your {@link BaseMvpContractorView}
     */
    @Override
    public abstract void onViewDestroy();

    /**
     * put the some code when view was started
     * <p>
     * make sure that you call getPresenter().onViewStart() inside your {@link BaseMvpContractorView}
     */
    @Override
    public abstract void onViewStart();

    /**
     * put the some code when view was stopped e.g.
     * <p>
     * make sure that you call getPresenter().onViewStop() inside your {@link BaseMvpContractorView}
     */
    @Override
    public abstract void onViewStop();
}
