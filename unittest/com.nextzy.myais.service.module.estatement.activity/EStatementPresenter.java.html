package com.nextzy.myais.service.module.estatement.activity;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;
import com.nextzy.myais.service.helper.estatement.event.GoToRegisPageEvent;

/**
 * Created by kajornsak on 10/11/2016 AD.
 */

public class EStatementPresenter extends NextzyActivityPresenter<EStatementContractor.View> implements EStatementContractor.Presenter {

    public EStatementPresenter(EStatementContractor.View view) {
        super(view);
    }

    public static EStatementPresenter createPresenter(EStatementContractor.View view){
        return new EStatementPresenter(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
        RxBus.get().register(this);
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
        RxBus.get().unregister(this);
    }

    @Subscribe
    public void GoToRegisPage(GoToRegisPageEvent goToRegisPageEvent){
        getView().goToRegisPage();
    }

}
