package com.ais.mimo.eservice.module.social;

import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;

import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.NextzyAndroidUtility;

/**
 * Created by Akexorcist on 10/31/2016 AD.
 */

/**
 * Deprecated For now
 * Users seems to like webview Approach.
 */

@Deprecated
public class SocialPortalActivity extends NextzyMvpActivity<SocialPortalContractor.Presenter> implements SocialPortalContractor.View {
    public static final String KEY_VALUE = "key_value";
    private static final String FACEBOOK_PACKAGE_NAME = "com.facebook.katana";
    private static final String TWITTER_PACKAGE_NAME = "com.twitter.android";
    private static final String LINE_PACKAGE_NAME = "jp.naver.line.android";

    private static final String FACEBOOK_PREPAID_URL_SCHEME = "fb://page/251069560456";
    private static final String FACEBOOK_POSTPAID_URL_SCHEME = "fb://page/127434041553";
    private static final String TWITTER_PREPAID_URL_SCHEME = "twitter://user?user_id=59416157";
    private static final String TWITTER_POSTPAID_URL_SCHEME = "twitter://user?user_id=59028840";
    private static final String LINE_URL = "http://line.me/ti/p/@aisofficialline";

    @Override
    public int getLayoutView() {
        return 0;
    }

    @Override
    protected void createPresenter() {
        SocialPortalPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {

    }

    @Override
    public void setupView() {

    }

    @Override
    public void restoreArgument(Bundle bundle) {
        getPresenter().setKeyType(bundle.getString(KEY_VALUE));
    }

    @Override
    public void initialize() {

    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        getPresenter().setKeyType(savedInstanceState.getString(KEY_VALUE));
    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }


    @Override
    public void openAisFacebook() {
        Mobile.getInstance().logEvent("APP_VIEW_AIS FACEBOOK", "5012", "5127");
        if (NextzyAndroidUtility.getInstance().isAppInstalled(this, FACEBOOK_PACKAGE_NAME)) {
            openAisFacebookPage();
        } else {
            NextzyAndroidUtility.getInstance().openAppInPlayStore(this, FACEBOOK_PACKAGE_NAME);
        }
        finish();
    }

    @Override
    public void openAisTwitter() {
        Mobile.getInstance().logEvent("APP_VIEW_AIS_TWITTER", "5012", "5128");
        if (NextzyAndroidUtility.getInstance().isAppInstalled(this, TWITTER_PACKAGE_NAME)) {
            openAisTwitterPage();
        } else {
            NextzyAndroidUtility.getInstance().openAppInPlayStore(this, TWITTER_PACKAGE_NAME);
        }
        finish();
    }

    @Override
    public void openAisLine() {
        Mobile.getInstance().logEvent("APP_VIEW_AIS LINE", "5012", "5129");
        if (NextzyAndroidUtility.getInstance().isAppInstalled(this, LINE_PACKAGE_NAME)) {
            openAisLineOfficialAccount();
        } else {
            NextzyAndroidUtility.getInstance().openAppInPlayStore(this, LINE_PACKAGE_NAME);
        }
        finish();
    }

    @Override
    public void closeActivityWithNothingHappened() {
        finish();
    }

    private void openAisFacebookPage() {
        if (Mobile.getInstance().isPrepaid()) {
            Uri uri = Uri.parse(FACEBOOK_PREPAID_URL_SCHEME);
            startActivity(new Intent(Intent.ACTION_VIEW, uri));
        } else {
            Uri uri = Uri.parse(FACEBOOK_POSTPAID_URL_SCHEME);
            startActivity(new Intent(Intent.ACTION_VIEW, uri));
        }
    }

    private void openAisTwitterPage() {
        if (Mobile.getInstance().isPrepaid()) {
            Uri uri = Uri.parse(TWITTER_PREPAID_URL_SCHEME);
            startActivity(new Intent(Intent.ACTION_VIEW, uri));
        } else {
            Uri uri = Uri.parse(TWITTER_POSTPAID_URL_SCHEME);
            startActivity(new Intent(Intent.ACTION_VIEW, uri));
        }
    }

    private void openAisLineOfficialAccount() {
        Uri uri = Uri.parse(LINE_URL);
        startActivity(new Intent(Intent.ACTION_VIEW, uri));
    }
}
