package com.ais.mimo.eservice.module.social;

import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by Akexorcist on 10/31/2016 AD.
 */

public class SocialPortalPresenter extends NextzyActivityPresenter<SocialPortalContractor.View> implements SocialPortalContractor.Presenter {
    private String keyType;

    public static SocialPortalPresenter createPresenter(SocialPortalContractor.View view) {
        return new SocialPortalPresenter(view);
    }

    public SocialPortalPresenter(SocialPortalContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {

    }

    @Override
    public void onViewDestroy() {

    }

    @Override
    public void onViewStart() {
        setSocialRedirect();
    }

    @Override
    public void onViewStop() {

    }

    @Override
    public void setKeyType(String keyType) {
        this.keyType = keyType;
    }

    @Override
    public String getKeyType() {
        return keyType;
    }

    public void setSocialRedirect() {
        if (SocialPortalKey.KEY_FACEBOOK.equals(getKeyType())) {
            getView().openAisFacebook();
        } else if (SocialPortalKey.KEY_LINE.equals(getKeyType())) {
            getView().openAisLine();
        } else if (SocialPortalKey.KEY_TWITTER.equals(getKeyType())) {
            getView().openAisTwitter();
        } else {
            getView().closeActivityWithNothingHappened();
        }
    }
}
