package com.ais.mimo.eservice.helper.register.network.operator;


import com.nextzy.myais.common.network.model.GenericResult;
import com.nextzy.myais.common.network.tool.NextzyPredicate;

import rx.Single;
import rx.functions.Func1;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public abstract class RegisterResultHasDataCheck<T extends GenericResult, E extends Throwable> implements Func1<T, Single<T>> {

    protected abstract E getThrowable(String message);

    @Override
    public Single<T> call(T t) {
        if ( isNot20000nSSBE309OrNoData(t) ) {
            return Single.error(getThrowable(t.getResultDescription()));
        }
        return Single.just(t);
    }

    private NextzyPredicate<T> dataIsSSBE309 = t -> t.getResultCode().equalsIgnoreCase( "SSBE309" );
    private NextzyPredicate<T> dataIsNot20000 = t -> !t.getResultCode().equalsIgnoreCase( "20000" );
    private NextzyPredicate<T> dataIsNull = t -> t.getData() == null;
    private NextzyPredicate<T> dataListIsNull = t -> t.getDataList() == null;
    private NextzyPredicate<T> dataListIs0 = t -> t.getDataList().size() == 0;
    private NextzyPredicate<T> dataResultCodeIsNot20000nSSBE309
            = t -> !t.getResultCode().equalsIgnoreCase( "20000" )
            && !t.getResultCode().equalsIgnoreCase("SSBE309");

    protected boolean validate(NextzyPredicate<T> nextzyPredicate, T t) {
        return nextzyPredicate.isTrue(t);
    }

    protected boolean isNot20000nSSBE309OrNoData(T t) {
        return validate( dataResultCodeIsNot20000nSSBE309, t) || validate(dataIsNull, t);
    }

    protected boolean isNot20000nSSBE309HasNoDataListAndSizeZero(T t) {
        return validate( dataResultCodeIsNot20000nSSBE309, t) || validate(dataListIsNull, t) || validate(dataListIs0, t);
    }
    protected boolean isNot20000nSSBE309OrNoDataList(T t) {
        return validate( dataResultCodeIsNot20000nSSBE309, t) || validate(dataListIsNull, t);
    }
    protected boolean isNot20000nSSBE309( T t) {
        return validate( dataResultCodeIsNot20000nSSBE309, t);
    }

    protected boolean isNot20000( T t) {
        return validate(dataIsNot20000, t);
    }

    protected boolean isSSBE309nHaveData( T t) {
        return validate( dataIsSSBE309, t) && !validate(dataIsNull, t);
    }

    protected boolean isNot20000nNoData(T t) {
        return validate(dataIsNot20000, t) || validate(dataIsNull, t);
    }

    protected boolean isNot20000nNoDataList(T t) {
        return validate(dataIsNot20000, t) || validate(dataListIsNull, t);
    }

}
