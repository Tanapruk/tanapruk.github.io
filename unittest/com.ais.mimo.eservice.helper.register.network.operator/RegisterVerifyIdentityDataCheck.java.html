package com.ais.mimo.eservice.helper.register.network.operator;


import com.ais.mimo.eservice.helper.register.network.exception.RegisterSSBE309Exception;
import com.ais.mimo.eservice.helper.register.network.exception.RegisterVerifyIdentityDataException;
import com.ais.mimo.eservice.helper.register.network.response.RegisterVerifyIdentityResult;

import rx.Single;

/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class RegisterVerifyIdentityDataCheck extends RegisterResultHasDataCheck<RegisterVerifyIdentityResult, RegisterVerifyIdentityDataException>{
    @Override
    protected RegisterVerifyIdentityDataException getThrowable( String message) {
        return new RegisterVerifyIdentityDataException(message);
    }

    @Override
    public Single<RegisterVerifyIdentityResult> call(RegisterVerifyIdentityResult t) {
        if ( isSSBE309nHaveData(t)
                && t.getData().getCaptcha() != null
                && t.getData().getReferenceId() != null){
            return Single.error(new RegisterSSBE309Exception(
                    t.getResultDescription(),
                    t.getData().getCaptcha(),
                    t.getData().getReferenceId()));
        }

        if ( isNot20000nNoData(t)) {
            return Single.error(getThrowable(t.getResultDescription()));
        }
        if( t.getData().getReferenceId() == null){
            return Single.error(getThrowable(t.getResultDescription()));
        }
        return Single.just(t);
    }

}
