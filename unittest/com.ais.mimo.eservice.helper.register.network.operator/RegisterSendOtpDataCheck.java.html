package com.ais.mimo.eservice.helper.register.network.operator;


import com.ais.mimo.eservice.helper.register.network.RegisterManager;
import com.ais.mimo.eservice.helper.register.network.exception.RegisterSendOtpCaptchaException;
import com.ais.mimo.eservice.helper.register.network.exception.RegisterSendOtpDataException;
import com.ais.mimo.eservice.helper.register.network.response.RegisterSendOtpResult;

import rx.Single;


/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class RegisterSendOtpDataCheck extends RegisterResultHasDataCheck<RegisterSendOtpResult, RegisterSendOtpDataException> {

    @Override
    protected RegisterSendOtpDataException getThrowable(String message) {
        return new RegisterSendOtpDataException(message);
    }

    @Override
    public Single<RegisterSendOtpResult> call(RegisterSendOtpResult registerSendOtpResult) {
        if (RegisterManager.SSBE309.equals(registerSendOtpResult.getResultCode())) {
            RegisterSendOtpCaptchaException exception = new RegisterSendOtpCaptchaException(registerSendOtpResult.getResultDescription());
            exception.setCaptcha(registerSendOtpResult.getData().getCaptcha());
            exception.setReferenceId(registerSendOtpResult.getData().getReferenceId());
            return Single.error(exception);
        }
        return super.call(registerSendOtpResult);
    }
}
