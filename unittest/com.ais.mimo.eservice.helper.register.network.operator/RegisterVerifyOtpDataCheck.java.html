package com.ais.mimo.eservice.helper.register.network.operator;


import com.ais.mimo.eservice.helper.register.network.exception.RegisterVerifyOtpDataException;
import com.ais.mimo.eservice.helper.register.network.response.RegisterVerifyOtpResult;

import rx.Single;

/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class RegisterVerifyOtpDataCheck extends RegisterResultHasDataCheck<RegisterVerifyOtpResult, RegisterVerifyOtpDataException>{

    @Override
    protected RegisterVerifyOtpDataException getThrowable( String message) {
        return new RegisterVerifyOtpDataException(message);
    }

    @Override
    public Single<RegisterVerifyOtpResult> call(RegisterVerifyOtpResult t) {
        if ( isNot20000(t)) {
            return Single.error(getThrowable(t.getResultDescription()));
        }
        return Single.just(t);
    }
}
