package com.ais.mimo.eservice.helper.register.network.operator;


import com.ais.mimo.eservice.helper.register.network.exception.ChangePasswordWrongFormatException;
import com.ais.mimo.eservice.helper.register.network.response.ChangePasswordResult;
import com.nextzy.myais.common.network.tool.ResultWithoutDataCheck;

import rx.Single;

/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class ChangePasswordDataCheck extends ResultWithoutDataCheck<ChangePasswordResult, ChangePasswordWrongFormatException> {

    @Override
    protected ChangePasswordWrongFormatException getThrowable(String message) {
        return new ChangePasswordWrongFormatException(message);
    }

    @Override
    public Single<ChangePasswordResult> call(ChangePasswordResult changePasswordResult) {
        if ("SSBE002".equals(changePasswordResult.getResultCode())) {
            return Single.error(new ChangePasswordWrongFormatException(changePasswordResult.getDeveloperMessage()));
        }

        return super.call(changePasswordResult);
    }
}
