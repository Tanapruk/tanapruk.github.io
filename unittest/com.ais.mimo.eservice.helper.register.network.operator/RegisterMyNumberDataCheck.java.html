package com.ais.mimo.eservice.helper.register.network.operator;


import com.ais.mimo.eservice.helper.register.network.exception.RegisterMyNumberDataException;
import com.ais.mimo.eservice.helper.register.network.response.RegisterMyNumberResult;

import rx.Single;


/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class RegisterMyNumberDataCheck extends RegisterResultHasDataCheck<RegisterMyNumberResult, RegisterMyNumberDataException>{
    @Override
    protected RegisterMyNumberDataException getThrowable( String message) {
        return new RegisterMyNumberDataException(message);
    }

    @Override
    public Single<RegisterMyNumberResult> call( RegisterMyNumberResult numberResult ){
        if ( isNot20000nNoDataList( numberResult )){
            return Single.error(getThrowable( numberResult.getResultDescription() ));
        }
        return Single.just( numberResult );
    }
}
