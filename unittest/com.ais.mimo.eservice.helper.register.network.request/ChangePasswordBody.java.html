public class ChangePasswordBody {
    String user;
    String password;
    String email;
    String referenceId;

    public ChangePasswordBody(String user, String password, String email, String referenceId) {
        this.user = user;
        this.password = password;
        this.email = email;
        this.referenceId = referenceId;
    }

    public ChangePasswordBody(String user, String password, String referenceId) {
        this.user = user;
        this.password = password;
        this.referenceId = referenceId;
    }
}
