package com.ais.mimo.eservice.utility;

import android.util.Base64;

/**
 * Created by TheKhaeng on 10/12/2016.
 */

public class CaptchaUtility {

    private static CaptchaUtility captchaUtil;

    public static CaptchaUtility getInstance() {
        if (captchaUtil == null) {
            captchaUtil = new CaptchaUtility();
        }
        return captchaUtil;
    }

    public String decodeBase64(String base64) {
        byte[] valueDecoded = Base64.decode(base64, Base64.DEFAULT);
        return new String(valueDecoded);
    }

    public String encodeBase64(String string) {
        byte[] valueEncode = Base64.encode(string.getBytes(), Base64.DEFAULT);
        return new String(valueEncode);
    }
}
