package com.ais.mimo.eservice;

import android.content.Context;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.preference.PreferenceManager;
import android.support.multidex.MultiDex;

import com.ais.mimo.eservice.module.login.activity.LoginActivity;
import com.akexorcist.localizationactivity.LanguageSetting;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.balanceusage.module.cqs.activity.CqsActivity;
import com.nextzy.myais.common.NextzyApplication;
import com.nextzy.myais.common.event.GoToRoamingCurrentPackageEvent;
import com.nextzy.myais.common.event.GoToRoamingPackageEvent;
import com.nextzy.myais.common.event.GotoChangePackageEvent;
import com.nextzy.myais.common.event.GotoCqsEvent;
import com.nextzy.myais.common.event.GotoCurrentPackageEvent;
import com.nextzy.myais.common.event.GotoIswopAcitivtyEvent;
import com.nextzy.myais.common.event.GotoLoginActivityEvent;
import com.nextzy.myais.common.utility.NextzyLanguage;
import com.nextzy.myais.promotion.module.changepackage.ChangePackageActivity;
import com.nextzy.myais.promotion.module.changepackage.constant.ChangePackageKey;
import com.nextzy.myais.promotion.module.currentpackage.activity.PackageInfoActivity;
import com.nextzy.myais.promotion.module.roaming.activity.RoamingApplyPackageActivity;
import com.nextzy.myais.promotion.module.roaming.activity.RoamingCheckSubscriptionActivity;
import com.nextzy.myais.service.module.iswop.activity.IswopActivity;
import com.orhanobut.hawk.Hawk;

import static android.content.Intent.FLAG_ACTIVITY_NEW_TASK;

/**
 * Created by trusttanapruk on 8/15/2016.
 */
public class MainApplication extends NextzyApplication {

    protected Object onGoToEventListener = new Object() {

        @Subscribe
        public void onGotoLoginActivity(GotoLoginActivityEvent event) {
            Bundle bundle = new Bundle();
            bundle.putBoolean(LoginActivity.SHOULD_BYPASS, false);
            Intent intent = new Intent(getApplicationContext(), LoginActivity.class);
            intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TASK | FLAG_ACTIVITY_NEW_TASK);
            intent.putExtras(bundle);
            getApplicationContext().startActivity(intent);
        }

        @Subscribe
        public void onGotoChangeMainActivity(GotoChangePackageEvent event) {
            Bundle bundle = new Bundle();
            bundle.putString(ChangePackageKey.KEY_VALUE, ChangePackageKey.TYPE_MAIN);
            Intent intent = new Intent(getApplicationContext(), ChangePackageActivity.class);
            intent.addFlags(FLAG_ACTIVITY_NEW_TASK);
            intent.putExtras(bundle);
            getApplicationContext().startActivity(intent);
        }

        @Subscribe
        public void onGotoCurrentPackageActivity(GotoCurrentPackageEvent event) {
            openActivity(PackageInfoActivity.class);
        }

        @Subscribe
        public void onGotoCqsActivity(GotoCqsEvent event) {
            openActivity(CqsActivity.class);
        }

        @Subscribe
        public void onGotoIswopActivity(GotoIswopAcitivtyEvent event) {
            openActivity(IswopActivity.class);
        }

        @Subscribe
        public void onGotoRoamingPackageActivity(GoToRoamingPackageEvent event) {
            openActivity(RoamingApplyPackageActivity.class);
        }

        @Subscribe
        public void onGotoRoamingCurrentPackageActivity(GoToRoamingCurrentPackageEvent event) {
            openActivity(RoamingCheckSubscriptionActivity.class);
        }
    };

    private void openActivity(Class className) {
        Intent intent = new Intent(getApplicationContext(), className);
        intent.addFlags(FLAG_ACTIVITY_NEW_TASK);
        getApplicationContext().startActivity(intent);
    }


    @Override
    protected void attachBaseContext(Context base) {
        super.attachBaseContext(base);
        MultiDex.install(this);
        RxBus.get().register(onGoToEventListener);
        setThaiLanguageFirstTime();
    }

    private void setThaiLanguageFirstTime() {
        SharedPreferences sharedPreferences = PreferenceManager.getDefaultSharedPreferences(this);
        final String IS_FIRST_TIME = "is_first_time";
        if (sharedPreferences.getBoolean(IS_FIRST_TIME, true)) {
            LanguageSetting.setLanguage(this, NextzyLanguage.THAI);
            sharedPreferences.edit().putBoolean(IS_FIRST_TIME, false).apply();
        }
    }

    @Override
    public void onCreate() {
        super.onCreate();
        Hawk.init(this).build();
    }

    @Override
    public void onTerminate() {
        RxBus.get().unregister(onGoToEventListener);
        super.onTerminate();
    }

}
