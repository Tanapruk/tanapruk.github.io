package com.ais.mimo.eservice.module.tutorial;


import android.graphics.Bitmap;
import android.graphics.Point;
import android.os.Bundle;
import android.widget.FrameLayout;
import android.widget.ImageView;

import com.ais.mimo.eservice.module.main.activity.MainActivity;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.DefaultPreference;
import com.nextzy.myais.common.utility.NextzyAndroidUtility;
import com.nextzy.myais.common.utility.NextzyBitmapUtility;
import com.nextzy.myais.common.utility.NextzyLanguage;
import com.nextzy.myais.myais.R;

public class TutorialActivity extends NextzyMvpActivity<TutorialContractor.Presenter> implements TutorialContractor.View {
    private static final String TAG = TutorialActivity.class.getSimpleName();
    public static final String TUTORIAL_VERSION = "1.0";

    private FrameLayout layoutContainer;
    private ImageView ivScreen;

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_tutorial_activity;
    }

    @Override
    protected void createPresenter() {
        TutorialPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        layoutContainer = (FrameLayout) findViewById(R.id.my_ais_tutorial_layout_container);
        ivScreen = (ImageView) findViewById(R.id.my_ais_tutorial_iv_screen);
    }

    @Override
    public void setupView() {
        layoutContainer.setOnClickListener(v -> clickTutorialLayout());
        ivScreen.setOnClickListener(v -> clickTutorialLayout());
        ivScreen.setImageBitmap(getTutorialBitmapFromRawResource());
    }

    @Override
    public void onBackPressed() {
        clickTutorialLayout();
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
    }


    @Override
    public void restoreView() {
    }

    @Override
    public void saveInstanceState(Bundle outState) {
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
    }

    private void clickTutorialLayout() {
        DefaultPreference.getInstance().saveFalseIsFirstOpenTutorialActivity(TUTORIAL_VERSION);
        openActivityFadeInOut(MainActivity.class, true);
    }

    private Bitmap getTutorialBitmapFromRawResource() {
        Point point = NextzyAndroidUtility.getInstance().getDeviceScreenSize(this);
        float ratio = NextzyAndroidUtility.getInstance().calculateRatio(point);
        int resId = getTutorialImageResourceNameByRatio(ratio);
        int width = point.x;
        int height = point.y;
        return NextzyBitmapUtility.decodeBitmapFromResource(getApplicationContext().getResources(), resId, width, height);
    }

    private int getTutorialImageResourceNameByRatio(float ratio) {
        String fileName = "my_ais_tutorial_";
        if (ratio == NextzyAndroidUtility.RATIO_16_10 ||
                ratio == NextzyAndroidUtility.RATIO_3_2) {
            fileName += "1600x2560";
        } else if (ratio == NextzyAndroidUtility.RATIO_16_9) {
            fileName += "1080x1920";
        } else if (ratio == NextzyAndroidUtility.RATIO_4_3) {
            fileName += "1536x2048";
        } else if (ratio == NextzyAndroidUtility.RATIO_5_3) {
            fileName += "480x800";
        } else {
            fileName += "480x800";
        }
        fileName += "_" + NextzyLanguage.getInstance().getCurrentLanguage().toLowerCase();
        int resId = NextzyAndroidUtility.getInstance().getDrawableResourceByFilename(this, fileName);
        if (resId == -1 || resId == 0) {
            return R.drawable.my_ais_tutorial_480x800_en;
        }
        return resId;
    }
}
