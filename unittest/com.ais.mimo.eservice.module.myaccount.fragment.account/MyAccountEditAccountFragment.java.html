public class MyAccountEditAccountFragment extends NextzyMvpFragment<MyAccountEditAccountContractor.Presenter> implements MyAccountEditAccountContractor.View, MyNumberDropDownDialog.MyNumberListener {
    private String KEY_MY_NUMBER_RESULT = "myNumberResultKey";
    private String KEY_MY_NUMBER = "myNumberKey";
    private String KEY_SHOW_OTP_VIEW = "showOtpViewKey";
    private String KEY_MOBILE_TYPE = "mobileTypeKey";
    private String KEY_REFERENCE_ID = "referenceIdKey";
    private String KEY_EMAIL = "emailKey";
    private String KEY_NTYPE = "ntypeKey";
    private String KEY_SHOW_CHANGE_LOGIN_SUCCESS_DIALOG = "showChangeLoginSuccessDialogKey";
    private String KEY_SHOW_EMAIL = "showEmail";

    private boolean showOtp = false;
    private boolean showChangeLoginSuccessDialog = false;
    private boolean showEmail = false;

    public final static String TAG = MyAccountEditAccountFragment.class.getSimpleName();

    private LinearLayout llContainer;
    private TextView tvMainNumber;
    private MyAisRegisterEditText edtOtp;
    private MyAisRegisterEditText edtMobileNumber;
    private LinearLayout llRequestOTP;
    private LinearLayout llOK;
    private LinearLayout llOtp;
    private LinearLayout llDropdown;
    private LinearLayout llDropdownView;
    private AISTextView tvMobileNumber;
    private ImageView imvSpinner;
    private AISButton btnCancel;
    private AISButton btnRequestOTP_OK;
    private MyNumberDropDownDialog myNumberDropDownDialog;
    private LinearLayout llEmail;
    private AISTextView tvEmail;

    public MyAccountEditAccountFragment() {
        super();
    }

    public static MyAccountEditAccountFragment newInstance() {
        MyAccountEditAccountFragment fragment = new MyAccountEditAccountFragment();
        Bundle args = new Bundle();
        fragment.setArguments(args);
        return fragment;
    }

    public static MyAccountEditAccountFragment newInstance(Bundle bundle) {
        MyAccountEditAccountFragment fragment = new MyAccountEditAccountFragment();
        fragment.setArguments(bundle);
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter() {
        MyAccountEditAccountPresenter.createPresenter(this);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView(View view) {
        llContainer = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_container);
        llRequestOTP = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_next);
        llOK = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_ok);
        llOtp = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_otp);
        tvMainNumber = (TextView) view.findViewById(R.id.my_ais_account_tv_column_r);
        edtOtp = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_account_otp);
        edtMobileNumber = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_account_edt_mobile_number);
        llDropdown = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_drop_down);
        llDropdownView = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_drop_down_view);
        tvMobileNumber = (AISTextView) view.findViewById(R.id.my_ais_account_tv_mobile_number);
        imvSpinner = (ImageView) view.findViewById(R.id.my_ais_account_imv_spinner);
        btnCancel = (AISButton) view.findViewById(R.id.my_ais_account_btn_cancel);
        btnRequestOTP_OK = (AISButton) view.findViewById(R.id.my_ais_account_btn_requrst_otp_ok);
        llEmail = (LinearLayout) view.findViewById(R.id.my_ais_account_ll_email);
        tvEmail = (AISTextView) view.findViewById(R.id.my_ais_edit_account_tv_email);
    }

    @Override
    public void setupView() {
        llRequestOTP.setVisibility(View.GONE);
        edtOtp.setInputType(InputType.TYPE_CLASS_NUMBER | InputType.TYPE_NUMBER_VARIATION_PASSWORD);
        edtOtp.setMaxLengthEdt(4);
        edtOtp.setHintText(getString(R.string.my_ais_account_hint_sms_otp));
        btnCancel.setOnClickListener(view -> clickBack());

        if (isArgumentHasChangeMobileNumber()) {
            edtMobileNumber.setText(getArguments().getString(KEY_CHANGE_MOBILE_NUMBER));
            edtMobileNumber.setEditable(false);
            keyInView();
        }
    }


    @Override
    public void initialize() {
        getPresenter().updateMyAccountUi();
        if (!isArgumentHasChangeMobileNumber()) {
            getPresenter().requestMyNumber();
        }
    }

    @Override
    public void restoreView() {
        updateUiByResult(getPresenter().getMyNumberResult());
        if (showOtp)
            showOtpView();
        if (showChangeLoginSuccessDialog)
            showChangeLoginSuccessDialog(getPresenter().getMobileNumber());
        if (showEmail)
            showEmail(getPresenter().getEmail());
    }

    /**
     * onCreateView
     **/
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_account_fragment_edit_mobile_number;
    }

    @Override
    public void onResume() {
        super.onResume();
    }

    /**
     * State
     **/
    @Override
    public void saveInstanceState(Bundle outState) {
        // Save Instance State here
        outState.putParcelable(KEY_MY_NUMBER_RESULT, getPresenter().getMyNumberResult());
        outState.putBoolean(KEY_SHOW_OTP_VIEW, showOtp);
        outState.putString(KEY_MY_NUMBER, getPresenter().getMobileNumber());
        outState.putString(KEY_REFERENCE_ID, getPresenter().getReferenceId());
        outState.putString(KEY_EMAIL, getPresenter().getEmail());
        outState.putString(KEY_NTYPE, getPresenter().getNtype());
        outState.putBoolean(KEY_SHOW_CHANGE_LOGIN_SUCCESS_DIALOG, showChangeLoginSuccessDialog);
        outState.putBoolean(KEY_SHOW_EMAIL, showEmail);
        MobileType mobileType = getPresenter().getMobileType();
        if (mobileType != null)
            outState.putInt(KEY_MOBILE_TYPE, mobileType.ordinal());
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        // Restore instance state here
        getPresenter().setMyNumber(savedInstanceState.getParcelable(KEY_MY_NUMBER_RESULT));
        showOtp = savedInstanceState.getBoolean(KEY_SHOW_OTP_VIEW);
        showChangeLoginSuccessDialog = savedInstanceState.getBoolean(KEY_SHOW_CHANGE_LOGIN_SUCCESS_DIALOG);
        showEmail = savedInstanceState.getBoolean(KEY_SHOW_EMAIL);
        getPresenter().setMobileNumber(savedInstanceState.getString(KEY_MY_NUMBER));
        getPresenter().setMobileType(MobileType.values()[savedInstanceState.getInt(KEY_MOBILE_TYPE)]);
        getPresenter().setReferenceId(savedInstanceState.getString(KEY_REFERENCE_ID));
        getPresenter().setEmail(savedInstanceState.getString(KEY_EMAIL));
        getPresenter().setNtype(savedInstanceState.getString(KEY_NTYPE));
    }

    @Override
    public void updateUi(String mobileNumber) {
        tvMainNumber.setText(mobileNumber);
    }

    @Override
    public void updateUiByResult(MyNumberResult myNumberResult) {
        getPresenter().setMyNumber(myNumberResult);
        if (myNumberResult.hasDataList() && myNumberResult.getDataList().size() > 0) {
            if (getPresenter().getMyNumberDropDownList().size() > 0) {
                dropDownView();
                return;
            }
        }
        keyInView();
    }

    @Override
    public void showResponseException(String resultDescription) {
        showAlertDialogDismissOnClick(resultDescription);
    }

    @Override
    public void showDialogWithRes(int resId) {
        showAlertDialogDismissOnClick(getString(resId));
    }

    @Override
    public void showDialogDuplicateMaster() {
        String s = getString(R.string.my_ais_account_detail_dialog_same_master_mobile);
        showDialog(s);
    }

    @Override
    public void showDialog(String s) {
        showAlertDialogDismissOnClick(s);
    }

    @Override
    public void showOtpView() {
        showOtp = true;
        dismissAllDialogs();
        llDropdownView.setClickable(false);
        imvSpinner.setVisibility(View.GONE);
        llOtp.setVisibility(View.VISIBLE);
        llRequestOTP.setVisibility(View.GONE);
        edtMobileNumber.setEditable(false);
        llOK.setVisibility(View.VISIBLE);
        btnCancel = (AISButton) llOK.findViewById(R.id.my_ais_account_btn_cancel);
        btnCancel.setOnClickListener(view -> clickBack());
        btnRequestOTP_OK = (AISButton) llOK.findViewById(R.id.my_ais_account_btn_ok);
        btnRequestOTP_OK.setOnClickListener(view -> getPresenter().verifyOtp(edtOtp.getOtp()));
        edtOtp.setOnEditorActionListener((v, actionId, event) -> {
            if ((event != null && (event.getKeyCode() == KeyEvent.KEYCODE_ENTER)) || (actionId == EditorInfo.IME_ACTION_DONE)) {
                getPresenter().verifyOtp(edtOtp.getOtp());
            }
            return false;
        });
    }

    @Override
    public void showChangeLoginSuccessDialog(String mobileNumber) {
        showChangeLoginSuccessDialog = true;
        String message = getString(R.string.my_ais_account_change_main_number_success)
                + FormatUtility.getInstance().getStringWithFormat(mobileNumber, FORMAT_MOBILE_NUMBER)
                + getString(R.string.my_ais_account_successfully)
                + getString(R.string.my_ais_account_you_can_login_to_use_my_ais_right_away);
        AisDialog.getInstance().showSimpleDialog(getFragmentManager(),
                null,
                message,
                getString(R.string.my_ais_account_login),
                data -> getPresenter().callLogout());
    }

    @Override
    public void showDialogSetNewPasscode(boolean isFibre) {
        AisDialog.getInstance().showChoiceDialog(
                getFragmentManager(),
                null,
                getString(R.string.my_ais_account_detail_dialog_set_new_passcode),
                getString(R.string.ok),
                getString(R.string.cancel),
                new ChoiceDialogClickListener() {
                    @Override
                    public void onDialogPositiveButtonClick(@Nullable Bundle data) {
                        getPresenter().requestVerifyIdentity(getPresenter().getMobileNumber(), isFibre);
                    }

                    @Override
                    public void onDialogNegativeButtonClick(@Nullable Bundle data) {
                        clickBack();
                    }
                }
        );
    }

    @Override
    public void showMobileNumberDefaultDropDown(String mobileNumber) {
        tvMobileNumber.setText(mobileNumber);
    }

    @Override
    public void showUpdateEmailDialog() {
        Timber.d("myTimber : showUpdateEmailDialog");
        AisDialog.getInstance().showChoiceDialog(
                getFragmentManager(),
                null,
                getString(R.string.my_ais_account_detail_dialog_update_email),
                getString(R.string.ok),
                getString(R.string.cancel),
                new ChoiceDialogClickListener() {
                    @Override
                    public void onDialogPositiveButtonClick(@Nullable Bundle data) {
                        AisDialog.getInstance().dismissDialog();
                        getPresenter().goToEditEmailFragment();
                    }

                    @Override
                    public void onDialogNegativeButtonClick(@Nullable Bundle data) {
                        clickBack();
                    }
                }
        );
    }

    @Override
    public void showOtpError() {
        edtOtp.showError();
    }

    @Override
    public void showDefaultOtpEdt() {
        edtOtp.showDefault();
    }

    @Override
    public void showEmail(String email) {
        showEmail = true;
        llEmail.setVisibility(View.VISIBLE);
        tvEmail.setText(email);
    }

    @Override
    public void popBack() {
        getActivity().getSupportFragmentManager().popBackStack();
    }

    @Override
    public void onMyNumberSelected(String mobileNumber, int position) {
        dismissMyNumberDropDownDialog();
        tvMobileNumber.setText(mobileNumber);
    }

    @Override
    public void onDismiss() {
        dismissMyNumberDropDownDialog();
    }

    private void clickRequestOtpAndValidateMobileNumber() {
        String mobileNumber = edtMobileNumber.getStringFromEditText();
        getPresenter().validateMobileNumber(mobileNumber);
    }

    private void clickRequestOtp() {
        String mobileNumber = tvMobileNumber.getText().toString();
        getPresenter().setMobileNumber(mobileNumber);
        getPresenter().verifyPriority();

    }

    private void clickBack() {
        dismissDialog();
        getActivity().onBackPressed();
    }

    private void keyInView() {
        edtMobileNumber.setVisibility(View.VISIBLE);
        llDropdown.setVisibility(View.GONE);
        btnRequestOTP_OK.setText(R.string.my_ais_account_request_otp);
        llRequestOTP.setVisibility(View.VISIBLE);
        btnRequestOTP_OK.setOnClickListener(view -> clickRequestOtpAndValidateMobileNumber());
        edtMobileNumber.setOnEditorActionListener((v, actionId, event) -> {
            if ((event != null && (event.getKeyCode() == KeyEvent.KEYCODE_ENTER)) || (actionId == EditorInfo.IME_ACTION_DONE)) {
                clickRequestOtpAndValidateMobileNumber();
            }
            return false;
        });
    }

    private void dropDownView() {
        edtMobileNumber.setVisibility(View.GONE);
        llDropdown.setVisibility(View.VISIBLE);
        List<MyNumberDropDownDialog.ContentItem> contentItemList = getPresenter().getMyNumberDropDownList();
        llDropdownView.setOnClickListener(view -> showMyNumberDropDownDialog(contentItemList));
        btnRequestOTP_OK.setText(R.string.my_ais_account_request_otp);
        llRequestOTP.setVisibility(View.VISIBLE);
        btnRequestOTP_OK.setOnClickListener(view -> clickRequestOtp());
        getPresenter().setMobileNumberDefaultDropDown(contentItemList);
    }

    private void showMyNumberDropDownDialog(List<MyNumberDropDownDialog.ContentItem> contentItemList) {
        myNumberDropDownDialog = new MyNumberDropDownDialog(tvMobileNumber.getText().toString());
        myNumberDropDownDialog.setItemList(contentItemList);
        myNumberDropDownDialog.setMyNumberListener(this);
        myNumberDropDownDialog.showPopupMenu(getContext());
    }

    public void dismissMyNumberDropDownDialog() {
        if (myNumberDropDownDialog != null) {
            myNumberDropDownDialog.closePopupMenu();
            myNumberDropDownDialog = null;
        }
    }

    private boolean isArgumentHasChangeMobileNumber() {
        return !NextzyUtility.getInstance().isEmptyString(getArguments().getString(KEY_CHANGE_MOBILE_NUMBER));
    }
}
