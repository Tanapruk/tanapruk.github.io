package com.ais.mimo.eservice.helper.register.network.operator.filter;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.helper.register.network.response.RegisterMyNumberResult;
import com.google.gson.Gson;
import com.nextzy.myais.common.database.NextzyDatabase;

import java.util.List;

import rx.Single;
import rx.functions.Func1;

import static com.ais.mimo.eservice.helper.register.network.RegisterManager.MASTER_MOBILE_NUMBER;

/**
 * Created by TheKhaeng on 10/27/2016.
 */

public class RegisterMyNumberFilter implements Func1<RegisterMyNumberResult, Single<RegisterMyNumberResult>> {

    private String number;

    public RegisterMyNumberFilter(String number) {
        this.number = number;
    }

    @Override
    public Single<RegisterMyNumberResult> call(RegisterMyNumberResult registerMyNumberResult) {
        return Single.just(filterOwnNumber(registerMyNumberResult));
    }

    private RegisterMyNumberResult filterOwnNumber(RegisterMyNumberResult result) {
        if (number != null) {
            List<MobileData> mobiles = result.getDataList();
            for (int i = 0; i < mobiles.size(); i++) {
                if (number.equals(mobiles.get(i).getNumber())) {
                    saveMainNumberRegister(mobiles.remove(i).getNumber());
                    return result;
                }
            }
            saveMainNumberRegister(number);
        }
        return result;
    }

    private void saveMainNumberRegister(String mobileNumber) {
        String className = mobileNumber.getClass().getSimpleName();
        String json = new Gson().toJson(mobileNumber);
        NextzyDatabase.getInstance().add(MASTER_MOBILE_NUMBER, className, json);
    }

}
