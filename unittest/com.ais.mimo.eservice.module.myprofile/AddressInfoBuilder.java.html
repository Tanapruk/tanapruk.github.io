package com.ais.mimo.eservice.module.myprofile;

import android.content.Context;

import com.nextzy.myais.myais.R;

/**
 * Created by Akexorcist on 11/23/2016 AD.
 */

public class AddressInfoBuilder {
    private String houseNumber;
    private String room;
    private String floor;
    private String buildingName;
    private String alley;
    private String road;
    private String subDistrict;
    private String district;
    private String province;
    private String postalCode;

    public static AddressInfoBuilder create() {
        return new AddressInfoBuilder();
    }

    private AddressInfoBuilder() {
    }

    public AddressInfoBuilder setHouseNumber(String houseNumber) {
        this.houseNumber = houseNumber;
        return this;
    }

    public AddressInfoBuilder setAlley(String alley) {
        this.alley = alley;
        return this;
    }

    public AddressInfoBuilder setRoad(String road) {
        this.road = road;
        return this;
    }

    public AddressInfoBuilder setSubDistrict(String subDistrict) {
        this.subDistrict = subDistrict;
        return this;
    }

    public AddressInfoBuilder setDistrict(String district) {
        this.district = district;
        return this;
    }

    public AddressInfoBuilder setProvince(String province) {
        this.province = province;
        return this;
    }

    public AddressInfoBuilder setPostalCode(String postalCode) {
        this.postalCode = postalCode;
        return this;
    }

    public AddressInfoBuilder setRoom(String room) {
        this.room = room;
        return this;
    }

    public AddressInfoBuilder setFloor(String floor) {
        this.floor = floor;
        return this;
    }

    public AddressInfoBuilder setBuildingName(String buildingName) {
        this.buildingName = buildingName;
        return this;
    }

    public String build(Context context) {
        String address = "";
        if (houseNumber != null && !houseNumber.isEmpty()) {
            address += context.getString(R.string.my_ais_profile_your_address_house_number) + " " + houseNumber;
        }
        if (room != null && !room.isEmpty()) {
            address += room.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_room) + " " + room;
        }
//        if (floor != null && !floor.isEmpty()) {
//            address += floor.isEmpty() ? "" : ", ";
//            address += context.getString(R.string.my_ais_profile_your_address_floor) + " " + floor;
//        }
        if (buildingName != null && !buildingName.isEmpty()) {
            address += buildingName.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_building) + " " + buildingName;
        }
        if (alley != null && !alley.isEmpty()) {
            address += address.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_alley) + " " + alley;
        }
        if (road != null && !road.isEmpty()) {
            address += address.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_street) + " " + road;
        }
        if (subDistrict != null && !subDistrict.isEmpty()) {
            address += address.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_sub_district) + " " + subDistrict;
        }
        if (district != null && !district.isEmpty()) {
            address += address.isEmpty() ? "" : ", ";
            address += context.getString(R.string.my_ais_profile_your_address_district) + " " + district;
        }
        if (province != null && !province.isEmpty()) {
            address += address.isEmpty() ? "" : ", ";
            String language = getAddressLanguage();
            String actualProvince = ProvinceConverter.getInstance().convertProvinceCodeToName(province, language);
            address += context.getString(R.string.my_ais_profile_your_address_province) + " " + actualProvince;
        }
        if (postalCode != null && !postalCode.isEmpty()) {
            address += address.isEmpty() ? "" : " ";
            address += context.getString(R.string.my_ais_profile_your_address_postal_code) + " " + postalCode;
        }
        if (address.isEmpty()) {
            address = "-";
        }
        return address;
    }

    private String getAddressLanguage() {
        if (alley != null) {
            if (isContainNonEnglishLetter(alley)) {
                return "th";
            }
        }
        if (road != null) {
            if (isContainNonEnglishLetter(road)) {
                return "th";
            }
        }
        if (subDistrict != null) {
            if (isContainNonEnglishLetter(subDistrict)) {
                return "th";
            }
        }
        if (district != null) {
            if (isContainNonEnglishLetter(district)) {
                return "th";
            }
        }
        return "en";
    }

    private boolean isContainNonEnglishLetter(String text) {
        return text.matches("^(?![a-zA-Z0-9\\s]).*$");
    }
}
