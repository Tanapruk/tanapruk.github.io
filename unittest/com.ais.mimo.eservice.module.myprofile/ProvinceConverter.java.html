    private static ProvinceConverter converter;

    public static ProvinceConverter getInstance() {
        if (converter == null) {
            converter = new ProvinceConverter();
        }
        return converter;
    }

    public String convertProvinceCodeToName(String provinceCode, String language) {
        try {
            Province province = Province.valueOf(provinceCode.toUpperCase());
            if ("th".equals(language.toLowerCase())) {
                return province.getNameTh();
            } else {
                return province.getNameEn();
            }
        } catch (IllegalArgumentException ignored) {
        }
        return provinceCode;
    }
