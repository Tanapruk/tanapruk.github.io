package com.nextzy.myais.common.mvp;

import com.nextzy.myais.common.delegate.ErrorReceiver;

import java.util.List;

/**
 * Created by trusttanapruk on 9/15/2016.
 */
public abstract class NextzyActivityPresenter<CV extends NextzyActivityContractorView> extends NextzyPresenter<CV> implements NextzyActivityContractorPresenter {

    protected NextzyActivityPresenter(CV view) {
        super(view);
    }

    private ErrorReceiver mErrorReceiver = new ErrorReceiver(this, enableErrorReceiver());

    protected boolean enableErrorReceiver() {
        return true;
    }


    @Override
    public void onSessionExpired(Throwable throwable, List<String> serviceNameList) {
        getView().showSessionExpireDialog();
    }

    @Override
    public void onServiceUnavailable(Throwable throwable, List<String> serviceNameList) {
        getView().showServiceUnavailableDialog(false);
    }

    @Override
    public void onNoInternetConnection(Throwable throwable, List<String> serviceNameList) {
        getView().showNoInternetConnection();
    }

    @Override
    public void onUnHandledAisError(Throwable throwable, List<String> serviceNameList) {
        getView().showUnhandledAisError(throwable.getMessage());
    }

    @Override
    public void onBadRequestException(Throwable throwable, List<String> serviceNameList) {
        getView().showServiceUnavailableDialog(false);
    }

    /**
     * when throwable is specific to the service inside that activity e.g., MSISDNException
     * It will be here.
     *
     * @param throwable
     */
    @Override
    public void onNonGlobalException(Throwable throwable, List<String> serviceNameList) {

    }

    @Override
    public void onViewStart() {
        mErrorReceiver.start();
    }

    @Override
    public void onViewStop() {
        mErrorReceiver.stop();
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }
}
