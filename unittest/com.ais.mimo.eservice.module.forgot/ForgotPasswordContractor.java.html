package com.ais.mimo.eservice.module.forgot;

import com.nextzy.myais.common.mvp.NextzyActivityContractorPresenter;
import com.nextzy.myais.common.mvp.NextzyActivityContractorView;

/**
 * Created by Akexorcist on 10/13/2016 AD.
 */

public class ForgotPasswordContractor {
    public interface Presenter extends NextzyActivityContractorPresenter {
        void onRequestOtpButtonClick(String mobileNumber);

        void onRequestOtpButtonClick(String mobileNumber, String captcha);

        void onConfirmNewPasswordButtonClick(String mobileNumber, String otp, String newPassword, String confirmPassword);
    }

    public interface View extends NextzyActivityContractorView<Presenter> {
        void enableRequestOtp();

        void disableRequestOtp();

        void enableCaptcha();

        void disableCaptcha();

        void enableVerifyPassword();

        void disableVerifyPassword();

        void setCaptchaImage(String base64);

        void showMobileNumberInvalid();

        void showCaptchaInvalid();

        void showOtpPasswordInvalid();

        void showPasswordInvalid();

        void showPasswordFormatInvalid();

        void showConfirmPasswordInvalid();

        void showChangePasswordSuccessful(String message);

        void showOtpAndNewPassword();

        void hideOtpAndNewPassword();

        void showCaptcha();

        void hideCaptcha();

        String getMobileNumber();

        String getNewPassword();

        String getConfirmPassword();

        void setOtpPasswordViaSmsDescription();

        void setOtpPasswordViaEmailDescription();

        boolean isCaptchaShowing();

        void requestOtpPasswordFocus();

        void requestCaptchaFocus();
    }
}
