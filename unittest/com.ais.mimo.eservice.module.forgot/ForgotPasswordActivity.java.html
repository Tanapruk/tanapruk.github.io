package com.ais.mimo.eservice.module.forgot;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ImageView;
import android.widget.LinearLayout;
import android.widget.TextView;

import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.NextzyBitmapUtility;
import com.nextzy.myais.common.widget.dialog.AisDialog;
import com.nextzy.myais.common.widget.dialog.listener.SimpleDialogClickListener;
import com.nextzy.myais.common.widget.toolbar.simple.MyAisSimpleToolbar;
import com.nextzy.myais.myais.R;

/**
 * Created by Akexorcist on 10/13/2016 AD.
 */

public class ForgotPasswordActivity extends NextzyMvpActivity<ForgotPasswordContractor.Presenter> implements ForgotPasswordContractor.View, View.OnClickListener {
    public static final String KEY_ACCOUNT_NUMBER = "key_account_number";

    private MyAisSimpleToolbar tbHeader;
    private EditText etMobileNumber;
    private EditText etCaptcha;
    private EditText etOtpPassword;
    private EditText etNewPassword;
    private EditText etConfirmPassword;
    private TextView etOtpPasswordDescription;
    private ImageView ivCaptcha;
    private ImageView ivPasswordInfo;
    private LinearLayout layoutPasswordContainer;
    private LinearLayout layoutCaptchaContainer;
    private Button btnGetOtp;
    private Button btnCancel;
    private Button btnOk;

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_forgot_password;
    }

    @Override
    protected void createPresenter() {
        ForgotPasswordPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        tbHeader = (MyAisSimpleToolbar) findViewById(R.id.my_ais_forgot_password_tb_header);
        etMobileNumber = (EditText) findViewById(R.id.my_ais_forgot_password_et_mobile_number);
        etCaptcha = (EditText) findViewById(R.id.my_ais_forgot_password_et_captcha);
        etOtpPassword = (EditText) findViewById(R.id.my_ais_forgot_password_et_otp_password);
        etNewPassword = (EditText) findViewById(R.id.my_ais_forgot_password_et_new_password);
        etConfirmPassword = (EditText) findViewById(R.id.my_ais_forgot_password_et_confirm_new_password);
        etOtpPasswordDescription = (TextView) findViewById(R.id.my_ais_forgot_password_et_otp_password_description);
        ivCaptcha = (ImageView) findViewById(R.id.my_ais_forgot_password_iv_captcha);
        ivPasswordInfo = (ImageView) findViewById(R.id.my_ais_forgot_password_iv_password_info);
        layoutPasswordContainer = (LinearLayout) findViewById(R.id.my_ais_forgot_password_layout_new_password_container);
        layoutCaptchaContainer = (LinearLayout) findViewById(R.id.my_ais_forgot_password_layout_captcha_container);
        btnGetOtp = (Button) findViewById(R.id.my_ais_forgot_password_btn_request_otp);
        btnCancel = (Button) findViewById(R.id.my_ais_forgot_password_btn_cancel);
        btnOk = (Button) findViewById(R.id.my_ais_forgot_password_btn_ok);
    }

    @Override
    public void setupView() {
        tbHeader.setTitle(R.string.my_ais_forgot_password_title);
        tbHeader.setNavigationOnClickListener(this);
        btnGetOtp.setOnClickListener(this);
        btnCancel.setOnClickListener(this);
        ivPasswordInfo.setOnClickListener(this);
        btnOk.setOnClickListener(this);
        enableRequestOtp();
        hideCaptcha();
        hideOtpAndNewPassword();
    }

    @Override
    public void restoreArgument(Bundle bundle) {
        String accountNumber = bundle.getString(KEY_ACCOUNT_NUMBER);
        etMobileNumber.setText(accountNumber);
    }

    @Override
    public void initialize() {
//        disableCaptcha();
//        disableVerifyPassword();
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    @Override
    public void onClick(View view) {
        if (view == tbHeader) {
            super.onBackPressed();
        } else if (view == btnGetOtp) {
            onGetOtpButtonClick();
        } else if (view == btnOk) {
            onOkButtonClick();
        } else if (view == btnCancel) {
            super.onBackPressed();
        } else if (view == ivPasswordInfo) {
            showPasswordInfoDialog();
        }
    }

    @Override
    public void enableRequestOtp() {
        etMobileNumber.setEnabled(true);
        btnGetOtp.setEnabled(true);
    }

    @Override
    public void disableRequestOtp() {
        etMobileNumber.setEnabled(false);
        btnGetOtp.setEnabled(false);
    }

    @Override
    public void enableCaptcha() {
        etCaptcha.setEnabled(true);
    }

    @Override
    public void disableCaptcha() {
        etCaptcha.setEnabled(false);
    }

    @Override
    public void enableVerifyPassword() {
        etOtpPassword.setEnabled(true);
        etNewPassword.setEnabled(true);
        etConfirmPassword.setEnabled(true);
        btnCancel.setEnabled(true);
        btnOk.setEnabled(true);
    }

    @Override
    public void disableVerifyPassword() {
        etOtpPassword.setEnabled(false);
        etNewPassword.setEnabled(false);
        etConfirmPassword.setEnabled(false);
        btnCancel.setEnabled(false);
        btnOk.setEnabled(false);
    }

    @Override
    public void setCaptchaImage(String base64) {
        ivCaptcha.setImageBitmap(NextzyBitmapUtility.decodeBase64Image(base64));
    }

    @Override
    public void showMobileNumberInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_mobile_number_invalid));
    }

    @Override
    public void showCaptchaInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_captcha_invalid));
    }

    @Override
    public void showOtpPasswordInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_otp_password_invalid));
    }

    @Override
    public void showPasswordInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_password_invalid));
    }

    @Override
    public void showPasswordFormatInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_password_format_invalid));
    }

    @Override
    public void showConfirmPasswordInvalid() {
        showSimpleDialog(getString(R.string.my_ais_forgot_password_confirm_password_invalid));
    }

    @Override
    public void showChangePasswordSuccessful(String message) {
        showSimpleDialog(message, data -> finish());
    }

    @Override
    public void showOtpAndNewPassword() {
        layoutPasswordContainer.setVisibility(View.VISIBLE);
    }

    @Override
    public void hideOtpAndNewPassword() {
        layoutPasswordContainer.setVisibility(View.GONE);
    }

    @Override
    public void showCaptcha() {
        layoutCaptchaContainer.setVisibility(View.VISIBLE);
    }

    @Override
    public void hideCaptcha() {
        layoutCaptchaContainer.setVisibility(View.GONE);
    }

    @Override
    public String getMobileNumber() {
        return etMobileNumber.getText().toString();
    }

    @Override
    public String getNewPassword() {
        return etNewPassword.getText().toString();
    }

    @Override
    public String getConfirmPassword() {
        return etConfirmPassword.getText().toString();
    }

    @Override
    public void setOtpPasswordViaSmsDescription() {
        etOtpPasswordDescription.setText(R.string.my_ais_forgot_password_verify_password_via_sms_description);
    }

    @Override
    public void setOtpPasswordViaEmailDescription() {
        etOtpPasswordDescription.setText(R.string.my_ais_forgot_password_verify_password_via_email_description);
    }

    @Override
    public boolean isCaptchaShowing() {
        return layoutCaptchaContainer.getVisibility() == View.VISIBLE;
    }

    @Override
    public void requestOtpPasswordFocus() {
        etOtpPassword.requestFocus();
        etOtpPassword.requestFocusFromTouch();
    }

    @Override
    public void requestCaptchaFocus() {
        etCaptcha.requestFocus();
        etCaptcha.requestFocusFromTouch();
    }

    private void showPasswordInfoDialog() {
        AisDialog.getInstance().showHtmlDialog(
                getSupportFragmentManager(),
                getString(R.string.my_ais_register_dialog_password_title),
                getString(R.string.my_ais_register_dialog_password_info),
                this
        );
    }

    private void onGetOtpButtonClick() {
        if (!isCaptchaShowing()) {
            String mobileNumber = etMobileNumber.getText().toString();
            getPresenter().onRequestOtpButtonClick(mobileNumber);
        } else {
            String mobileNumber = etMobileNumber.getText().toString();
            String captcha = etCaptcha.getText().toString();
            getPresenter().onRequestOtpButtonClick(mobileNumber, captcha);
        }
    }

    private void onOkButtonClick() {
        String mobileNumber = etMobileNumber.getText().toString();
        String password = etOtpPassword.getText().toString();
        String newPassword = etNewPassword.getText().toString();
        String confirmPassword = etConfirmPassword.getText().toString();
        getPresenter().onConfirmNewPasswordButtonClick(mobileNumber, password, newPassword, confirmPassword);
    }

    private void showSimpleDialog(String message) {
        showAlertDialogDismissOnClick(message);
    }

    private void showSimpleDialog(String message, SimpleDialogClickListener listener) {
        showAlertDialogCustomOnClick(message,listener);
    }
}
