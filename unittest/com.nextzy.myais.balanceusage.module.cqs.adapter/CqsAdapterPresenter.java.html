package com.nextzy.myais.balanceusage.module.cqs.adapter;

import com.nextzy.myais.common.adapter.item.NextzyBaseItem;
import com.nextzy.myais.common.mvp.NextzyPresenter;

import java.util.List;

/**
 * Created by TheKhaeng on 10/3/2016.
 */

public class CqsAdapterPresenter extends NextzyPresenter<CqsAdapterContractor.View> implements CqsAdapterContractor.Presenter{

    private List<NextzyBaseItem> cqsItems;

    public static CqsAdapterPresenter createPresenter(CqsAdapterContractor.View view ){
        return new CqsAdapterPresenter( view );
    }

    private CqsAdapterPresenter( CqsAdapterContractor.View view ){
        super( view );
    }


    @Override
    public void onViewCreate(){

    }

    @Override
    public void onViewDestroy(){

    }

    @Override
    public void onViewStart(){

    }

    @Override
    public void onViewStop(){

    }


    @Override
    public int getItemCount() {
        if(cqsItems == null) return 0;
        return cqsItems.size();
    }

    @Override
    public void setCqsItems(List<NextzyBaseItem> cqsItems ){
        this.cqsItems = cqsItems;
    }


    @Override
    public NextzyBaseItem getCqsItem(int pos) {
        return cqsItems.get(pos);
    }

    @Override
    public int getItemViewType( int position ){
        return cqsItems.get(position).getType();
    }

}
