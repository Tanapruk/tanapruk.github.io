package com.ais.mimo.eservice.helper.register.network.response;

import com.ais.mimo.eservice.helper.register.network.response.data.RegisterVerifyIdentityData;
import com.nextzy.myais.common.config.mobile.MobileUtility;

import lombok.EqualsAndHashCode;
import lombok.ToString;

/**
 * Created by TheKhaeng on 10/11/2016.
 */

@ToString
@EqualsAndHashCode( callSuper = false )
public class RegisterVerifyIdentityResult extends RegisterGenericResult<RegisterVerifyIdentityData>{

    public boolean isPostpaid() {
        if (getData() != null) {
            String nType = getData().getNtype();
            if (isNTypeIsPostpaid(nType)) {
                return true;
            }
        }
        return false;
    }


    public boolean isPrepaid() {
        if (getData() != null) {
            String nType = getData().getNtype();
            if (isNTypeIsPrepaid(nType)) {
                return true;
            }
        }
        return false;
    }


    public boolean isFibre() {
        if (getData() != null) {
            String nType = getData().getNtype();
            if (isNTypeIsFbb(nType)) {
                return true;
            }
        }
        return false;
    }

    private boolean isNTypeIsPostpaid(String nType) {
        return MobileUtility.getInstance().isNTypeIsPostpaid(nType);
    }

    private boolean isNTypeIsPrepaid(String nType) {
        return MobileUtility.getInstance().isNTypeIsPrepaid(nType);
    }

    private boolean isNTypeIsFbb(String nType) {
        return MobileUtility.getInstance().isNTypeIsFbb(nType);
    }
}
