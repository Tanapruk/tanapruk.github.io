package com.ais.mimo.eservice.helper.register.network.response;

import com.ais.mimo.eservice.helper.register.network.response.data.RegisterVerifyMyAisData;
import com.nextzy.myais.common.network.model.GenericResult;

import lombok.EqualsAndHashCode;
import lombok.ToString;

/**
 * Created by TheKhaeng on 10/11/2016.
 */

@ToString
@EqualsAndHashCode( callSuper = false )
public class RegisterVerifyMyAisResult extends GenericResult<RegisterVerifyMyAisData>{

    private boolean isMyAis(String yOrN) {
        return hasData() && yOrN.equalsIgnoreCase(getData().getIsMyAIS());
    }

    private boolean isMaster(String yOrN) {
        return hasData() && yOrN.equalsIgnoreCase(getData().getIsMaster());
    }

    private boolean isMyAisNo() {
        return isMyAis("N");
    }

    private boolean isMasterNo() {
        return isMaster("N");
    }

    public boolean isCanRequestVerifyIdentity(){
        return isMasterNo() && isMyAisNo();
    }
}
