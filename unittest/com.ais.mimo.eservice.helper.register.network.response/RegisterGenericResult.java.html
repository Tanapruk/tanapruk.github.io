package com.ais.mimo.eservice.helper.register.network.response;

import com.nextzy.myais.common.network.model.GenericData;
import com.nextzy.myais.common.network.model.GenericResult;

/**
 * Created by TheKhaeng on 10/31/2016.
 */

public abstract class RegisterGenericResult<D extends GenericData> extends GenericResult<D> {
    public static final String SSBE309 = "SSBE309";

    public boolean isSSBE309() {
        return hasData() && SSBE309.equalsIgnoreCase(getResultCode());
    }

    public boolean isCanNextStep(){
        return isSuccess();
    }
}
