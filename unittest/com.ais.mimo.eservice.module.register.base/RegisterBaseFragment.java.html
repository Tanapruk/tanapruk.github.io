package com.ais.mimo.eservice.module.register.base;

import android.os.Handler;
import android.support.v4.widget.NestedScrollView;
import android.view.View;

import com.ais.mimo.eservice.helper.register.network.RegisterApi;
import com.ais.mimo.eservice.module.register.main.RegisterActivity;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.mvp.NextzyMvpFragment;
import com.nextzy.myais.common.utility.NextzyAndroidUtility;
import com.nextzy.myais.myais.R;

/**
 * Created by TheKhaeng on 10/15/2016.
 */

public abstract class RegisterBaseFragment<CP extends RegisterBaseMvpContractorPresenter>
        extends NextzyMvpFragment<CP> {

    public static final String KEY_MOBILE_NUMBER = "key_mobile_number";
    public static final String KEY_COME_FROM_TAG = "key_COME_FROM";
    public static final String KEY_ID_CARD = "key_id_card";
    public static final String KEY_EMAIL = "key_email";
    public static final String KEY_PASS = "key_pass";
    public static final String KEY_REPASS = "key_repass";

    private RegisterApi regisApi;

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">
    @Override
    public void onResume() {
        super.onResume();
        getPresenter().registerRx();
    }

    @Override
    public void onPause() {
        super.onPause();
        getPresenter().unregisterRx();
    }
    //</editor-fold>


    public void setIndexIndicator(int index) {
        if (getActivity() instanceof RegisterActivity) {
            ((RegisterActivity) getActivity()).setIndexIndicator(index);
        }
    }

    public void hindIndicator() {
        if (getActivity() instanceof RegisterActivity) {
            ((RegisterActivity) getActivity()).hindIndicator();
        }
    }

    public void setBackgroundWhite() {
        if (getActivity() instanceof RegisterActivity) {
            ((RegisterActivity) getActivity()).setBackground(R.color.my_ais_white);
        }
    }

    public void setBackgroundLightGray() {
        if (getActivity() instanceof RegisterActivity) {
            ((RegisterActivity) getActivity()).setBackground(R.color.my_ais_extra_light_gray);
        }
    }

    public void showLoadingNoFade() {
        showLoadingDialog();
    }

    public void dismissLoadingNoFade() {
        final Handler handler = new Handler();
        handler.postDelayed(this::dismissDialog, 400);
    }

    public void dismissLoadingNoFade(int delay) {
        final Handler handler = new Handler();
        handler.postDelayed(this::dismissDialog, delay);
    }

    public void clickCancel() {
        getActivity().onBackPressed();
    }

    public void setupClickHideKeyboard(View v) {
        if (v != null) {
            v.setOnClickListener(
                    view -> hindKeyboard());
        }
    }


    public void showResponseException() {
        if (getActivity() instanceof NextzyMvpActivity) {
            ((NextzyMvpActivity) getActivity())
                    .showAlertDialogDismissOnClick(R.string.my_ais_server_unavailable);
        }
    }

    public void showResponseException(String resultDescription) {
        if (getActivity() instanceof NextzyMvpActivity) {
            ((NextzyMvpActivity) getActivity())
                    .showAlertDialogDismissOnClick(resultDescription);
        }
    }


    public void scrollToEnd(NestedScrollView scrollView, int delay) {
        if (getActivity() instanceof NextzyMvpActivity) {
            final Handler handler = new Handler();
            handler.postDelayed(() -> {
                scrollView.fullScroll(View.FOCUS_DOWN);
            }, delay);
        }
    }

    public void hindKeyboard() {
        NextzyAndroidUtility.getInstance().hideKeyboard(getActivity());
    }


    public String getArgumentsStringMobileNumber(){
        return getArguments().getString(KEY_MOBILE_NUMBER);
    }

    public String getArgumentsStringIdCard(){
        return getArguments().getString(KEY_ID_CARD);
    }

    public String getArgumentsStringEmail(){
        return getArguments().getString(KEY_EMAIL);
    }

    public String getArgumentsStringPassword(){
        return getArguments().getString(KEY_PASS);
    }
}
