package com.ais.mimo.eservice.module.setting.activity;

import android.os.Bundle;
import android.support.v7.widget.Toolbar;
import android.view.View;

import com.ais.mimo.eservice.module.setting.fragment.main.MainSettingFragment;
import com.ais.mimo.eservice.module.setting.fragment.passcode.PasscodeSettingFragment;
import com.nextzy.myais.common.delegate.FragmentHelperDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.ais.mimo.eservice.module.setting.changePasscode.ChangePasscodeActivity;
import com.nextzy.myais.myais.R;

import static com.ais.mimo.eservice.module.setting.changePasscode.ChangePasscodeActivity.KEY_OLD_PASSCODE;

public class SettingActivity extends NextzyMvpActivity<SettingContractor.Presenter>
        implements SettingContractor.View,
        View.OnClickListener {

    private FragmentHelperDelegate fragmentHelperDelegate = new FragmentHelperDelegate(this);
    private Toolbar tbHeader;

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_setting;
    }

    @Override
    protected void createPresenter() {
        SettingPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        fragmentHelperDelegate.bind(R.id.contentContainer);
        tbHeader = (Toolbar) findViewById(R.id.my_ais_library_simple_toolbar_tb_header);
    }

    @Override
    public void setupView() {
        setToolbarTitle(getString(R.string.my_ais_setting_title));
        tbHeader.setNavigationOnClickListener(this);
    }

    public void setToolbarTitle(String title){
        tbHeader.setTitle(title);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        fragmentHelperDelegate.addFragment(MainSettingFragment.newInstance(), false);
    }


    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {
    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    @Override
    public void onClick(View view) {
        if (view == tbHeader) {
            onBackPressed();
        }
    }


    @Override
    public void openPasscodeSettingFragment() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                PasscodeSettingFragment.newInstance(),
                null);
    }

    @Override
    public void openChangePasscodeFragment(String oldPasscode) {
        Bundle bundle = new Bundle();
        bundle.putString(KEY_OLD_PASSCODE, oldPasscode);
        openActivity(ChangePasscodeActivity.class, bundle);
    }

    @Override
    public void openFirstNewPasscodeActivity() {
        openActivity(ChangePasscodeActivity.class);
    }
}
