package com.ais.mimo.eservice.module.setting.activity;

import com.ais.mimo.eservice.helper.term.network.TermManager;
import com.ais.mimo.eservice.module.setting.event.GoToChangePasscodeFragmentEvent;
import com.ais.mimo.eservice.module.setting.event.GoToFirstNewPasscodeFragmentEvent;
import com.ais.mimo.eservice.module.setting.event.GoToPasscodeSettingFragmentEvent;
import com.ais.mimo.eservice.module.setting.event.SetLanguageEvent;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.eventlog.EventLogBlob;
import com.nextzy.myais.common.eventlog.EventLogParam;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by Akexorcist on 9/7/2016 AD.
 */

public class SettingPresenter extends NextzyActivityPresenter<SettingContractor.View> implements SettingContractor.Presenter {

    public static SettingPresenter createPresenter(SettingContractor.View view) {
        return new SettingPresenter(view);
    }

    public SettingPresenter(SettingContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
        EventLogBlob.getInstance().addEventLog(
                TermManager.SERVICE_NAME_BY_MOBILE_TERM_OF_CONDITION,
                new EventLogParam("APP_SETTING_TERMS_CONDITION", "5009", "5009"));
    }

    @Override
    public void onViewStop() {
        // Do nothing.
    }


    @Subscribe
    public void onSetLanguageEvent(SetLanguageEvent event) {
        getView().setLanguage(event.getLanguage());
    }

    @Subscribe
    public void onGoToPasscodeSettingFragmentEvent(GoToPasscodeSettingFragmentEvent event) {
        getView().openPasscodeSettingFragment();
    }

    @Subscribe
    public void onGoToChangePasscodeFragmentEvent(GoToChangePasscodeFragmentEvent event) {
        getView().openChangePasscodeFragment(event.getOldPasscode());
    }

    @Subscribe
    public void onGoToFirstNewPasscodeActivityEvent(GoToFirstNewPasscodeFragmentEvent event) {
        getView().openFirstNewPasscodeActivity();
    }

}
