package com.ais.mimo.eservice.helper.login.network.exception

import com.nextzy.myais.common.network.exception.SessionExpireException

/**
 * Created by trusttanapruk on 8/18/2016.
 */
object LoginExceptionValidator {
    fun Throwable.isHandled(): Boolean {
        when (this) {
            is MSISDNFailException -> return true
            is GetOtpNotAisNumberException -> return true
            is LoginWithOtpWrong3Exception -> return true
            is LoginWithOtpWrongException -> return true
            is MobileContactNotAisNumberException -> return true
            is SecondaryOtpExistsException -> return true
            is SessionExpireException -> return true
            else -> return false
        }
    }
}
