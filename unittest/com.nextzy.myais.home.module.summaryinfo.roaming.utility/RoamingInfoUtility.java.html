package com.nextzy.myais.home.module.summaryinfo.roaming.utility;

/**
 * Created by Akexorcist on 12/10/2016 AD.
 */

public class RoamingInfoUtility {
    private static RoamingInfoUtility roamingInfoUtility;

    public static RoamingInfoUtility getInstance() {
        if (roamingInfoUtility == null) {
            roamingInfoUtility = new RoamingInfoUtility();
        }
        return roamingInfoUtility;
    }

    public boolean isContainWithBasicRoamingInfo(String packageName, String startDate, String endDate) {
        return packageName != null && startDate != null && endDate != null;
    }

    public boolean isContainWithExtraRoamingInfo(String remaining, String max, String unit) {
        return remaining != null && max != null && unit != null;
    }

    public boolean isRoamingPackage(String packageName) {
        return packageName != null && (packageName.contains("Roaming") ||
                packageName.contains("โรมมิ่ง") ||
                packageName.contains("ข้ามแดน") ||
                packageName.equalsIgnoreCase("IR Package"));
    }
}
