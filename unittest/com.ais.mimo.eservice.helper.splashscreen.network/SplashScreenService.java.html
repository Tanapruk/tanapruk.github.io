package com.ais.mimo.eservice.helper.splashscreen.network;

import com.ais.mimo.eservice.helper.splashscreen.network.operator.IrResponseConverterFactory;
import com.nextzy.myais.common.network.worker.GenericApiService;

import retrofit2.Converter;

/**
 * Created by TheKhaeng on 11/17/2016.
 */

public class SplashScreenService extends GenericApiService<SplashScreenApi> {

    public static SplashScreenService newInstance(String baseUrl) {
        SplashScreenService service = new SplashScreenService();
        service.setBaseUrl(baseUrl);
        return service;
    }

    private SplashScreenService() {

    }

    @Override
    protected Class<SplashScreenApi> getApiClassType() {
        return SplashScreenApi.class;
    }

    @Override
    protected Converter.Factory addConverter() {
        return IrResponseConverterFactory.create();
    }

}
