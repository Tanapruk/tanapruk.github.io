package com.ais.mimo.eservice.module.myprofile.fragment.changeContact;

import android.os.Bundle;
import android.support.annotation.Nullable;
import android.view.View;

import com.ais.mimo.eservice.module.register.widget.MyAisRegisterEditText;
import com.nextzy.myais.common.delegate.FormatUtility;
import com.nextzy.myais.common.mvp.NextzyMvpFragment;
import com.nextzy.myais.common.widget.AISButton;
import com.nextzy.myais.common.widget.dialog.AisDialog;
import com.nextzy.myais.common.widget.dialog.listener.ChoiceDialogClickListener;
import com.nextzy.myais.myais.R;

import static com.nextzy.myais.common.delegate.FormatUtility.FORMAT_MOBILE_NUMBER;

/**
 * Created by TheKhaeng on 11/16/2016.
 */

public class MyProfileChangeContactNumberFragment extends NextzyMvpFragment<MyProfileChangeContactNumberContractor.Presenter> implements MyProfileChangeContactNumberContractor.View {
    public final static String TAG = MyProfileChangeContactNumberFragment.class.getSimpleName();
    public static final String KEY_OLD_CONTACT_NUMBER = "key_old_contact_number";
    private MyAisRegisterEditText edtCurrentNumber;
    private MyAisRegisterEditText edtNewNumber;
    private AISButton btnOk;
    private AISButton btnCancel;

    public MyProfileChangeContactNumberFragment() {
        super();
    }

    public static MyProfileChangeContactNumberFragment newInstance(Bundle data) {
        MyProfileChangeContactNumberFragment fragment = new MyProfileChangeContactNumberFragment();
        fragment.setArguments(data);
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter() {
        MyProfileChangeContactNumberPresenter.createPresenter(this);
    }

    @Override
    public void restoreArgument(Bundle bundle) {
    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView(View view) {
        edtCurrentNumber = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_profile_edt_current_number);
        edtNewNumber = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_profile_edt_new_number);
        btnOk = (AISButton) view.findViewById(R.id.my_ais_profile_btn_ok);
        btnCancel = (AISButton) view.findViewById(R.id.my_ais_profile_btn_cancel);
    }

    @Override
    public void setupView() {
        edtCurrentNumber.setText(getOldNumber());
        edtCurrentNumber.setEditable(false);

        btnOk.setOnClickListener(v -> clickOk());
        btnCancel.setOnClickListener(v -> clickCancel());
    }

    @Override
    public void initialize() {
    }

    @Override
    public void restoreView() {
    }

    /**
     * onCreateView
     **/
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_profile_fragment_edit_contact_number;
    }

    /**
     * State
     **/
    @Override
    public void saveInstanceState(Bundle outState) {
        // Save Instance State here
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        // Restore instance state here
    }

    //</editor-fold>

    @Override
    public void showChangeContactNumberSuccessDialog(){
        showAlertDialogCustomOnClick(
                R.string.my_ais_profile_edit_contact_change_success_dialog,
                data -> finishFragment());
    }

    @Override
    public void showDefaultNewNumberEdt() {
        edtNewNumber.showDefault();
    }

    @Override
    public void showInvalidateMobileNumberDialog() {
        edtNewNumber.showError();
        String error = getString(R.string.my_ais_profile_edit_error_mobile_format);
        showAlertDialogDismissOnClick(error);
    }

    @Override
    public void showRedundantMobileNumberDialog() {
        edtNewNumber.showError();
        String error = getString(R.string.my_ais_profile_edit_error_redundant);
        showAlertDialogDismissOnClick(error);
    }

    @Override
    public void showServiceUnavailable(String outErrorMessage) {
        showAlertDialogDismissOnClick(outErrorMessage);
    }

    @Override
    public void showServiceUnavailable() {
        showAlertDialogDismissOnClick(getString(R.string.my_ais_server_unavailable));
    }

    @Override
    public void finishFragment() {
        getActivity().onBackPressed();
    }

    public String getOldNumber() {
        String number = getArguments().getString(KEY_OLD_CONTACT_NUMBER);
        if (number == null)
            throw new NullPointerException("Must send old number to this fragment before.");
        return number;
    }

    private void clickOk() {
        if( getPresenter().validateMobileNumber(getOldNumber(), edtNewNumber.getStringFromEditText())){
            showConfirmToChangeContactNumberDialog();
        }
    }

    private void clickCancel() {
        getActivity().onBackPressed();
    }

    private void showConfirmToChangeContactNumberDialog() {
        AisDialog.getInstance().showChoiceDialog(
                getFragmentManager(),
                null,
                String.format(
                        getString(R.string.my_ais_profile_edit_contact_confirm_dialog),
                        FormatUtility.getInstance().getStringWithFormat(edtNewNumber.getStringFromEditText(), FORMAT_MOBILE_NUMBER)),
                getString(R.string.ok),
                getString(R.string.cancel),
                true,
                new ChoiceDialogClickListener() {
                    @Override
                    public void onDialogPositiveButtonClick(@Nullable Bundle data) {
                        getPresenter().requestUpdateContactMobileNumber(getOldNumber(), edtNewNumber.getStringFromEditText());
                    }

                    @Override
                    public void onDialogNegativeButtonClick(@Nullable Bundle data) {
                        dismissDialog();
                    }
                }
        );
    }
}
