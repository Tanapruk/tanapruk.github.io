package com.ais.mimo.eservice.module.myprofile.activity;

import com.ais.mimo.eservice.module.myprofile.event.GoToChangeContactNumberFragmentEvent;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class MyProfilePresenter extends NextzyActivityPresenter<MyProfileContractor.View> implements MyProfileContractor.Presenter {

    public MyProfilePresenter(MyProfileContractor.View view) {
        super(view);
    }

    public static MyProfilePresenter createPresenter(MyProfileContractor.View view) {
        return new MyProfilePresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
    }


    @Subscribe
    public void onGoToEditContactMobileNumberFragmentEvent(GoToChangeContactNumberFragmentEvent event) {
        getView().openEditContactMobileNumberFragment(event.getOldNumber());
    }
}
