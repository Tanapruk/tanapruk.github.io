package com.ais.mimo.eservice.module.myprofile.activity;


import android.os.Bundle;
import android.support.v7.widget.Toolbar;
import android.view.View;

import com.ais.mimo.eservice.module.myprofile.fragment.main.MyProfileMainFragment;
import com.ais.mimo.eservice.module.myprofile.fragment.changeContact.MyProfileChangeContactNumberFragment;
import com.nextzy.myais.common.delegate.FragmentHelperDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.myais.R;

import static com.ais.mimo.eservice.module.myprofile.fragment.changeContact.MyProfileChangeContactNumberFragment.KEY_OLD_CONTACT_NUMBER;

public class MyProfileActivity extends NextzyMvpActivity<MyProfileContractor.Presenter>
        implements MyProfileContractor.View, View.OnClickListener {
    private final static String TAG = MyProfileActivity.class.getSimpleName();
    private Toolbar tbHeader;
    private FragmentHelperDelegate fragmentHelperDelegate = new FragmentHelperDelegate(this);

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_my_ais_profile;
    }

    @Override
    protected void createPresenter() {
        MyProfilePresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        fragmentHelperDelegate.bind(R.id.contentContainer);
        tbHeader = (Toolbar) findViewById(R.id.my_ais_profile_tb_header);
    }

    @Override
    public void setupView() {
        tbHeader.setNavigationOnClickListener(this);
        tbHeader.setTitle(R.string.my_ais_profile_title);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        getSupportFragmentManager().beginTransaction()
                .add(R.id.contentContainer,
                        MyProfileMainFragment.newInstance(),
                        MyProfileMainFragment.TAG)
                .commit();
    }

    @Override
    public void onClick(View view) {
        if (view == tbHeader) {
            finish();
        }
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {
    }

    @Override
    public void onBackPressed() {
        super.onBackPressed();
    }

    @Override
    public void openEditContactMobileNumberFragment(String oldNumber) {
        Bundle bundle = new Bundle();
        bundle.putString(KEY_OLD_CONTACT_NUMBER, oldNumber);
        fragmentHelperDelegate.openFragmentSlideInOut(
                MyProfileChangeContactNumberFragment.newInstance(bundle),
                MyProfileChangeContactNumberFragment.TAG);
    }
    //</editor-fold>
}
