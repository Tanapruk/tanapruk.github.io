package com.ais.mimo.eservice.module.register.step2.model;

import com.ais.mimo.eservice.helper.register.network.response.RegisterMyNumberResult;
import com.ais.mimo.eservice.helper.register.network.response.RegisterVerifyOtpResult;

import lombok.Getter;

/**
 * Created by TheKhaeng on 10/15/2016.
 */

public class RegisterChangeMainNumberModel {

    @Getter
    private final RegisterVerifyOtpResult verifyOtpResult;
    @Getter
    private final RegisterMyNumberResult myNumberResult;

    public RegisterChangeMainNumberModel(RegisterVerifyOtpResult verifyOtpResult,
                                         RegisterMyNumberResult registerMyNumberResult) {
        this.verifyOtpResult = verifyOtpResult;
        this.myNumberResult = registerMyNumberResult;
    }


    public boolean isSSBE309() {
        return verifyOtpResult.isSSBE309() || myNumberResult.isSSBE309();
    }

    public String getErrorResultDescription(){
        if( !verifyOtpResult.isSuccess()){
            return verifyOtpResult.getResultDescription();
        }else if( !myNumberResult.isSuccess()){
            return myNumberResult.getResultDescription();
        }
        return "";
    }

    public boolean isCanNextStep(){
        return verifyOtpResult.isCanNextStep() && myNumberResult.isCanNextStep();
    }
}
