package com.nextzy.myais.service.module.iswop.activity;

import com.hwangjr.rxbus.RxBus;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class IswopActivityPresenter extends NextzyActivityPresenter<IswopActivityContractor.View> implements IswopActivityContractor.Presenter {

    protected IswopActivityPresenter( IswopActivityContractor.View view ){
        super( view );
    }

    public static IswopActivityPresenter createPresenter( IswopActivityContractor.View view) {
        return new IswopActivityPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart(){
        // for handle network error
        super.onViewStart();
    }

    @Override
    public void onViewStop(){
        super.onViewStop();
    }
}
