package com.ais.mimo.eservice.helper.pushnotification.adapter;

import android.support.v7.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ImageView;
import android.widget.TextView;

import com.ais.mimo.eservice.helper.pushnotification.MyAisNotificationUtility;
import com.ais.mimo.eservice.helper.pushnotification.event.GotoUrlSchemeEvent;
import com.nextzy.myais.common.database.model.NotificationItem;
import com.hwangjr.rxbus.RxBus;
import com.nextzy.myais.myais.R;

import java.util.List;

/**
 * Created by trusttanapruk on 11/23/2016.
 */

public class NotificationAdapter extends RecyclerView.Adapter<RecyclerView.ViewHolder> {

    public static final int VIEW_TYPE_NOTIFICATION_ITEM = 1;

    List<NotificationItem> mNotificationItemList;

    public NotificationAdapter(List<NotificationItem> notificationItemList) {
        mNotificationItemList = notificationItemList;
    }

    @Override
    public int getItemViewType(int position) {
        return VIEW_TYPE_NOTIFICATION_ITEM;
    }

    @Override
    public RecyclerView.ViewHolder onCreateViewHolder(ViewGroup parent, int viewType) {
        if (viewType == VIEW_TYPE_NOTIFICATION_ITEM) {
            View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.my_ais_view_notification, parent, false);
            return new NotificationItemViewHolder(view);
        }
        return null;
    }

    @Override
    public void onBindViewHolder(RecyclerView.ViewHolder holder, int position) {
        if (holder instanceof NotificationItemViewHolder) {
            NotificationItemViewHolder viewHolder = ((NotificationItemViewHolder) holder);
            NotificationItem notificationItem = mNotificationItemList.get(position);
            viewHolder.tvMessage.setText(notificationItem.getMessage());
            int iconResource = MyAisNotificationUtility.getInstance().getIconResource(notificationItem.getIcon());
            viewHolder.ivIcon.setImageResource(iconResource);

            viewHolder.layout.setOnClickListener(v -> {
                RxBus.get().post(new GotoUrlSchemeEvent(notificationItem.getDestinationPage()));
            });
        }

    }

    @Override
    public int getItemCount() {
        return mNotificationItemList.size();
    }

    public class NotificationItemViewHolder extends RecyclerView.ViewHolder {
        ImageView ivIcon;
        TextView tvMessage;
        View layout;


        public NotificationItemViewHolder(View view) {
            super(view);
            ivIcon = (ImageView) view.findViewById(R.id.my_ais_notification_iv_icon);
            tvMessage = (TextView) view.findViewById(R.id.my_ais_notification_tv_message);
            layout = (View) view.findViewById(R.id.my_ais_notification_layout_container);


        }
    }
}
