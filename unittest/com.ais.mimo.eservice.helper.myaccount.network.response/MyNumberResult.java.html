package com.ais.mimo.eservice.helper.myaccount.network.response;

import android.os.Parcel;
import android.os.Parcelable;

import com.nextzy.myais.common.network.model.GenericResult;

import java.util.ArrayList;

/**
 * Created by RoyleKoonlert on 3/1/2017 AD.
 */

public class MyNumberResult extends GenericResult<MyNumberData> implements Parcelable {

    @Override
    public int describeContents() {
        return 0;
    }

    @Override
    public void writeToParcel(Parcel dest, int flags) {
        dest.writeParcelable(this.data, flags);
        dest.writeList(this.dataList);
        dest.writeString(this.resultCode);
        dest.writeString(this.resultDescription);
        dest.writeString(this.developerMessage);
    }

    public MyNumberResult() {
    }

    protected MyNumberResult(Parcel in) {
        this.data = in.readParcelable(MyNumberData.class.getClassLoader());
        this.dataList = new ArrayList<>();
        in.readList(this.dataList, MyNumberData.class.getClassLoader());
        this.resultCode = in.readString();
        this.resultDescription = in.readString();
        this.developerMessage = in.readString();
    }

    public static final Parcelable.Creator<MyNumberResult> CREATOR = new Parcelable.Creator<MyNumberResult>() {
        @Override
        public MyNumberResult createFromParcel(Parcel source) {
            return new MyNumberResult(source);
        }

        @Override
        public MyNumberResult[] newArray(int size) {
            return new MyNumberResult[size];
        }
    };
}
