package com.ais.mimo.eservice.module.myaccount.addmobile.activity;

import com.ais.mimo.eservice.module.myaccount.event.ChangeToolbarTitleEvent;
import com.ais.mimo.eservice.module.myaccount.event.GoToEditAccountWithMobileNumberEvent;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class MyAccountAddNumberPresenter extends NextzyActivityPresenter<MyAccountAddNumberContractor.View> implements MyAccountAddNumberContractor.Presenter {

    protected MyAccountAddNumberPresenter(MyAccountAddNumberContractor.View view) {
        super(view);
    }

    public static MyAccountAddNumberPresenter createPresenter(MyAccountAddNumberContractor.View view) {
        return new MyAccountAddNumberPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {
        // for handle network error
        super.onViewStart();
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
    }
    //</editor-fold>

    @Subscribe
    public void onGoToEditAccountWithMobileNumberEvent(GoToEditAccountWithMobileNumberEvent event) {
        getView().openEditAccountFragment(event.getMobileNumber());
    }

    @Subscribe
    public void onChangeToolbarTitleEvent(ChangeToolbarTitleEvent event) {
        getView().setTitle(event.getTitle());
    }
}
