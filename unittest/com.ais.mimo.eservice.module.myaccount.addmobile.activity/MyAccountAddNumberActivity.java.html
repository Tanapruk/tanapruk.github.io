package com.ais.mimo.eservice.module.myaccount.addmobile.activity;

import android.os.Bundle;

import com.ais.mimo.eservice.module.myaccount.addmobile.fragment.MyAccountAddNumberFragment;
import com.ais.mimo.eservice.module.myaccount.fragment.account.MyAccountEditAccountFragment;
import com.nextzy.myais.common.delegate.FragmentHelperDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.NextzyAndroidUtility;
import com.nextzy.myais.common.widget.toolbar.simple.MyAisSimpleToolbar;
import com.nextzy.myais.myais.R;

import static com.ais.mimo.eservice.module.register.step4.RegisterStep4Fragment.EXTRA_DATA;

public class MyAccountAddNumberActivity extends NextzyMvpActivity<MyAccountAddNumberContractor.Presenter> implements MyAccountAddNumberContractor.View{
    private final static String TAG = MyAccountAddNumberActivity.class.getSimpleName();
    public static final String KEY_CHANGE_MOBILE_NUMBER = "key_change_mobile_number";
    private MyAisSimpleToolbar tbHeader;
    private FragmentHelperDelegate fragmentHelperDelegate = new FragmentHelperDelegate(this);

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView(){
        return R.layout.my_ais_account_activity_add_number;
    }

    @Override
    protected void createPresenter(){
        MyAccountAddNumberPresenter.createPresenter( this );
    }

    @Override
    public void bindView(){
        fragmentHelperDelegate.bind(R.id.contentContainer);
        tbHeader = (MyAisSimpleToolbar) findViewById( R.id.my_ais_library_simple_toolbar_tb_header );
    }

    @Override
    public void setupView(){
        tbHeader.setNavigationOnClickListener( v -> {
            if(v == tbHeader){
                onBackPressed();
            }
        } );
        tbHeader.setTitle(R.string.my_ais_register_add_mobile_number);
    }

    @Override
    public void restoreArgument( Bundle bundle ){

    }

    @Override
    public void initialize(){
        getSupportFragmentManager().beginTransaction()
                .add( R.id.contentContainer,
                        MyAccountAddNumberFragment.newInstance(
                                getIntent().getExtras().getBundle( EXTRA_DATA )),
                        MyAccountAddNumberFragment.TAG )
                .commit();
    }

    @Override
    public void restoreInstanceState( Bundle savedInstanceState ){
    }

    @Override
    public void restoreView(){
    }

    @Override
    public void saveInstanceState( Bundle outState ){
    }

    @Override
    public void onBackPressed() {
        NextzyAndroidUtility.getInstance().hideKeyboard(this);
        super.onBackPressed();
    }
    //</editor-fold>

    @Override
    public void openEditAccountFragment(String mobileNumber) {
        Bundle bundle = new Bundle();
        bundle.putString(KEY_CHANGE_MOBILE_NUMBER, mobileNumber);
        fragmentHelperDelegate.openFragmentSlideInOut(
                MyAccountEditAccountFragment.newInstance(bundle));
    }

    @Override
    public void setTitle(String title) {
        tbHeader.setTitle(title);
    }
}
