package com.ais.mimo.eservice.module.logout;

import android.os.Bundle;

import com.ais.mimo.eservice.module.login.activity.LoginActivity;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.mvp.webview.MyAisWebViewActivity;
import com.nextzy.myais.common.network.tool.NextzyWebkitCookieJar;
import com.nextzy.myais.common.utility.DataUpdatePreference;
import com.nextzy.myais.myais.R;

public class LogoutActivity extends NextzyMvpActivity<LogoutContractor.Presenter> implements LogoutContractor.View {

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_logout;
    }

    @Override
    protected void createPresenter() {
        LogoutPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {

    }

    @Override
    public void setupView() {

    }

    @Override
    protected void onPause() {
        super.onPause();
        overridePendingTransition(0, 0);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        showLoadingDialog();
        getPresenter().callService();
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }


    @Override
    public void openWebView(String url) {
        dismissDialog();
        Bundle bundle = new Bundle();
        openActivity(LogoutSurveyWebViewActivity.class, bundle);
        bundle.putString(MyAisWebViewActivity.KEY_URL, url);
        bundle.putString(MyAisWebViewActivity.KEY_TITLE, getString(R.string.my_ais_survey_title));
        openActivity(LogoutSurveyWebViewActivity.class, bundle, true, true);
        clearCookie();
    }

    @Override
    public void logoutSuccess() {
        dismissDialog();
        clearCookie();
        DataUpdatePreference.getInstance().clear(this);
        Bundle bundle = new Bundle();
        bundle.putBoolean(LoginActivity.SHOULD_BYPASS, false);
        openActivity(LoginActivity.class, bundle, true, true);
    }

    private void clearCookie() {
        NextzyWebkitCookieJar.getInstance().clearCookie(this);
    }
}
