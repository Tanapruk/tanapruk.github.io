public class RegisterWelcomePresenter extends RegisterBasePresenter<RegisterWelcomeContractor.View> implements RegisterWelcomeContractor.Presenter{

    }

    }

    public void unregisterRx(){
        RxBus.get().unregister( this );
    }

    public void clickLogin(){
        if(isLogin()){
            getView().logout();

            getView().finishActivity();
