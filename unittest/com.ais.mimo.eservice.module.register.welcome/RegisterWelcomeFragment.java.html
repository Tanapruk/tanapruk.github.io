package com.ais.mimo.eservice.module.register.welcome;

import android.os.Bundle;
import android.view.View;

import com.ais.mimo.eservice.module.logout.LogoutActivity;
import com.ais.mimo.eservice.module.register.base.RegisterBaseFragment;
import com.ais.mimo.eservice.module.register.widget.MyAisRegisterTickItemView;
import com.nextzy.myais.common.widget.AISButton;
import com.nextzy.myais.myais.R;


public class RegisterWelcomeFragment extends RegisterBaseFragment<RegisterWelcomeContractor.Presenter> implements RegisterWelcomeContractor.View{
    public final static String TAG = RegisterWelcomeFragment.class.getSimpleName();
    private AISButton btnLogin;
    private MyAisRegisterTickItemView vTick1;
    private MyAisRegisterTickItemView vTick2;
    private MyAisRegisterTickItemView vTick3;
    private MyAisRegisterTickItemView vTick4;


    public RegisterWelcomeFragment(){
        super();
    }

    public static RegisterWelcomeFragment newInstance( Bundle data ){
        RegisterWelcomeFragment fragment = new RegisterWelcomeFragment();
        fragment.setArguments( data );
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter(){
        RegisterWelcomePresenter.createPresenter( this );
    }

    @Override
    public void restoreArgument( Bundle bundle ){

    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView( View view ){
        btnLogin = (AISButton) view.findViewById( R.id.my_ais_register_btn_login);
        vTick1 = (MyAisRegisterTickItemView) view.findViewById( R.id.my_ais_tick_item_1 );
        vTick2 = (MyAisRegisterTickItemView) view.findViewById( R.id.my_ais_tick_item_2 );
        vTick3 = (MyAisRegisterTickItemView) view.findViewById( R.id.my_ais_tick_item_3 );
        vTick4 = (MyAisRegisterTickItemView) view.findViewById( R.id.my_ais_tick_item_4 );
    }

    @Override
    public void setupView(){
        hindIndicator();
        setBackgroundWhite();
        btnLogin.setOnClickListener( v -> clickLogin() );
        vTick1.animateTick( 500 );
        vTick2.animateTick( 700 );
        vTick3.animateTick( 900 );
        vTick4.animateTick( 1100 );
    }


    @Override
    public void initialize(){
    }


    @Override
    public void restoreView(){
    }

    /**
     * onCreateView
     **/
    @Override
    public int getLayoutView(){
        return R.layout.my_ais_register_fragment_welcome;
    }


    /**
     * State
     **/
    @Override
    public void saveInstanceState( Bundle outState ){
        // Save Instance State here
    }

    @Override
    public void restoreInstanceState( Bundle savedInstanceState ){
        // Restore instance state here
    }
    //</editor-fold>

    @Override
    public void finishActivity(){
        getActivity().finish();
    }

    @Override
    public void logout() {
        openActivity(LogoutActivity.class);
    }


    /********************/
    /** Private method **/
    /********************/
    //<editor-fold desc="Private function folding">
    private void clickLogin(){
        getPresenter().clickLogin();
    }
    //</editor-fold>

}
