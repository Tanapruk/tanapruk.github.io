package com.ais.mimo.eservice.module.term;

import android.os.Bundle;
import android.view.View;
import android.webkit.WebView;
import android.widget.Button;

import com.ais.mimo.eservice.module.main.activity.MainActivity;
import com.ais.mimo.eservice.module.tutorial.TutorialActivity;
import com.akexorcist.roundcornerprogressbar.RoundCornerProgressBar;
import com.nextzy.myais.common.delegate.WebViewClientDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.utility.DefaultPreference;
import com.nextzy.myais.myais.R;

import static com.ais.mimo.eservice.module.tutorial.TutorialActivity.TUTORIAL_VERSION;

/**
 * Created by trusttanapruk on 9/15/2016.
 */
public class TermActivity extends NextzyMvpActivity<TermContractor.Presenter> implements TermContractor.View {
    private WebView wvTermBody;
    View containerBtnAcceptBtnDecline;
    private Button btnOk;
    private Button btnAccept;
    private Button btnDecline;
    private RoundCornerProgressBar pbLoading;
    public static final String KEY_SHOW_ACCEPT_AND_DECLINE_BUTTON = "SHOW_ACCEPT_AND_DECLINE_BUTTON";
    boolean showAcceptAndDeclineButton = false;

    private WebViewClientDelegate webViewDelegate = new WebViewClientDelegate(this);

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_term;
    }

    @Override
    protected void createPresenter() {
        TermPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        wvTermBody = (WebView) findViewById(R.id.my_ais_wv_term_condition);
        btnOk = (Button) findViewById(R.id.my_ais_btn_ok_term_condition);
        containerBtnAcceptBtnDecline = findViewById(R.id.my_ais_container_btn_accept_decline_term_condition);
        btnAccept = (Button) findViewById(R.id.my_ais_btn_accept_term_condition);
        btnDecline = (Button) findViewById(R.id.my_ais_btn_decline_term_condition);
        pbLoading = (RoundCornerProgressBar) findViewById(R.id.my_ais_library_web_view_pb_loading);
    }

    @Override
    public void setupView() {
        wvTermBody.setScrollBarStyle(View.SCROLLBARS_INSIDE_OVERLAY);
        webViewDelegate.bind(wvTermBody, pbLoading);

        btnAccept.setOnClickListener(v -> getPresenter().sendTermOfConditionYes());
        btnDecline.setOnClickListener(v -> getPresenter().onUserDeclineTermAgreement());
        btnOk.setOnClickListener(v -> finish());

    }

    @Override
    public void restoreArgument(Bundle bundle) {
        showAcceptAndDeclineButton = bundle.getBoolean(KEY_SHOW_ACCEPT_AND_DECLINE_BUTTON, false);
    }

    @Override
    public void initialize() {
        if (showAcceptAndDeclineButton) {
            btnOk.setVisibility(View.GONE);
            animateFadeShow(containerBtnAcceptBtnDecline);
            getPresenter().callTermOfConditionAgreement();
        } else {
            getPresenter().callTermOfCondition();
        }
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    @Override
    public void loadWebView(String term) {
        wvTermBody.loadData(term, "text/html; charset=utf-8", "UTF-8");
    }

    @Override
    public void gotoNextActivity() {
        if (DefaultPreference.getInstance().isFirstOpenTutorialActivity(TUTORIAL_VERSION)) {
            openActivityFadeInOut(TutorialActivity.class, true, true);
        } else {
            openActivity(MainActivity.class, null, true, true);
            dismissDialog();
        }
    }
}
