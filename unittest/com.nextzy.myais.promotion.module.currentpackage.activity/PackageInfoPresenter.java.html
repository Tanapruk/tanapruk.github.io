package com.nextzy.myais.promotion.module.currentpackage.activity;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;
import com.nextzy.myais.promotion.module.currentpackage.event.GotoApplyPackageEvent;
import com.nextzy.myais.promotion.module.currentpackage.event.GotoCancelPackageEvent;
import com.nextzy.myais.promotion.module.currentpackage.event.GotoChangeMainPackageEvent;
import com.nextzy.myais.promotion.module.currentpackage.event.ShowCurrentPackageTermEvent;

import java.util.List;

/**
 * Created by trusttanapruk on 9/10/2016.
 */
public class PackageInfoPresenter extends NextzyActivityPresenter<PackageInfoContractor.View> implements PackageInfoContractor.Presenter {
    protected PackageInfoPresenter(PackageInfoContractor.View view) {
        super(view);
    }

    public static PackageInfoPresenter createPresenter(PackageInfoContractor.View view) {
        return new PackageInfoPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {
        super.onViewStart();

    }

    @Override
    public void onViewStop() {
        super.onViewStop();
    }


    @Subscribe
    public void onGotoChangeMainPackageEvent(GotoChangeMainPackageEvent event) {
        getView().gotoChangeMainPackage();
    }

    @Subscribe
    public void onGotoApplyPackageEvent(GotoApplyPackageEvent event) {
        getView().gotoApplyPackage();
    }

    @Subscribe
    public void onGotoCancelPackageEvent(GotoCancelPackageEvent event) {
        getView().gotoCancelPackage();
    }

    @Subscribe
    public void onShowTermCondition(ShowCurrentPackageTermEvent event) {
        getView().showTermCondition();
    }

    @Override
    public void onServiceUnavailable(Throwable throwable, List<String> serviceNameList) {
        getView().showServiceUnavailableDialog(true);
    }
}
