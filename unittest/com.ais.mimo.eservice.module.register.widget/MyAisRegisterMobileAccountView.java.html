package com.ais.mimo.eservice.module.register.widget;

public class MyAisRegisterMobileAccountView extends NextzyViewGroup {
    public static final int LINEAR_LAYOUT_MANAGER = 0;
    public static final int GRID_LAYOUT_MANAGER = 1;
    private AISButton btnAdd;
    private RegisterAdapter postAdapter;
    private RegisterAdapter preAdapter;
    private RegisterAdapter fbbAdapter;
    private RecyclerView rvPost;
    private RecyclerView rvPre;
    private RecyclerView rvFbb;
    private LinearLayout llPost;
    private LinearLayout llPre;
    private LinearLayout llFbb;
    private LinearLayout llBtnAdd;
    private AISDivider divider;
    private LinearLayout llBase;
    private AISTextView tvHeader;
    private AISTextView tvDescAddMobile;
    private LinearLayout llTitle;
    private boolean canShowTitle = false;

    public MyAisRegisterMobileAccountView(Context context) {
        super(context);
    }

    public MyAisRegisterMobileAccountView(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    public MyAisRegisterMobileAccountView(Context context, AttributeSet attrs, int defStyleAttr) {
        super(context, attrs, defStyleAttr);
    }

    @TargetApi(Build.VERSION_CODES.LOLLIPOP)
    public MyAisRegisterMobileAccountView(Context context, AttributeSet attrs, int defStyleAttr, int defStyleRes) {
        super(context, attrs, defStyleAttr, defStyleRes);
    }

    @Override
    protected int getLayoutRes() {
        return R.layout.my_ais_register_view_mobile_number_account;
    }

    @Override
    protected void bindView() {
        llBase = (LinearLayout) findViewById(R.id.my_ais_register_ll_base);
        llPost = (LinearLayout) findViewById(R.id.my_ais_regis_ll_post_group);
        llTitle = (LinearLayout) findViewById(R.id.my_ais_register_ll_title);
        llPost.setVisibility(View.GONE);
        llPre = (LinearLayout) findViewById(R.id.my_ais_regis_ll_pre_group);
        llPre.setVisibility(View.GONE);
        llFbb = (LinearLayout) findViewById(R.id.my_ais_regis_ll_fbb_group);
        tvHeader = (AISTextView) findViewById(R.id.my_ais_register_tv_header_add_mobile);
        tvDescAddMobile = (AISTextView) findViewById(R.id.my_ais_register_tv_desc_add_mobile);
        llFbb.setVisibility(View.GONE);
        rvPost = (RecyclerView) findViewById(R.id.my_ais_regis_rv_post_list);
        rvPre = (RecyclerView) findViewById(R.id.my_ais_regis_rv_pre_list);
        rvFbb = (RecyclerView) findViewById(R.id.my_ais_regis_rv_fbb_list);
        btnAdd = (AISButton) findViewById(R.id.my_ais_register_btn_add);
        llBtnAdd = (LinearLayout) findViewById(R.id.my_ais_register_ll_btn_add);
        divider = (AISDivider) findViewById(R.id.my_ais_register_divider);
    }

    @Override
    protected void setupView() {
        rvPost.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
        rvPre.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
        rvFbb.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
        showViewGroupIfCan();
    }

    @Override
    protected void setupStyleables(AttributeSet attrs, int defStyleAttr, int defStyleRes) {

    }

    @Override
    protected Parcelable onSaveInstanceState() {
        Parcelable superState = super.onSaveInstanceState();
        SavedState ss = (SavedState) onSaveChildInstanceState( new SavedState( superState ) );
        ss.canShowTitle = this.canShowTitle;
        return ss;
    }

    @Override
    protected void onRestoreInstanceState(Parcelable state) {
        if (!(state instanceof SavedState)) {
            super.onRestoreInstanceState(state);
            return;
        }
        SavedState ss = (SavedState) state;
        super.onRestoreInstanceState(ss.getSuperState());
        this.canShowTitle = ss.canShowTitle;
    }


    public void hideAddButton() {
        llBtnAdd.setVisibility(View.GONE);
    }

    public void showAddButton() {
        llBtnAdd.setVisibility(View.VISIBLE);
    }

    public void showTitle() {
        llTitle.setVisibility(View.VISIBLE);
    }

    public void hideTitle() {
        llTitle.setVisibility(View.GONE);
    }

    public void setLayoutManager(int layoutManager) {
        switch (layoutManager) {
            case LINEAR_LAYOUT_MANAGER:
                rvPost.setLayoutManager(new MyAisLinearManager(getContext()));
                rvPre.setLayoutManager(new MyAisLinearManager(getContext()));
                rvFbb.setLayoutManager(new MyAisLinearManager(getContext()));
                break;
            case GRID_LAYOUT_MANAGER:
                rvPost.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
                rvPre.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
                rvFbb.setLayoutManager(new MyAisRegisterGridManager(getContext(), 2, false));
                break;
        }
    }

    public void setPostAdapter(RegisterAdapter mobileAdapter) {
        postAdapter = mobileAdapter;
        rvPost.setAdapter(postAdapter);
        setVisibleIsCan(llPost, mobileAdapter.getItemCount());
        showViewGroupIfCan();
    }


    public void setPreAdapter(RegisterAdapter mobileAdapter) {
        preAdapter = mobileAdapter;
        rvPre.setAdapter(preAdapter);
        setVisibleIsCan(llPre, mobileAdapter.getItemCount());
        showViewGroupIfCan();
    }

    public void setFbbAdapter(RegisterAdapter mobileAdapter) {
        fbbAdapter = mobileAdapter;
        rvFbb.setAdapter(fbbAdapter);
        setVisibleIsCan(llFbb, mobileAdapter.getItemCount());
        showViewGroupIfCan();
    }

    public void setHeader(String header){
       tvHeader.setText(header);
    }

    public void setDescAddMobile(String desc){
        tvDescAddMobile.setText(desc);
    }

    public void setOnClickAddListener(View.OnClickListener listener) {
        btnAdd.setVisibility(VISIBLE);
        btnAdd.setOnClickListener(listener);
    }

    public void notifyDataSetChanged() {
        if (postAdapter != null) {
            postAdapter.notifyDataSetChanged();
            setVisibleIsCan(llPost, postAdapter.getItemCount());
            showViewGroupIfCan();
        }

        if (preAdapter != null) {
            preAdapter.notifyDataSetChanged();
            setVisibleIsCan(llFbb, fbbAdapter.getItemCount());
            showViewGroupIfCan();
        }

        if (fbbAdapter != null) {
            fbbAdapter.notifyDataSetChanged();
            setVisibleIsCan(llPre, preAdapter.getItemCount());
            showViewGroupIfCan();
        }

    }

    //<editor-fold desc="Private function folding">
    private void showViewGroupIfCan() {
        setVisibleBaseLayoutIsCan();
        showDividerIfCan();
        showTitleIfCan();
    }

    private void setVisibleBaseLayoutIsCan() {
        if ((preAdapter != null && postAdapter.getItemCount() > 0)
                || (preAdapter != null && preAdapter.getItemCount() > 0)
                || (fbbAdapter != null && fbbAdapter.getItemCount() > 0)) {
            NextzyAnimation.getInstance().animateFadeShow(llBase);
        } else if (btnAdd.getVisibility() == View.VISIBLE) {
            NextzyAnimation.getInstance().animateFadeShow(llBase);
        } else {
            NextzyAnimation.getInstance().animateFadeGone(llBase);
        }
    }

    private void setVisibleIsCan(View v, int arraySize) {
        if (arraySize == 0) {
            v.setVisibility(View.GONE);
        } else {
            v.setVisibility(View.VISIBLE);
        }
    }

    private void showDividerIfCan() {
        if ((preAdapter != null && postAdapter.getItemCount() > 0)
                || (preAdapter != null && preAdapter.getItemCount() > 0)
                || (fbbAdapter != null && fbbAdapter.getItemCount() > 0)) {
            divider.setVisibility(View.VISIBLE);
        } else {
            divider.setVisibility(View.GONE);
        }
    }

    private void showTitleIfCan() {
        if (((preAdapter != null && postAdapter.getItemCount() > 0)
                || (preAdapter != null && preAdapter.getItemCount() > 0)
                || (fbbAdapter != null && fbbAdapter.getItemCount() > 0))
                && canShowTitle) {
            llTitle.setVisibility(View.VISIBLE);
        } else {
            llTitle.setVisibility(View.GONE);
        }
    }

    private static class SavedState extends ChildSavedState {

        public boolean canShowTitle;

        SavedState(Parcelable superState) {
            super(superState);
        }

        private SavedState(Parcel in, ClassLoader classLoader ) {
            super(in, classLoader);
            this.canShowTitle = (in.readInt() == 1);
        }

        @Override
        public void writeToParcel(Parcel out, int flags) {
            super.writeToParcel(out, flags);
            out.writeInt((canShowTitle) ? 1 : 0);
        }

        public static final ClassLoaderCreator<SavedState> CREATOR = new ClassLoaderCreator<SavedState>(){
            @Override
            public SavedState createFromParcel( Parcel source, ClassLoader loader ){
                return new SavedState( source, loader );
            }

            public SavedState createFromParcel( Parcel in ){
                return null;
            }


            public SavedState[] newArray( int size ){
                return new SavedState[size];
            }
        };
    }
