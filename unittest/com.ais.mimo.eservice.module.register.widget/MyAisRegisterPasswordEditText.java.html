package com.ais.mimo.eservice.module.register.widget;

import android.annotation.TargetApi;
import android.content.Context;
import android.os.Build;
import android.os.Parcel;
import android.os.Parcelable;
import android.util.AttributeSet;
import android.view.View;
import android.widget.EditText;
import android.widget.ImageView;

import com.nextzy.myais.common.widget.NextzyViewGroup;
import com.nextzy.myais.myais.R;


public class MyAisRegisterPasswordEditText extends NextzyViewGroup {


    private EditText edtPassword;
    private EditText edtRePassword;
    private ImageView imgTerm;

    public MyAisRegisterPasswordEditText(Context context ){
        super( context );
    }

    public MyAisRegisterPasswordEditText(Context context, AttributeSet attrs ){
        super( context, attrs );
    }

    public MyAisRegisterPasswordEditText(Context context, AttributeSet attrs, int defStyleAttr ){
        super( context, attrs, defStyleAttr );
    }

    @TargetApi( Build.VERSION_CODES.LOLLIPOP )
    public MyAisRegisterPasswordEditText(Context context, AttributeSet attrs, int defStyleAttr, int defStyleRes ){
        super( context, attrs, defStyleAttr, defStyleRes );
    }

    @Override
    protected int getLayoutRes() {
        return R.layout.my_ais_register_view_password;
    }

    @Override
    protected void bindView() {
        edtPassword = (EditText) findViewById( R.id.my_ais_regis_edt_password );
        edtRePassword = (EditText) findViewById( R.id.my_ais_regis_edt_re_password );
        imgTerm = (ImageView) findViewById( R.id.my_ais_regis_term_password );
    }

    @Override
    protected void setupView() {

    }

    @Override
    protected void setupStyleables(AttributeSet attrs, int defStyleAttr, int defStyleRes) {

    }

    @Override
    protected Parcelable onSaveInstanceState() {
        Parcelable superState = super.onSaveInstanceState();
        SavedState ss = (SavedState) onSaveChildInstanceState( new SavedState( superState ) );
        ss.passwordString = edtPassword.getText().toString();
        ss.rePasswordString = edtRePassword.getText().toString();
        return ss;
    }

    @Override
    protected void onRestoreInstanceState(Parcelable state) {
        if (!(state instanceof SavedState)) {
            super.onRestoreInstanceState(state);
            return;
        }
        SavedState ss = (SavedState) state;
        super.onRestoreInstanceState(ss.getSuperState());
        edtPassword.setText(ss.passwordString);
        edtRePassword.setText(ss.rePasswordString);
    }


    public String getPasswordStringFromEditText(){
        return edtPassword.getText().toString();
    }
    public String getRePasswordStringFromEditText(){
       return edtRePassword.getText().toString();
    }

    public void setPassText(String text){
        edtPassword.setText( text );
    }

    public void setRePassText(String text){
        edtRePassword.setText( text );
    }

    public void setOnClickTermListener( View.OnClickListener listener ){
        imgTerm.setClickable( true );
        imgTerm.setOnClickListener( listener );
    }

    public void showDefault(){
        edtPassword.setBackgroundResource( R.drawable.my_ais_library_selector_edit_text_green );
        edtRePassword.setBackgroundResource( R.drawable.my_ais_library_selector_edit_text_green );
    }

    public void showError(){
        edtPassword.setBackgroundResource( R.drawable.my_ais_library_shape_edit_text_white_red_border );
        edtRePassword.setBackgroundResource( R.drawable.my_ais_library_shape_edit_text_white_red_border );
    }

    private static class SavedState extends ChildSavedState {

        String passwordString;
        String rePasswordString;

        SavedState(Parcelable superState) {
            super(superState);
        }

        private SavedState(Parcel in, ClassLoader classLoader) {
            super(in, classLoader);
            this.passwordString = in.readString();
            this.rePasswordString = in.readString();
        }

        @Override
        public void writeToParcel(Parcel out, int flags) {
            super.writeToParcel(out, flags);
            out.writeString(this.passwordString);
            out.writeString(this.rePasswordString);
        }

        public static final ClassLoaderCreator<SavedState> CREATOR = new ClassLoaderCreator<SavedState>(){
            @Override
            public SavedState createFromParcel( Parcel source, ClassLoader loader ){
                return new SavedState( source, loader );
            }

            public SavedState createFromParcel( Parcel in ){
                return null;
            }


            public SavedState[] newArray( int size ){
                return new SavedState[size];
            }
        };
    }
}
