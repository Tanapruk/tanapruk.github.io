package com.ais.mimo.eservice.helper.adobeanalytics;

import com.adobe.mobile.Analytics;

import java.util.HashMap;

/**
 * Created by RoyleKoonlert on 2/7/2017 AD.
 */

public class AisAdobeAnalytics {
    private static AisAdobeAnalytics aisAdobeAnalytics;

    public static AisAdobeAnalytics getInstance() {
        if (aisAdobeAnalytics == null) {
            aisAdobeAnalytics = new AisAdobeAnalytics();
        }
        return aisAdobeAnalytics;
    }

    //secretCode = encrypt mobile number from api
    public void track(String secretCode) {
        HashMap<String, Object> contextData = new HashMap<>();
        contextData.put("ais.user.loginStatus", "logged-in");
        contextData.put("ais.user.loginType", "authenticated");
        contextData.put("ais.user.loginId", secretCode);
        contextData.put("successful_login", "success");
        Analytics.trackState("Login Page", contextData);
        Analytics.trackAction("Successful Login", contextData);
    }

}
