package com.nextzy.myais.balanceusage.module.cqs.activity;

import com.hwangjr.rxbus.RxBus;
import com.nextzy.myais.balanceusage.helper.cqs.network.CqsManager;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class CqsActivityPresenter extends NextzyActivityPresenter<CqsActivityContractor.View> implements CqsActivityContractor.Presenter {

    private final CqsManager mManager;

    protected CqsActivityPresenter( CqsActivityContractor.View view ){
        super( view );
        mManager = CqsManager.getInstance();
    }

    public static CqsActivityPresenter createPresenter( CqsActivityContractor.View view) {
        return new CqsActivityPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart(){
        // for handle network error
        super.onViewStart();
    }

    @Override
    public void onViewStop(){
        super.onViewStop();
    }
}
