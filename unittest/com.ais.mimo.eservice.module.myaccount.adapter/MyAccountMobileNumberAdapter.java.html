package com.ais.mimo.eservice.module.myaccount.adapter;

import android.support.v7.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.ViewGroup;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.base.RegisterAdapter;
import com.ais.mimo.eservice.module.register.step4.adapter.holder.MobileItemHolder;
import com.nextzy.myais.myais.R;

import java.util.ArrayList;


/**
 * Created by Nonthawit on 7/30/2016.
 */

public class MyAccountMobileNumberAdapter extends RegisterAdapter<MyAccountMobileNumberContractor.Presenter> implements MyAccountMobileNumberContractor.View {
    private final static String TAG = MyAccountMobileNumberAdapter.class.getSimpleName();

    private ArrayList<MobileData> mobileList;

    @Override
    public void createPresenter() {
        MyAccountMobileNumberPresenter.createPresenter(this);
    }

    public MyAccountMobileNumberAdapter() {
        this.mobileList = new ArrayList<>();
    }

    public MyAccountMobileNumberAdapter(ArrayList<MobileData> mobileNumberList) {
        if (mobileNumberList == null) {
            this.mobileList = new ArrayList<>();
        } else {
            this.mobileList = mobileNumberList;
        }
    }

    @Override
    public int getItemCount() {
        return mobileList.size();
    }


    @Override
    public RecyclerView.ViewHolder onCreateViewHolder(ViewGroup viewGroup, int viewType) {
        LayoutInflater mInflater = LayoutInflater.from(viewGroup.getContext());

        ViewGroup vItem = (ViewGroup) mInflater.inflate(R.layout.my_ais_register_view_mobile_item, viewGroup, false);
        return new MobileItemHolder(vItem);
    }

    @Override
    public void onBindViewHolder(RecyclerView.ViewHolder viewHolder, int pos) {
        MobileItemHolder item = (MobileItemHolder) viewHolder;
        item.setText(mobileList.get(pos).getNumber());
    }

    public void addAllMobileNumber(ArrayList<MobileData> mobileList) {
        if (getItemCount() != 0) {
            for (MobileData mobile : mobileList) {
                if (!this.mobileList.contains(mobile)) {
                    this.mobileList.add(mobile);
                }
            }
        } else {
            this.mobileList = mobileList;
        }
        notifyDataSetChanged();
    }

    public ArrayList<MobileData> getConfirmMobileList() {
        return mobileList;
    }

    @Override
    public ArrayList<MobileData> getNoConfirmMobileList() {
        return new ArrayList<>();
    }
}
