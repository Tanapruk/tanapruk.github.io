package com.nextzy.myais.home.module;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.home.module.summaryinfo.event.DisableViewScrollingEvent;
import com.nextzy.myais.home.module.summaryinfo.event.EnableViewScrollingEvent;

/**
 * Created by nextzy on 8/21/2016 AD.
 */
public class HomePresenter extends NextzyPresenter<HomeContractor.View> implements HomeContractor.Presenter {
    private String menuKey;

    public static HomePresenter createPresenter(HomeContractor.View view) {
        return new HomePresenter(view);
    }

    public HomePresenter(HomeContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {

    }

    @Override
    public void setMenuKey(String key) {
        this.menuKey = key;
    }

    @Override
    public String getMenuKey() {
        return menuKey;
    }

    @Subscribe
    public void onEnableViewScrollingEvent(EnableViewScrollingEvent event) {
        getView().enableNestedScrolling();
    }

    @Subscribe
    public void onDisableViewScrollingEvent(DisableViewScrollingEvent event) {
        getView().disableNestedScrolling();
    }

}
