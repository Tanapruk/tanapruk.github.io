package com.ais.mimo.eservice.config.menu.side;

public enum SideMenuKey {
    KEY_AIS_ACCOUNT(MyAccountActivity.class),
    KEY_AIS_PROFILE(MyProfileActivity.class),
    KEY_AIS_REGISTER(RegisterIntroActivity.class),
    KEY_AIS_WEBSITE("http://www.ais.co.th", "AIS Website", "APP_VIEW_AIS_WEBSITE", "5012", "5126"),
    KEY_AIS_ONLINE_STORE("https://store.ais.co.th", "AIS Online Store", "APP_VIEW_AIS_ONLINE_STORE   ", "5012", "5125"),
    KEY_AIS_FACEBOOK_POSTPAID("https://www.facebook.com/AIS/", "Facebook", "APP_VIEW_AIS FACEBOOK", "5012", "5127"),
    KEY_AIS_FACEBOOK_PREPAID("https://www.facebook.com/12call/", "Facebook", "APP_VIEW_AIS FACEBOOK", "5012", "5127"),
    KEY_AIS_FACEBOOK(SocialPortalActivity.class, SocialPortalKey.KEY_FACEBOOK),
    KEY_AIS_TWITTER_POSTPAID("https://twitter.com/AIS_Thailand", "Twitter", "APP_VIEW_AIS_TWITTER", "5012", "5128"),
    KEY_AIS_TWITTER_PREPAID("https://twitter.com/12Call_Fan", "Twitter", "APP_VIEW_AIS_TWITTER", "5012", "5128"),
    KEY_AIS_TWITTER(SocialPortalActivity.class, SocialPortalKey.KEY_TWITTER),
    KEY_AIS_LINE_POSTPAID("http://line.me/ti/p/@aisofficialline", "Line", "APP_VIEW_AIS LINE", "5012", "5129"),
    KEY_AIS_LINE_PREPAID("http://line.me/ti/p/@aisofficialline", "Line", "APP_VIEW_AIS LINE", "5012", "5129"),
    KEY_AIS_LINE(SocialPortalActivity.class, SocialPortalKey.KEY_LINE),
    KEY_SAMPLE(SampleActivity.class),
    KEY_SETTING(SettingActivity.class),
    KEY_DEVMODE(DeveloperModeActivity.class),
    KEY_LOGOUT(LogoutActivity.class);

    private Class<? extends FragmentActivity> activity;
    private String url;
    private String value;
    private String title;

    /**
     * eventLog related strings
     */
    private String eventName;
    private String menuId;
    private String eventId;


    SideMenuKey() {
    }

    SideMenuKey(String url) {
        this.url = url;
    }

    SideMenuKey(String url, String title) {
        this.url = url;
        this.title = title;
    }

    SideMenuKey(String url, String title, String eventName, String menuId, String eventId) {
        this.url = url;
        this.title = title;
        this.eventName = eventName;
        this.menuId = menuId;
        this.eventId = eventId;
    }

    SideMenuKey(Class<? extends FragmentActivity> activity) {
        this.activity = activity;
    }

    SideMenuKey(Class<? extends FragmentActivity> activity, String value) {
        this.activity = activity;
        this.value = value;
    }

    public boolean isActivity() {
        return activity != null;
    }

    public boolean isContainValue() {
        return value != null;
    }

    public boolean isUrl() {
        return url != null;
    }

    public Class getActivity() {
        return activity;
    }

    public String getValue() {
        return value;
    }

    public String getUrl() {
        return url;
    }

    public String getTitle() {
        return title;
    }

    public String getEventName() {
        return eventName;
    }

    public String getMenuId() {
        return menuId;
    }

    public String getEventId() {
        return eventId;
    }
}
