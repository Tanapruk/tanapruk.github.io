package com.ais.mimo.eservice.helper.myaccount.network.operator;

import com.ais.mimo.eservice.helper.myaccount.network.exception.ChangeEmailException;
import com.ais.mimo.eservice.helper.myaccount.network.operator.base.MyAccountHasDataCheck;
import com.ais.mimo.eservice.helper.myaccount.network.response.ChangeEmailResult;

import rx.Single;

/**
 * Created by TheKhaeng on 11/15/2016.
 */

public class ChangeEmailDataCheck extends MyAccountHasDataCheck<ChangeEmailResult, ChangeEmailException> {
    @Override
    protected ChangeEmailException getThrowable(String message) {
        return new ChangeEmailException(message);
    }
    @Override
    public Single<ChangeEmailResult> call(ChangeEmailResult remarkResult ){
        if ( isNot20000( remarkResult )){
            return Single.error(getThrowable( remarkResult.getResultDescription() ));
        }
        return Single.just( remarkResult );
    }
}
