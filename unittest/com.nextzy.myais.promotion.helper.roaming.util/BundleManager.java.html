package com.nextzy.myais.promotion.helper.roaming.util;

import android.content.Intent;

public class BundleManager {


    final String CUSTOMER_TYPE = "chosenCustomerType";
    final String COUNTRY_NAME = "chosenCountryName";
    final String COUNTRY_NAME_EN = "chosenCountryNameEn";
    final String PACKAGE_TYPE = "chosenPackageType";
    final String UPPACK_CDATA = "upPackCDATA";
    final String UPSUPPORT_RESULT = "UpSupportResult";

    final String IS3G_STRING = "is3GString";
    final String ISPREPAID_STRING = "isPrepaidString";
    final String PACKAGE_TYPE_STRING = "packageTypeString";
    final String PACKAGE_NO = "packageNumber";

    final String START_DATE_TIME_IN_THAI = "startDateTimeInThailand";
    final String END_DATE_TIME_IN_THAI = "endDateTimeInThailand";
    final String DT = "dt";

    final Bundle bundle;
    Intent intent;

    public BundleManager() {
        bundle = new Bundle();
    }

    public BundleManager(Intent intent) {
        bundle = new Bundle();
        this.intent = intent;
    }

    public void setIntent(Intent intent) {
        this.intent = intent;
    }


    public Bundle build() {
        return bundle;
    }


    public BundleManager addIs3GString(String is3GString) {
        bundle.putString(IS3G_STRING, is3GString);
        return this;
    }

    public BundleManager addIsPrepaidString(String isPrepaidString) {
        bundle.putString(ISPREPAID_STRING, isPrepaidString);
        return this;
    }

    public BundleManager addPackageTypeString(String packageTypeString) {
        bundle.putString(PACKAGE_TYPE_STRING, packageTypeString);
        return this;
    }

    public BundleManager addChosenCustomerType(String chosenCustomerType) {
        bundle.putString(CUSTOMER_TYPE, chosenCustomerType);
        return this;
    }

    public BundleManager addChosenCountryNameEn(String chosenCustomerNameEn) {
        bundle.putString(COUNTRY_NAME_EN, chosenCustomerNameEn);

        return this;
    }

    public BundleManager addChosenCountryName(String chosenCustomerName) {
        bundle.putString(COUNTRY_NAME, chosenCustomerName);

        return this;
    }

    public BundleManager addChosenPackageType(String chosenPackageType) {
        bundle.putString(PACKAGE_TYPE, chosenPackageType);

        return this;
    }

    public BundleManager addUpPackCDATA(UpPackCDATA upPackCDATA) {

        return this;
    }

    public BundleManager addPackageNo(int packageNo) {
        bundle.putInt(PACKAGE_NO, packageNo);
        return this;
    }

    public BundleManager addStartDateTimeInThai(String startDateTimeInThailand) {
        bundle.putString(START_DATE_TIME_IN_THAI, startDateTimeInThailand);
        return this;
    }

    public BundleManager addEndDateTimeInThai(String endDateTimeInThailand) {
        bundle.putString(END_DATE_TIME_IN_THAI, endDateTimeInThailand);
        return this;
    }

    public BundleManager addDt(String dt) {
        bundle.putString(DT, dt);
        return this;
    }


    public String getStartDateTimeInThai() {
        return intent.getStringExtra(START_DATE_TIME_IN_THAI);
    }

    public String getEndDateTimeInThai() {
        return intent.getStringExtra(END_DATE_TIME_IN_THAI);
    }

    public String getDt() {
        return intent.getStringExtra(DT);
    }


    public String getIs3GString() {
        return intent.getStringExtra(IS3G_STRING);
    }

    public String getIsPrepaidString() {
        return intent.getStringExtra(ISPREPAID_STRING);

    public String getPackageTypeString() {
        return intent.getStringExtra(PACKAGE_TYPE_STRING);
    }

    public String getChosenCustomerType() {
        return intent.getStringExtra(CUSTOMER_TYPE);
    }


    public String getChosenCountryName() {
        return intent.getStringExtra(COUNTRY_NAME);
    }

    public String getChosenCountryNameEn() {
        return intent.getStringExtra(COUNTRY_NAME_EN);
    }

    public String getChosenPackageType() {
        return intent.getStringExtra(PACKAGE_TYPE);
    }

    public int getPackageNo() {
        return intent.getIntExtra(PACKAGE_NO, 1);
    }
