package com.nextzy.myais.promotion.helper.roaming.util;

import com.nextzy.myais.promotion.helper.roaming.model.TwoColumnCountryName;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by trusttanapruk on 9/11/15 AD.
 */
public class ListManager {

    public static ListManager instance;

    public static ListManager getInstance() {
        if (instance == null) {
            instance = new ListManager();
        }
        return instance;
    }




    public static List<TwoColumnCountryName> convertTwoColumn(List<String> countryNameList) {

        List<TwoColumnCountryName> twoColumnCountryNameList = new ArrayList<>();
        for (int i = 0; i < countryNameList.size(); i += 2) {

            TwoColumnCountryName twoColumnCountryName = new TwoColumnCountryName();

            String countryNameLeft = countryNameList.get(i);

            twoColumnCountryName.setCountryLeft(countryNameLeft);

            int countryNameRightLocation = i + 1;

            if (countryNameRightLocation < countryNameList.size()) {
                String countryNameRight = countryNameList.get(i + 1);
                twoColumnCountryName.setCountryRight(countryNameRight);
            }
            twoColumnCountryNameList.add(twoColumnCountryName);


        }

        return twoColumnCountryNameList;
    }



}
