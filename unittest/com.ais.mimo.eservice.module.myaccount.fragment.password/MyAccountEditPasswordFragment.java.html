package com.ais.mimo.eservice.module.myaccount.fragment.password;

import android.os.Bundle;
import android.text.InputType;
import android.view.View;

import com.ais.mimo.eservice.module.register.widget.MyAisRegisterEditText;
import com.nextzy.myais.common.mvp.NextzyMvpFragment;
import com.nextzy.myais.common.widget.AISButton;
import com.nextzy.myais.common.widget.dialog.AisDialog;
import com.nextzy.myais.myais.R;

/**
 * Created by TheKhaeng on 11/16/2016.
 */

public class MyAccountEditPasswordFragment extends NextzyMvpFragment<MyAccountEditPasswordContractor.Presenter> implements MyAccountEditPasswordContractor.View {
    public final static String TAG = MyAccountEditPasswordFragment.class.getSimpleName();
    private MyAisRegisterEditText etOldPassword;
    private MyAisRegisterEditText etNewPassword;
    private MyAisRegisterEditText etConfirmNewPassword;
    private AISButton btnSaveChanges;
    private AISButton btnCancelSaveChanges;

    public MyAccountEditPasswordFragment() {
        super();
    }

    public static MyAccountEditPasswordFragment newInstance() {
        MyAccountEditPasswordFragment fragment = new MyAccountEditPasswordFragment();
        Bundle args = new Bundle();
        fragment.setArguments(args);
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter() {
        MyAccountEditPasswordPresenter.createPresenter(this);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView(View view) {
        etOldPassword = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_account_et_old_password);
        etNewPassword = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_account_et_new_password);
        etConfirmNewPassword = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_account_et_confirm_new_password);
        btnSaveChanges = (AISButton) view.findViewById(R.id.my_ais_account_btn_save_changes);
        btnCancelSaveChanges = (AISButton) view.findViewById(R.id.my_ais_account_btn_cancel_save_changes);
    }

    @Override
    public void setupView() {
        etOldPassword.setInputType(InputType.TYPE_CLASS_TEXT | InputType.TYPE_TEXT_VARIATION_PASSWORD);
        etNewPassword.setInputType(InputType.TYPE_CLASS_TEXT | InputType.TYPE_TEXT_VARIATION_PASSWORD);
        etConfirmNewPassword.setInputType(InputType.TYPE_CLASS_TEXT | InputType.TYPE_TEXT_VARIATION_PASSWORD);
        etNewPassword.setOnClickTermListener(view -> showPasswordTerm());
        btnSaveChanges.setOnClickListener(view -> clickSaveChanges());
        btnCancelSaveChanges.setOnClickListener(view -> clickCancel());
    }


    @Override
    public void initialize() {
    }


    @Override
    public void restoreView() {
    }

    /**
     * onCreateView
     **/
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_account_fragment_edit_password;
    }


    /**
     * State
     **/
    @Override
    public void saveInstanceState(Bundle outState) {
        // Save Instance State here
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        // Restore instance state here
    }
    //</editor-fold>


    @Override
    public void back() {
        dismissDialog();
        getActivity().getSupportFragmentManager().popBackStack();
    }

    @Override
    public void showDefaultNewPasswordEdt(){
        etNewPassword.showDefault();
        etConfirmNewPassword.showDefault();
    }

    @Override
    public void showDefaultOldPasswordEdt(){
        etOldPassword.showDefault();
    }

    @Override
    public void showFormatErrorDialogNewPasswordEdt(){
        etNewPassword.showError();
        etConfirmNewPassword.showError();
        String error;
        if( etNewPassword.getStringFromEditText().isEmpty()
                || etConfirmNewPassword.getStringFromEditText().isEmpty()){
            error = getString(R.string.my_ais_register_dialog_password_none);
        }else{
            error = getString(R.string.my_ais_register_dialog_password_fault);
        }
        showResponseExceptionDismiss(error);
    }

    @Override
    public void showFormatErrorDialogCurrentPasswordEdt(){
        etOldPassword.showError();
        String error;
        if( etOldPassword.getStringFromEditText().isEmpty()){
            error = getString(R.string.my_ais_register_dialog_password_none);
        }else{
            error = getString(R.string.my_ais_register_dialog_password_fault);
        }
        showResponseExceptionDismiss(error);
    }

    @Override
    public void showNotMatchErrorDialogNewPasswordEdt() {
        etNewPassword.showError();
        etConfirmNewPassword.showError();
        String error = getString(R.string.my_ais_register_dialog_password_not_match);
        showResponseExceptionDismiss(error);
    }

    @Override
    public void showResponseError(String resultDescription) {
        showResponseExceptionDismiss(resultDescription);
    }

    @Override
    public void showChangePasswordSuccess(String resultDescription) {
        dismissDialog();
        showAlertDialogCustomOnClick(resultDescription, data -> getActivity().getSupportFragmentManager().popBackStack());
    }

    @Override
    public void showDialogErrorAlreadyUsedPassword() {
        etNewPassword.showError();
        etConfirmNewPassword.showError();
        String error = getString(R.string.my_ais_register_dialog_password_already_used);
        showResponseExceptionDismiss(error);
    }

    /********************/
    /** Private method **/
    /********************/
    //<editor-fold desc="Private function folding">
    private void clickCancel() {
        back();
    }

    private void clickSaveChanges() {
        getPresenter().onRequestChangePassword(
                etOldPassword.getStringFromEditText(),
                etNewPassword.getStringFromEditText(),
                etConfirmNewPassword.getStringFromEditText()
        );
    }

    private void showPasswordTerm() {
        AisDialog.getInstance().showHtmlDialog(
                getFragmentManager(),
                getString(R.string.my_ais_register_dialog_password_title),
                getString(R.string.my_ais_register_dialog_password_info),
                getContext()
        );
    }
    //</editor-fold>
}
