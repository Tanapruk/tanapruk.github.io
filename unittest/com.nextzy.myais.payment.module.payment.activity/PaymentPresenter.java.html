package com.nextzy.myais.payment.module.payment.activity;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;
import com.nextzy.myais.payment.helper.payment.event.GoToBillEvent;
import com.nextzy.myais.payment.helper.payment.event.GotoBillConfirmEvent;
import com.nextzy.myais.payment.helper.payment.event.GotoBillOtherEvent;
import com.nextzy.myais.payment.helper.payment.event.GotoPayAllBillConfirmEvent;
import com.nextzy.myais.payment.helper.payment.event.GotoPayAllBillEvent;

import java.lang.Override;

public class PaymentPresenter extends NextzyActivityPresenter<PaymentContractor.View> implements PaymentContractor.Presenter {
    public PaymentPresenter(PaymentContractor.View view) {
        super(view);
    }

    public static PaymentPresenter createPresenter(PaymentContractor.View view) {
        return new PaymentPresenter(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
        RxBus.get().register(this);
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
        RxBus.get().unregister(this);
    }

    @Subscribe
    public void onGotoBillConfirmEvent(GotoBillConfirmEvent event) {
        getView().gotoBillConfirmEvent();
    }

    @Subscribe
    public void onGotoBillOtherEvent(GotoBillOtherEvent event) {
        getView().gotoBillOtherEvent();
    }

    @Subscribe
    public void onGotoBillEvent(GoToBillEvent event) {
        getView().gotoBillEvent();
    }

    @Subscribe
    public void onGotoPayAllBillEvent(GotoPayAllBillEvent event) {
        getView().gotoPayAllBillEvent();
    }

    @Subscribe
    public void onGotoPayAllBillConfirmEvent(GotoPayAllBillConfirmEvent event) {
        getView().gotoPayAllBillConfirmEvent();
    }
}
