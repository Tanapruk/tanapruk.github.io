package com.nextzy.myais.payment.module.topup.activity;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;
import com.nextzy.myais.payment.helper.topup.event.GotoTopUpConfirmPostpaidEvent;

public class TopupPresenter extends NextzyActivityPresenter<TopupContractor.View> implements TopupContractor.Presenter {
    public TopupPresenter(TopupContractor.View view) {
        super(view);
    }

    public static TopupPresenter createPresenter(TopupContractor.View view) {
        return new TopupPresenter(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
        RxBus.get().register(this);
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
        RxBus.get().unregister(this);
    }

    @Subscribe
    public void onGotoTopupConfirm(GotoTopUpConfirmPostpaidEvent event) {
        getView().gotoTopUpConfirmation();
    }
}
