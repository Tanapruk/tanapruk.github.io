package com.ais.mimo.eservice.module.register.step4.adapter.holder;

import android.support.v7.widget.RecyclerView;
import android.view.View;

import com.nextzy.myais.common.widget.AISTextView;
import com.nextzy.myais.myais.R;

import static com.nextzy.myais.common.delegate.FormatUtility.FORMAT_MOBILE_NUMBER;

/**
 * Created by TheKhaeng on 10/13/2016.
 */

public class MobileItemHolder extends RecyclerView.ViewHolder {
    public AISTextView tvText;

    public MobileItemHolder(View itemView) {
        super(itemView);
        tvText = (AISTextView) itemView.findViewById( R.id.my_ais_register_tv_text);
    }

    public void setText(String text ){
        tvText.setText( text, FORMAT_MOBILE_NUMBER );
    }
}
