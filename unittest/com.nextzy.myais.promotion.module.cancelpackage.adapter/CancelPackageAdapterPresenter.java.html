package com.nextzy.myais.promotion.module.cancelpackage.adapter;

import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.promotion.module.cancelpackage.adapter.model.BaseCancelItem;

import java.util.List;

/**
 * Created by TheKhaeng on 10/3/2016.
 */

public class CancelPackageAdapterPresenter extends NextzyPresenter<CancelAdapterContractor.View>
        implements CancelAdapterContractor.Presenter{


    private List<BaseCancelItem> cancelItems;

    public static CancelPackageAdapterPresenter createPresenter(CancelAdapterContractor.View view ){
        return new CancelPackageAdapterPresenter( view );
    }

    private CancelPackageAdapterPresenter(CancelAdapterContractor.View view ){
        super( view );
    }


    @Override
    public void onViewCreate(){

    }

    @Override
    public void onViewDestroy(){

    }

    @Override
    public void onViewStart(){

    }

    @Override
    public void onViewStop(){

    }


    @Override
    public int getItemCount() {
        if(cancelItems == null) return 0;
        return cancelItems.size();
    }

    @Override
    public void setCancelPackageItems(List<BaseCancelItem> cancelItems ){
        this.cancelItems = cancelItems;
    }


    @Override
    public BaseCancelItem getCancelPackageItem(int pos) {
        return cancelItems.get(pos);
    }

    @Override
    public int getItemViewType( int position ){
        return cancelItems.get(position).getType();
    }

}
