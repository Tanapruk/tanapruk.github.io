package com.ais.mimo.eservice.helper.myaccount.utility;

import com.ais.mimo.eservice.helper.myaccount.network.response.MyNumberData;
import com.ais.mimo.eservice.helper.myaccount.network.response.MyNumberResult;
import com.ais.mimo.eservice.widget.MyNumberDropDownDialog;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.config.mobile.MobileType;
import com.nextzy.myais.common.config.mobile.MobileUtility;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

/**
 * Created by RoyleKoonlert on 3/6/2017 AD.
 */

public class MyNumberArranger {

    private MyNumberResult myNumberResult;
    private List<MyNumberDropDownDialog.ContentItem> postpaidContentItemList = new ArrayList<>();
    private List<MyNumberDropDownDialog.ContentItem> prepaidContentItemList = new ArrayList<>();
    private List<MyNumberDropDownDialog.ContentItem> fibreContentItemList = new ArrayList<>();

    private MyNumberResult getMyNumberResult() {
        return myNumberResult;
    }

    private void setMyNumberResult(MyNumberResult myNumberResult) {
        this.myNumberResult = myNumberResult;
    }

    public List<MyNumberDropDownDialog.ContentItem> getContentItemList(MyNumberResult myNumberResult) {
        setMyNumberResult(myNumberResult);
        List<MyNumberDropDownDialog.ContentItem> contentItems = new ArrayList<>();
        if (isData()) {
            divideGroupsByPriority();
            sortMyNumberAllGroups();
            addContentItems(contentItems);
        }
        return contentItems;
    }

    private List<MyNumberDropDownDialog.ContentItem> addContentItems(List<MyNumberDropDownDialog.ContentItem> contentItems) {
        if (Mobile.getInstance().getMobileType() == MobileType.POSTPAID) {
            contentItems.addAll(postpaidContentItemList);
        } else if (Mobile.getInstance().getMobileType() == MobileType.FBB) {
            if (postpaidContentItemList.size() > 0) {
                contentItems.addAll(postpaidContentItemList);
                return contentItems;
            }
            contentItems.addAll(fibreContentItemList);
        } else {
            if (postpaidContentItemList.size() > 0) {
                contentItems.addAll(postpaidContentItemList);
                return contentItems;
            }
            if (fibreContentItemList.size() > 0) {
                contentItems.addAll(fibreContentItemList);
                return contentItems;
            }
            contentItems.addAll(prepaidContentItemList);
        }

        return contentItems;
    }

    private void sortMyNumberAllGroups() {
        postpaidContentItemList = sortDataASC(postpaidContentItemList);
        fibreContentItemList = sortDataASC(fibreContentItemList);
        prepaidContentItemList = sortDataASC(prepaidContentItemList);
    }

    private List<MyNumberDropDownDialog.ContentItem> sortDataASC(List<MyNumberDropDownDialog.ContentItem> contentItemList) {
        if (contentItemList.size() > 0)
            Collections.sort(contentItemList, (contentItem1, contentItem2) -> contentItem1
                    .getMobileNumber().compareTo(contentItem2.getMobileNumber()));
        return contentItemList;
    }

    private boolean isData() {
        return myNumberResult.hasDataList() && myNumberResult.getDataList().size() > 0;
    }

    private boolean isMobileNumberNotMaster(MyNumberData myNumberData) {
        return !myNumberData.getNumber().equals(Mobile.getInstance().getMasterMobileNumber());
    }

    private void divideGroupsByPriority() {
        for (MyNumberData myNumberData : myNumberResult.getDataList()) {
            if (isMobileNumberNotMaster(myNumberData)) {
                if (isPostpaid(myNumberData))
                    addPostpaidGroup(myNumberData);
                if (isPrepaid(myNumberData))
                    addPrepaidGroup(myNumberData);
                if (isFibre(myNumberData))
                    addFibreGroup(myNumberData);
            }
        }
    }

    private boolean isPostpaid(MyNumberData myNumberData) {
        return MobileUtility.getInstance().getMobileTypeByMobileData(myNumberData.getChargeType(),
                myNumberData.getMobileType()) == MobileType.POSTPAID;
    }

    private boolean isPrepaid(MyNumberData myNumberData) {
        return MobileUtility.getInstance().getMobileTypeByMobileData(myNumberData.getChargeType(),
                myNumberData.getMobileType()) == MobileType.PREPAID;
    }

    private boolean isFibre(MyNumberData myNumberData) {
        return MobileUtility.getInstance().getMobileTypeByMobileData(myNumberData.getChargeType(),
                myNumberData.getMobileType()) == MobileType.FBB;
    }

    private void addPostpaidGroup(MyNumberData myNumberData) {
        postpaidContentItemList.add(new MyNumberDropDownDialog.ContentItem(myNumberData
                .getNumber()));
    }

    private void addPrepaidGroup(MyNumberData myNumberData) {
        prepaidContentItemList.add(new MyNumberDropDownDialog.ContentItem(myNumberData
                .getNumber()));
    }

    private void addFibreGroup(MyNumberData myNumberData) {
        fibreContentItemList.add(new MyNumberDropDownDialog.ContentItem(myNumberData
                .getNumber()));
    }

}
