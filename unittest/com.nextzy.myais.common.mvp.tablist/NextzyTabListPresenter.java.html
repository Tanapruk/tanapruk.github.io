package com.nextzy.myais.common.mvp.tablist;

import com.nextzy.myais.common.menu.MyAisMenu;
import com.nextzy.myais.common.mvp.NextzyPresenter;
import com.nextzy.myais.common.widget.tabmenu.MyAisTabMenu;
import com.nextzy.myais.common.widget.tabmenu.MyAisTabMenuUtility;

import java.util.List;

/**
 * Created by nextzy on 8/21/2016 AD.
 */
public class NextzyTabListPresenter extends NextzyPresenter<NextzyTabListContractor.View> implements NextzyTabListContractor.Presenter {

    public static NextzyTabListPresenter createPresenter(NextzyTabListContractor.View view) {
        return new NextzyTabListPresenter(view);
    }

    private NextzyTabListPresenter(NextzyTabListContractor.View view) {
        super(view);
    }

    @Override
    public void onViewCreate() {

    }

    @Override
    public void onViewDestroy() {

    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {

    }

    @Override
    public void onCreateMenu(String key) {
        MyAisMenu.Container menuList = getView().getMenuFromDatabase();
        MyAisMenu.Menu myAisMenuMenu = MyAisTabMenuUtility.getInstance().getSectionMenuByKey(menuList, key);
        if (myAisMenuMenu != null) {
            List<MyAisTabMenu> tabMenuList = MyAisTabMenuUtility.getInstance().convertMenuToTabMenu(myAisMenuMenu);
            getView().setMenu(tabMenuList);
        }
    }

    @Override
    public void onSubMenuClick(MyAisTabMenu menu) {
    }
}
