package com.nextzy.myais.common.widget.tabmenu;

import android.graphics.Color;

import com.nextzy.myais.common.menu.MyAisMenu;
import com.nextzy.myais.common.utility.NextzyLanguage;
import com.nextzy.myais.common.utility.NextzyUtility;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by Akexorcist on 9/5/2016 AD.
 */

public class MyAisTabMenuUtility {
    private static MyAisTabMenuUtility menuUtility;

    public static MyAisTabMenuUtility getInstance() {
        if (menuUtility == null) {
            menuUtility = new MyAisTabMenuUtility();
        }
        return menuUtility;
    }

    public List<MyAisTabMenu> convertMenuToTabMenu(MyAisMenu.Menu menu) {
        List<MyAisTabMenu> myAisTabMenuList = new ArrayList<>();
        if (menu != null && menu.isContainSectionMenu()) {
            myAisTabMenuList.add(new MyAisTabMenu());
            for (MyAisMenu.SectionMenu sectionMenu : menu.getSectionMenuList()) {
                String sectionMenuTitle = NextzyUtility.getInstance().getLocalMessage(sectionMenu.getNameEn(), sectionMenu.getNameTh(), NextzyLanguage.getInstance().getCurrentLanguage());
                String sectionMenuKey = sectionMenu.getKey();
                int sectionMenuIcon = sectionMenu.getImage();
                int sectionMenuColor = Color.parseColor(sectionMenu.getColor());
                myAisTabMenuList.add(new MyAisTabMenu(sectionMenuTitle, sectionMenuKey, sectionMenuIcon, sectionMenuColor));
                if (sectionMenu.isContainSubMenu()) {
                    for (MyAisMenu.SubMenu subMenu : sectionMenu.getSubMenuList()) {
                        String subMenuTitle = NextzyUtility.getInstance().getLocalMessage(subMenu.getNameEn(), subMenu.getNameTh(), NextzyLanguage.getInstance().getCurrentLanguage());
                        String subMenuKey = subMenu.getKey();
                        int subMenuColor = Color.parseColor(subMenu.getColor());
                        myAisTabMenuList.add(new MyAisTabMenu(subMenuTitle, subMenuKey, subMenuColor));
                    }
                }
            }
            myAisTabMenuList.add(new MyAisTabMenu());
        }
        return myAisTabMenuList;
    }

    public MyAisMenu.Menu getSectionMenuByKey(MyAisMenu.Container menuList, String key) {
        for (MyAisMenu.Menu menu : menuList.getMenuList()) {
            if (menu.getKey().equals(key)) {
                return menu;
            }
        }
        return null;
    }
}
