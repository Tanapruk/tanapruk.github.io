package com.nextzy.myais.common.database;

import com.google.gson.Gson;
import com.google.gson.GsonBuilder;
import com.nextzy.myais.common.database.model.NextzyRealmObject;

/**
 * Created by Akexorcist on 9/6/2016 AD.
 */

public class RealmUtility {
    private static RealmUtility realmUtility;

    public static RealmUtility getInstance() {
        if (realmUtility == null) {
            realmUtility = new RealmUtility();
        }
        return realmUtility;
    }

    public <T> T convertToClass(NextzyRealmObject realmMenu, Class<T> classOfT) {
        return new Gson().fromJson(realmMenu.getJson(), classOfT);
    }

    public NextzyRealmObject createRealmObject(String name, Object data) {
        String className = data.getClass().getSimpleName();
        String json = new GsonBuilder().setPrettyPrinting().create().toJson(data);
        return new NextzyRealmObject(name, className, json);
    }
}
