package com.ais.mimo.eservice.helper.term.network;

import com.nextzy.myais.common.network.worker.GenericApiService;

import okhttp3.Request;

 * Created by trusttanapruk on 9/15/2016.

public class TermApiService extends GenericApiService<TermApi> {
    @Override
    protected Class<TermApi> getApiClassType() {
        return TermApi.class;
    }

    public static TermApiService newInstance(String baseUrl) {
        TermApiService service = new TermApiService();
        service.setBaseUrl(baseUrl);
        return service;
    }
