package com.ais.mimo.eservice.module.register.event.data;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;

import java.util.ArrayList;

import lombok.EqualsAndHashCode;
import lombok.Getter;

/**
 * Created by TheKhaeng on 11/29/2016.
 */

@EqualsAndHashCode
public class SelectPostpaidEventData {
    @Getter
    private final String password;
    @Getter
    private final String rePassword;
    @Getter
    private ArrayList<MobileData> postpaidGroup;
    @Getter
    private ArrayList<MobileData> fbbGroup;

    public SelectPostpaidEventData(String password, String rePassword,
                                   ArrayList<MobileData> postpaidGroup,
                                   ArrayList<MobileData> fbbGroup) {
        this.password = password;
        this.rePassword = rePassword;
        this.postpaidGroup = postpaidGroup;
        this.fbbGroup = fbbGroup;
    }
}
