package com.ais.mimo.eservice.helper.appupdate.network.response;

import com.google.gson.annotations.SerializedName;
import com.nextzy.myais.common.network.model.GenericResult;

/**
 * Created by trusttanapruk on 8/25/2016.
 */
public class AppUpdateResult extends GenericResult {

    public AppUpdateResult() {
    }

    @SerializedName("isforce")
    String isForce;
    String version;
    String url;
    String message;

    public boolean isForce() {
        return isForce != null && "true".equalsIgnoreCase(isForce);
    }

    public void setForce(boolean isForce) {
        this.isForce = isForce + "";
    }

    public String getVersion() {
        return version;
    }

    public void setVersion(String version) {
        this.version = version;
    }

    public String getUrl() {
        return url;
    }

    public void setUrl(String url) {
        this.url = url;
    }

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }
}
