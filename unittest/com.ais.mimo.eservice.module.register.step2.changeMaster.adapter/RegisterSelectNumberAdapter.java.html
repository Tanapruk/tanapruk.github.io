package com.ais.mimo.eservice.module.register.step2.changeMaster.adapter;

import android.support.v7.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.ViewGroup;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.step2.changeMaster.adapter.holder.SelectMobileNumberItemHolder;
import com.nextzy.myais.common.mvp.NextzyRecyclerAdapter;
import com.nextzy.myais.myais.R;

import java.util.List;


/**
 * Created by Nonthawit on 7/30/2016.
 */

public class RegisterSelectNumberAdapter extends NextzyRecyclerAdapter<RecyclerView.ViewHolder, RegisterSelectMobileContractor.Presenter> implements RegisterSelectMobileContractor.View{
    private final static String TAG = RegisterSelectNumberAdapter.class.getSimpleName();
    private final boolean[] selects;

    private List<MobileData> array;
    private OnSelectMobileListener selectListener;


    public interface OnSelectMobileListener{
        void onSelect( RecyclerView.ViewHolder v, String mobileNumber, int pos );
    }

    @Override
    public void createPresenter(){
        RegisterSelectMobilePresenter.createPresenter( this );
    }

    public RegisterSelectNumberAdapter(List<MobileData> array ){
        this.array = array;
        selects = new boolean[array.size()];
        for( int i = 0; i < selects.length; i++ ){
            selects[i] = false;
        }
    }

    @Override
    public int getItemCount(){
        if( array == null ){
            return 0;
        }
        return array.size();
    }


    @Override
    public RecyclerView.ViewHolder onCreateViewHolder( ViewGroup viewGroup, int viewType ){
        LayoutInflater mInflater = LayoutInflater.from( viewGroup.getContext() );

        ViewGroup vItem = (ViewGroup) mInflater.inflate( R.layout.my_ais_register_view_select_mobile_item, viewGroup, false );
        return new SelectMobileNumberItemHolder( vItem );
    }

    @Override
    public void onBindViewHolder( RecyclerView.ViewHolder viewHolder, int pos ){
        SelectMobileNumberItemHolder item = (SelectMobileNumberItemHolder) viewHolder;
        item.setText( array.get( pos ).getNumber() );
        item.setOnClickListener( v -> {
            setSelect( pos );
            if( selectListener != null ){
                selectListener.onSelect( item, array.get( pos ).getNumber(), pos );
            }
        } );
        item.setSelect( selects[pos] );
    }

    public void setOnSelectListener( OnSelectMobileListener listener ){
        selectListener = listener;
    }

    private void setSelect( int select ){
        for( int i = 0; i < selects.length; i++ ){
            if( i == select ){
                selects[i] = true;
            }else{
                selects[i] = false;
            }
        }
        notifyDataSetChanged();
    }
}
