package com.nextzy.myais.common.config.mobile;

import android.support.annotation.NonNull;

import com.nextzy.myais.common.constant.BusinessType;
import com.nextzy.myais.common.constant.NetworkType;

import java.util.Arrays;
import java.util.List;

/**
 * Created by Akexorcist on 9/7/2016 AD.
 */

@SuppressWarnings("unused")
public class MobileUtility {
    private static MobileUtility mobileUtility;

    public static MobileUtility getInstance() {
        if (mobileUtility == null) {
            mobileUtility = new MobileUtility();
        }
        return mobileUtility;
    }

    public String getBusinessTypeByNetworkType(String networkType) {
        String businessType = "";
        if (getPrepaidList().contains(networkType)) {
            businessType += BusinessType.PREPAID;
        }
        if (getPostpaidList().contains(networkType)) {
            businessType += BusinessType.POSTPAID;
        }
        if (get2GList().contains(networkType)) {
            businessType += BusinessType.OLD_2G;
        }
        if (getCorporateList().contains(networkType)) {
            businessType += BusinessType.CORPORATE;
        }
        if (getBosList().contains(networkType)) {
            businessType += BusinessType.BOS;
        }
        if (getInterstellarList().contains(networkType)) {
            businessType += BusinessType.INTERSTELLAR;
        }
        if (getUnknownList().contains(networkType)) {
            businessType = BusinessType.UNKNOWN;
        }
        if (businessType.isEmpty()) {
            businessType = BusinessType.FIBRE;
        }
        return businessType;
    }

    public MobileType getMobileTypeByNetworkType(String networkType) {
        String businessType = getBusinessTypeByNetworkType(networkType);
        if (businessType.contains(BusinessType.POSTPAID)) {
            return MobileType.POSTPAID;
        }else if (businessType.equalsIgnoreCase(NetworkType.FIBRE_FBB) || businessType.equalsIgnoreCase(NetworkType.FIBRE_FBO)){
            return MobileType.FBB;
        }else if (businessType.contains(BusinessType.PREPAID)) {
            return MobileType.PREPAID;
        }
        return null;
    }

    public boolean isNTypeIsPostpaid(String nType) {
        return MobileType.POSTPAID == getMobileTypeByNetworkType(nType);
    }

    public boolean isNTypeIsPrepaid(String nType) {
        return MobileType.PREPAID == getMobileTypeByNetworkType(nType);
    }

    public boolean isNTypeIsFbb(String nType) {
        return MobileType.FBB == getMobileTypeByNetworkType(nType);
    }

    public MobileType getMobileTypeByMobileData(String chargeType, String mobileType) {
        if ("FBB".equalsIgnoreCase(mobileType) || "FBO".equalsIgnoreCase(mobileType)) {
            return MobileType.FBB;
        } else if (chargeType.toLowerCase().contains("post")) {
            return MobileType.POSTPAID;
        } else if (chargeType.toLowerCase().contains("pre")) {
            return MobileType.PREPAID;
        }

        return null;
    }

    boolean getSwopByNetworkType(String networkType) {
        return getBosList().contains(networkType);
    }

    @NonNull
    public List<String> getPrepaidList() {
        return Arrays.asList(
                NetworkType.PREPAID_2G_CORP,
                NetworkType.PREPAID_2G,
                NetworkType.PREPAID_3G_CORP,
                NetworkType.PREPAID_3G,
                NetworkType.PREPAID_3G_BOS,
                NetworkType.PREPAID_3G_BOS2,
                NetworkType.PREPAID_INTERSTALLAR,
                NetworkType.PREPAID_INTERSTALLAR_2
        );
    }

    @NonNull
    public List<String> getPostpaidList() {
        return Arrays.asList(
                NetworkType.POSTPAID_3G_BOS_CORP,
                NetworkType.POSTPAID_3G_BOS,
                NetworkType.POSTPAID_3G_CORP,
                NetworkType.POSTPAID_3G,
                NetworkType.POSTPAID_3G_BOS_CORP_HYBRID,
                NetworkType.POSTPAID_INTERSTALLAR,
                NetworkType.POSTPAID_INTERSTALLAR_CORP,
                NetworkType.POSTPAID_2G_CORP,
                NetworkType.POSTPAID_2G,
                NetworkType.POSTPAID_2G_1800
        );
    }

    @NonNull
    private List<String> get3GList() {
        return Arrays.asList(
                NetworkType.PREPAID_3G_CORP,
                NetworkType.PREPAID_3G,
                NetworkType.PREPAID_3G_BOS,
                NetworkType.PREPAID_3G_BOS2,
                NetworkType.POSTPAID_3G_BOS_CORP,
                NetworkType.POSTPAID_3G_BOS,
                NetworkType.POSTPAID_3G_CORP,
                NetworkType.POSTPAID_3G,
                NetworkType.POSTPAID_3G_BOS_CORP_HYBRID
        );
    }

    private List<String> get2GList() {
        return Arrays.asList(
                NetworkType.PREPAID_2G_CORP,
                NetworkType.PREPAID_2G,
                NetworkType.POSTPAID_2G_CORP,
                NetworkType.POSTPAID_2G,
                NetworkType.POSTPAID_2G_1800
        );
    }

    @NonNull
    private List<String> getBosList() {
        return Arrays.asList(
                NetworkType.PREPAID_3G_BOS,
                NetworkType.PREPAID_3G_BOS2,
                NetworkType.POSTPAID_3G_BOS_CORP,
                NetworkType.POSTPAID_3G_BOS,
                NetworkType.POSTPAID_3G_BOS_CORP_HYBRID
        );
    }

    @NonNull
    private List<String> getCorporateList() {
        return Arrays.asList(
                NetworkType.PREPAID_2G_CORP,
                NetworkType.PREPAID_3G_CORP,
                NetworkType.POSTPAID_3G_BOS_CORP,
                NetworkType.POSTPAID_3G_CORP,
                NetworkType.POSTPAID_3G_BOS_CORP_HYBRID,
                NetworkType.POSTPAID_INTERSTALLAR_CORP,
                NetworkType.POSTPAID_2G_CORP
        );
    }

    @NonNull
    private List<String> getInterstellarList() {
        return Arrays.asList(
                NetworkType.POSTPAID_INTERSTALLAR,
                NetworkType.POSTPAID_INTERSTALLAR_CORP,
                NetworkType.PREPAID_INTERSTALLAR,
                NetworkType.PREPAID_INTERSTALLAR_2
        );
    }

    @NonNull
    private List<String> getUnknownList() {
        return Arrays.asList(
                NetworkType.UNKOWN_1
        );
    }
}
