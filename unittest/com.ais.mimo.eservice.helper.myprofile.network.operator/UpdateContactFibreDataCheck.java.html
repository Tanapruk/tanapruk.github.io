package com.ais.mimo.eservice.helper.myprofile.network.operator;

import com.ais.mimo.eservice.helper.myprofile.network.exception.UpdateContactMobileNumberException;
import com.ais.mimo.eservice.helper.myprofile.network.response.UpdateContactMobileNumberResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public class UpdateContactFibreDataCheck extends ResultHasDataCheck<UpdateContactMobileNumberResult, UpdateContactMobileNumberException> {
    @Override
    protected UpdateContactMobileNumberException getThrowable(String message) {
        return new UpdateContactMobileNumberException(message);
    }

    @Override
    public Single<UpdateContactMobileNumberResult> call(UpdateContactMobileNumberResult result) {
        if (isNot20000HasNoData(result)) {
            return Single.error(new UpdateContactMobileNumberException(result.getResultDescription()));
        }
        return Single.just(result);
    }
}
