package com.ais.mimo.eservice.helper.myprofile.network.response;

import com.nextzy.myais.common.network.model.GenericData;

/**
 * Created by Akexorcist on 11/22/2016 AD.
 */

public class BillingCycleData extends GenericData {
    String object;
    String paymentDueDate;
    CurrentBillingCycle currentBillingCycle;

    public BillingCycleData() {
    }

    public String getObject() {
        return object;
    }

    public void setObject(String object) {
        this.object = object;
    }

    public String getPaymentDueDate() {
        return paymentDueDate;
    }

    public void setPaymentDueDate(String paymentDueDate) {
        this.paymentDueDate = paymentDueDate;
    }

    public CurrentBillingCycle getCurrentBillingCycle() {
        return currentBillingCycle;
    }

    public void setCurrentBillingCycle(CurrentBillingCycle currentBillingCycle) {
        this.currentBillingCycle = currentBillingCycle;
    }

    public static class CurrentBillingCycle {
        String code;
        String text;
        String startDate;
        String endDate;

        public CurrentBillingCycle() {
        }

        public String getCode() {
            return code;
        }

        public void setCode(String code) {
            this.code = code;
        }

        public String getText() {
            return text;
        }

        public void setText(String text) {
            this.text = text;
        }

        public String getStartDate() {
            return startDate;
        }

        public void setStartDate(String startDate) {
            this.startDate = startDate;
        }

        public String getEndDate() {
            return endDate;
        }

        public void setEndDate(String endDate) {
            this.endDate = endDate;
        }
    }
}
