package com.ais.mimo.eservice.helper.myprofile.network.response;

import com.google.gson.annotations.SerializedName;
import com.nextzy.myais.common.network.model.GenericData;

/**
 * Created by Akexorcist on 11/22/2016 AD.
 */

public class BillMediaData extends GenericData {
    String object;
    String mediaOption;
    String language;
    @SerializedName("isAppliedEStatement")
    boolean isAppliedEStatement;
    String email;
    String mobileNumber;

    public BillMediaData() {
    }

    public String getObject() {
        return object;
    }

    public void setObject(String object) {
        this.object = object;
    }

    public String getMediaOption() {
        return mediaOption;
    }

    public void setMediaOption(String mediaOption) {
        this.mediaOption = mediaOption;
    }

    public String getLanguage() {
        return language;
    }

    public void setLanguage(String language) {
        this.language = language;
    }

    public boolean isAppliedEstatement() {
        return isAppliedEStatement;
    }

    public void setAppliedEstatement(boolean appliedEStatement) {
        isAppliedEStatement = appliedEStatement;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public String getMobileNumber() {
        return mobileNumber;
    }

    public void setMobileNumber(String mobileNumber) {
        this.mobileNumber = mobileNumber;
    }
}
