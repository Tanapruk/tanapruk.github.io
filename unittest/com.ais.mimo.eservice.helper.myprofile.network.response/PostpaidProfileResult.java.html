package com.ais.mimo.eservice.helper.myprofile.network.response;

/**
 * Created by Akexorcist on 11/23/2016 AD.
 */

public class PostpaidProfileResult {
    private PostpaidInfoResult postpaidInfoResult;
    private BillingCycleResult billingCycleResult;
    private BillMediaResult billMediaResult;
    private PostpaidBillingAddressResult postpaidBillingAddressResult;
    private AddressRemarkResult addressRemarkResult;

    public PostpaidProfileResult(PostpaidInfoResult postpaidInfoResult,
                                 BillingCycleResult billingCycleResult,
                                 BillMediaResult billMediaResult,
                                 PostpaidBillingAddressResult postpaidBillingAddressResult,
                                 AddressRemarkResult addressRemarkResult) {
        this.postpaidInfoResult = postpaidInfoResult;
        this.billingCycleResult = billingCycleResult;
        this.billMediaResult = billMediaResult;
        this.postpaidBillingAddressResult = postpaidBillingAddressResult;
        this.addressRemarkResult = addressRemarkResult;
    }

    public PostpaidInfoResult getPostpaidInfoResult() {
        return postpaidInfoResult;
    }

    public BillingCycleResult getBillingCycleResult() {
        return billingCycleResult;
    }

    public BillMediaResult getBillMediaResult() {
        return billMediaResult;
    }

    public PostpaidBillingAddressResult getPostpaidBillingAddressResult() {
        return postpaidBillingAddressResult;
    }

    public AddressRemarkResult getAddressRemarkResult() {
        return addressRemarkResult;
    }
}
