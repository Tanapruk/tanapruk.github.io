package com.ais.mimo.eservice.module.myaccount.activity;

import com.ais.mimo.eservice.helper.login.network.LoginManager;
import com.ais.mimo.eservice.module.myaccount.event.GoToEditAccountFragmentEvent;
import com.ais.mimo.eservice.module.myaccount.event.GoToEditEmailFragmentEvent;
import com.ais.mimo.eservice.module.myaccount.event.GoToEditPasswordFragmentEvent;
import com.ais.mimo.eservice.module.myaccount.event.GotoLoginPageEvent;
import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.eventlog.EventLogBlob;
import com.nextzy.myais.common.eventlog.EventLogParam;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;

/**
 * Created by TheKhaeng on 9/20/2016.
 */

public class MyAccountPresenter extends NextzyActivityPresenter<MyAccountContractor.View> implements MyAccountContractor.Presenter {

    public MyAccountPresenter(MyAccountContractor.View view) {
        super(view);
    }

    public static MyAccountPresenter createPresenter(MyAccountContractor.View view) {
        return new MyAccountPresenter(view);
    }

    @Override
    public void onViewCreate() {
        RxBus.get().register(this);
    }

    @Override
    public void onViewDestroy() {
        RxBus.get().unregister(this);
    }

    @Override
    public void onViewStart() {
        // for handle network error
        super.onViewStart();
        EventLogBlob.getInstance().addEventLog(
                LoginManager.SERVICE_NAME_VERIFY_MYAIS,
                new EventLogParam("APP_VIEW_MYAIS_PROFILE", "5010", "5108")
        );
    }

    @Override
    public void onViewStop() {
        // for handle network error
        super.onViewStop();
    }


    @Subscribe
    public void onGoToEditAccountFragmentEvent(GoToEditAccountFragmentEvent event) {
        getView().openEditAccountFragment();
    }

    @Subscribe
    public void onGoToEditEmailFragmentEvent(GoToEditEmailFragmentEvent event) {
        getView().openEditEmailFragment();
    }

    @Subscribe
    public void onGoToEditPasswordFragmentEvent(GoToEditPasswordFragmentEvent event) {
        getView().openEditPasswordFragment();
    }

    @Subscribe
    public void onGotoLoginPageEvent(GotoLoginPageEvent event){
        getView().openLoginPage();
    }

}
