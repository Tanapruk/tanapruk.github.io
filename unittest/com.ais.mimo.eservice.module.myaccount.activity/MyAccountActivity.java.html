package com.ais.mimo.eservice.module.myaccount.activity;


import android.os.Bundle;
import android.support.v4.app.Fragment;
import android.support.v7.widget.Toolbar;

import com.ais.mimo.eservice.module.login.activity.LoginActivity;
import com.ais.mimo.eservice.module.myaccount.fragment.account.MyAccountEditAccountFragment;
import com.ais.mimo.eservice.module.myaccount.fragment.email.MyAccountEditEmailFragment;
import com.ais.mimo.eservice.module.myaccount.fragment.main.MyAccountMainFragment;
import com.ais.mimo.eservice.module.myaccount.fragment.password.MyAccountEditPasswordFragment;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.common.delegate.FragmentHelperDelegate;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.network.tool.NextzyWebkitCookieJar;
import com.nextzy.myais.common.utility.DataUpdatePreference;
import com.nextzy.myais.myais.R;

public class MyAccountActivity extends NextzyMvpActivity<MyAccountContractor.Presenter> implements MyAccountContractor.View {
    private final static String TAG = MyAccountActivity.class.getSimpleName();
    private Toolbar tbHeader;
    private FragmentHelperDelegate fragmentHelperDelegate = new FragmentHelperDelegate(this);

    /*************************/
    /** Activity life cycle **/
    /*************************/
    //<editor-fold desc="Activity life cycle folding">
    @Override
    public int getLayoutView() {
        return R.layout.my_ais_activity_my_ais_account;
    }

    @Override
    protected void createPresenter() {
        MyAccountPresenter.createPresenter(this);
    }

    @Override
    public void bindView() {
        fragmentHelperDelegate.bind(R.id.contentContainer);
        tbHeader = (Toolbar) findViewById(R.id.my_ais_library_simple_toolbar_tb_header);
    }

    @Override
    public void setupView() {
        tbHeader.setNavigationOnClickListener(v -> {
            if (v == tbHeader) {
                onBackPressed();
            }
        });
        tbHeader.setTitle(R.string.my_ais_account_title);
        Mobile.getInstance().logEvent("APP_VIEW_MYAIS_ACCOUNT", "5012", "5124");
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        getSupportFragmentManager().beginTransaction()
                .add(R.id.contentContainer,
                        MyAccountMainFragment.newInstance(),
                        MyAccountMainFragment.TAG)
                .commit();
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {
    }

    @Override
    public void onBackPressed() {
        Fragment fragment = fragmentHelperDelegate.getCurrentFragment();
        if (fragment instanceof MyAccountEditEmailFragment
                && fragment.isVisible()
                && !((MyAccountEditEmailFragment) fragment).isCanBackPressed()) {
            ((MyAccountEditEmailFragment) fragment).backStep();
        } else {
            super.onBackPressed();
        }
    }

    @Override
    public void openEditAccountFragment() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                MyAccountEditAccountFragment.newInstance(),
                MyAccountEditAccountFragment.TAG);
    }

    @Override
    public void openEditEmailFragment() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                MyAccountEditEmailFragment.newInstance(),
                MyAccountEditEmailFragment.TAG);
    }

    @Override
    public void openEditPasswordFragment() {
        fragmentHelperDelegate.openFragmentSlideInOut(
                MyAccountEditPasswordFragment.newInstance(),
                MyAccountEditPasswordFragment.TAG);
    }

    @Override
    public void openLoginPage() {
        DataUpdatePreference.getInstance().clear(this);
        NextzyWebkitCookieJar.getInstance().clearCookie(this);
        Bundle bundle = new Bundle();
        bundle.putBoolean(LoginActivity.SHOULD_BYPASS, false);
        openActivity(LoginActivity.class, bundle, true, true);
    }

    //</editor-fold>
}
