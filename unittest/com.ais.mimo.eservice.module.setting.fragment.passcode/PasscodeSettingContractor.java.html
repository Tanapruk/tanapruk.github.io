package com.ais.mimo.eservice.module.setting.fragment.passcode;

import com.nextzy.myais.common.passcode.network.response.BindFingerScanResult;

    public interface Presenter extends NextzyContractorPresenter{

        void dismissAllDialogs();

        void dismissDialog();

        void dismissDialog(int delay);

        void removeAllPasscodeData();

        boolean isFirstEnablePasscodeLock();

        void openFirstNewPasscodeActivity();
    }
}
