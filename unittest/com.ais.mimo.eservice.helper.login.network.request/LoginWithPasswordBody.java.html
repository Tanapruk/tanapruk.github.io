package com.ais.mimo.eservice.helper.login.network.request;

/**
 * Created by Oat on 10/13/2016 AD.
 */

public class LoginWithPasswordBody {

    String user;
    String password;
    String referenceId;
    String captcha;

    public LoginWithPasswordBody(String user, String password, String referenceId, String captcha) {
        this.user = user;
        this.password = password;
        this.referenceId = referenceId;
        this.captcha = captcha;
    }

    public String getUser() {
        return user;
    }

    public void setUser(String user) {
        this.user = user;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    public String getReferenceId() {
        return referenceId;
    }

    public void setReferenceId(String referenceId) {
        this.referenceId = referenceId;
    }

    public String getCaptcha() {
        return captcha;
    }

    public void setCaptcha(String captcha) {
        this.captcha = captcha;
    }
}
