package com.ais.mimo.eservice.helper.appupdate.network.operator;

import com.ais.mimo.eservice.helper.appupdate.network.exception.AppUpdateFailureException;
import com.ais.mimo.eservice.helper.appupdate.network.response.AppUpdateResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/25/2016.
 */
public class AppUpdateResultCheck extends ResultHasDataCheck<AppUpdateResult, AppUpdateFailureException> {
    @Override
    protected AppUpdateFailureException getThrowable(String message) {
        return new AppUpdateFailureException();
    }

    @Override
    public Single<AppUpdateResult> call(AppUpdateResult result) {
        //when data is valid it will have a message
        if (result != null && result.getMessage() != null) {
            return Single.just(result);
        }

        return super.call(result);
    }
}
