package com.ais.mimo.eservice.module.register.utility;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.utility.model.MobileTypeGroup;
import com.nextzy.myais.common.config.mobile.MobileType;
import com.nextzy.myais.common.config.mobile.MobileUtility;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by TheKhaeng on 10/15/2016.
 */

public class RegisterConverterUtility {
    private static RegisterConverterUtility util;

    public static RegisterConverterUtility getInstance() {
        if (util == null) {
            util = new RegisterConverterUtility();
        }
        return util;
    }

    public MobileTypeGroup groupMobileNumberByType(List<MobileData> array) {
        MobileTypeGroup model = new MobileTypeGroup();

        for (MobileData mobile : array) {
            MobileType type = MobileUtility.getInstance().getMobileTypeByMobileData(
                    mobile.getChargeType(), mobile.getMobileType());

            if (MobileType.FBB.equals(type)) {
                model.addFbbMobile(mobile);
            } else if (MobileType.PREPAID.equals(type)) {
                model.addPrepaidMobile(mobile);
            } else if (MobileType.POSTPAID.equals(type)) {
                model.addPostpaidMobile(mobile);
            }
        }

        return model;
    }
    public List<MobileData> getPostpaidGroup(List<MobileData> array) {
        return groupMobileNumberByType(array).getPostpaidList();
    }

    public List<MobileData> getPrepaidGroup(List<MobileData> array) {
        return groupMobileNumberByType(array).getPrepaidList();
    }

    public List<MobileData> getFbbGroup(List<MobileData> array) {
        return groupMobileNumberByType(array).getFbbList();
    }

    public List<String> conventMyNumberToStringArray(List<MobileData> myNumber) {
        ArrayList<String> mobiles = new ArrayList<>();
        for (MobileData MobileData : myNumber) {
            mobiles.add(MobileData.getNumber());
        }
        return mobiles;
    }

}
