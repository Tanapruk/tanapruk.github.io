package com.ais.mimo.eservice.module.register.step4.addmobilenumber.model;


import com.ais.mimo.eservice.helper.login.network.response.MobileData;

import javax.annotation.Nonnull;

import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.Setter;

/**
 * Created by TheKhaeng on 11/11/2016.
 */

@EqualsAndHashCode
public class AddMobileModel {
    @Getter
    private MobileData mobileData;
    private State state = new State();

    public AddMobileModel(@Nonnull MobileData mobileData, boolean isClick) {
        this.mobileData = mobileData;
        setClickRequestOtp(isClick);
    }

    public void setClickRequestOtp(boolean isClick ){
        state.setClick(isClick);
    }

    public String getNumber(){
        return mobileData.getNumber();
    }

    public boolean isClickRequestOtp(){
        return state.isClick();
    }


    private class State {
        @Getter @Setter
        private boolean isClick = false;
    }

}
