package com.ais.mimo.eservice.module.register.step3;

import android.os.Bundle;
import android.view.View;
import android.webkit.WebView;

import com.ais.mimo.eservice.module.register.base.RegisterBaseFragment;
import com.ais.mimo.eservice.module.register.widget.MyAisRegister2ButtonView;
import com.nextzy.myais.myais.R;


public class RegisterStep3Fragment extends RegisterBaseFragment<RegisterStep3Contractor.Presenter> implements RegisterStep3Contractor.View{
    public final static String TAG = RegisterStep3Fragment.class.getSimpleName();
    public static final String KEY_TERM = "key_term";
    private MyAisRegister2ButtonView vBtnNextCancel;
    private WebView webView;
    private String term;


    public RegisterStep3Fragment(){
        super();
    }

    public static RegisterStep3Fragment newInstance( Bundle data ){
        RegisterStep3Fragment fragment = new RegisterStep3Fragment();
        fragment.setArguments( data );
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter(){
        RegisterStep3Presenter.createPresenter( this );
    }

    @Override
    public void restoreArgument( Bundle bundle ){

    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView( View view ){
        webView = (WebView) view.findViewById( R.id.my_ais_regis_webview );
        vBtnNextCancel = (MyAisRegister2ButtonView) view.findViewById( R.id.my_ais_register_btn_next_cancel);
    }

    @Override
    public void setupView(){
        setIndexIndicator( 2 );
        setBackgroundWhite();
        vBtnNextCancel.setBtnFirstText( getContext().getString( R.string.my_ais_register_decline ) );
        vBtnNextCancel.setBtnSecondText( getContext().getString( R.string.my_ais_register_accept ) );
        vBtnNextCancel.setOnFirstButtonClickListener( v -> clickCancel());
        vBtnNextCancel.setOnSecondButtonClickListener( v ->
                getPresenter().goToFragmentStep4());
    }

    @Override
    public void initialize(){
        getPresenter().requestTerm();
    }


    @Override
    public void restoreView(){
        webView.loadDataWithBaseURL( "", term, "text/html", "UTF-8", "" );
    }

    /**
     * onCreateView
     **/
    @Override
    public int getLayoutView(){
        return R.layout.my_ais_register_fragment_step3;
    }

    /**
     * State
     **/
    @Override
    public void saveInstanceState( Bundle outState ){
        // Save Instance State here
        outState.putString(KEY_TERM, term);
    }

    @Override
    public void restoreInstanceState( Bundle savedInstanceState ){
        // Restore instance state here
        term = savedInstanceState.getString(KEY_TERM);
    }
    //</editor-fold>

    @Override
    public void showTerm( String term ){
        this.term = term;
        webView.loadDataWithBaseURL( "", term, "text/html", "UTF-8", "" );
        dismissLoadingNoFade(1000);
    }


}
