    @Override
    public void gotoUrlScheme(int currentTab, MyAisMenu.SubMenu subMenu) {
        setCurrentTab(currentTab);
        Observable.empty().delay(300, TimeUnit.MILLISECONDS).doOnCompleted(
                () -> {
                    final Fragment currentFragment = tabMenuAdapter.getFragment(currentTab);
                    if (currentFragment instanceof NextzyTabListFragment) {
                        ((NextzyTabListFragment) currentFragment).gotoMenuByUrlScheme(subMenu);
                    }
                }
        ).subscribe();
    }

    private void updateUserAccountGeneric(List<MobileContact> mobileContactList, boolean shouldReload) {
        List<MyAisToolbarUserDetailSwitchAccountMenu.ContentItem> contentItemList = UserAccountConverter.getInstance().convertUserAccountToSwitchAccountMenuItem(mobileContactList);
        String mobileNumber = Mobile.getInstance().getMobileNumber();
        if (shouldReload) {
            tbUserDetail.setMenuItem(contentItemList, mobileNumber);
        } else {
            tbUserDetail.setMenuItemWithoutReloadingMenu(contentItemList, mobileNumber);
        }
    }

    @Override
    public void updateUserAccount(List<MobileContact> mobileContactList) {
        updateUserAccountGeneric(mobileContactList, true);
    }

    @Override
    public void updateUserAccountWithoutReloading(List<MobileContact> mobileContactList) {
        updateUserAccountGeneric(mobileContactList, false);
    }

    @Override
    public void requestCameraPermission() {
        NextzyPermission.requestPermission(this, permissionResult -> {
            if (permissionResult.areAllPermissionsGranted()) {
                openActivityOnResult(QrScannerActivity.class, QrScannerActivity.REQUEST_CODE);
            } else {
                Observable.empty()
                        .delay(500, TimeUnit.MILLISECONDS)
                        .doOnCompleted(this::showCameraPermissionDenied)
                        .subscribe();
            }
        }, Manifest.permission.CAMERA);
    }

    @Override
    public void onTabSelected(TabLayout.Tab tab) {
        MyAisMenu.Menu menu = tabMenuAdapter.getMenu(tab.getPosition());
        tvTitle.setText(NextzyUtility.getInstance().getLocalMessage(menu.getNameEn(), menu.getNameTh(), getCurrentLanguage()));
        if (vsTitle.getDisplayedChild() == 0) {
            ivTitleSecond.setImageResource(menu.getImage());
        } else {
            ivTitleFirst.setImageResource(menu.getImage());
        }
    }

    @Override
    public void onTabUnselected(TabLayout.Tab tab) {
        MyAisMenu.Menu menu = tabMenuAdapter.getMenu(tab.getPosition());
        if (vsTitle.getDisplayedChild() == 0) {
            ivTitleFirst.setImageResource(menu.getImage());
        } else {
            ivTitleSecond.setImageResource(menu.getImage());
        }
        vsTitle.showNext();

        previousPos = tab.getPosition();
    }

    @Override
    public void onTabReselected(TabLayout.Tab tab) {
    }

    @Override
    protected void onActivityResult(int requestCode, int resultCode, Intent data) {
        if (isFromQrUrlScheme(requestCode, resultCode)) {
            String url = data.getStringExtra(QrScannerActivity.KEY_URL);
            setupUrlScheme(url);
            moveToPrivilegeTab();
        } else if (isFromAisUrlScheme(requestCode, resultCode)) {
            String url = data.getStringExtra(MyAisWebViewActivity.KEY_URL);
            setupUrlScheme(url);
            getPresenter().openUrlScheme();
        } else {
            super.onActivityResult(requestCode, resultCode, data);
        }
    }

    @Override
    public void onAfterLocaleChanged() {
        if (getSlidingMenu().isMenuShowing()) {
            toggle();
        }
        getPresenter().getUserProfile();
    }

    private boolean isFromAisUrlScheme(int requestCode, int resultCode) {
        return requestCode == MyAisWebViewActivity.REQUEST_CODE && resultCode == MyAisWebViewActivity.RESULT_URL_SCHEME;
    }

    private boolean isFromQrUrlScheme(int requestCode, int resultCode) {
        return requestCode == QrScannerActivity.REQUEST_CODE && resultCode == QrScannerActivity.RESULT_URL_SCHEME;
    }


    private void setupUrlScheme(String url) {
        if (url == null) {
            return;
        }
        if (url.contains("#")) {
            url = url.split("#")[0];
        }
        UrlSchemeUtils.getInstance().setupUrlScheme(Uri.parse(url));
    }

    private void moveToPrivilegeTab() {
        // TODO Fix this because we can't access to privilege module
        setCurrentTabItem(TabServiceMenuKey.KEY_SERVICE.toString());
    }

    private void setCurrentTabItem(String key) {
        int index = tabMenuAdapter.getTabIndexByKey(key);
        vpContent.setCurrentItem(index, true);
    }

    public void showCameraPermissionDenied() {
        showAlertDialogDismissOnClick(getString(com.nextzy.myais.common.R.string.my_ais_library_qr_camera_permission_denied));
    }
