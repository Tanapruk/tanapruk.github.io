package com.nextzy.myais.promotion.module.roaming.activity;

import com.hwangjr.rxbus.RxBus;
import com.hwangjr.rxbus.annotation.Subscribe;
import com.nextzy.myais.common.mvp.NextzyActivityPresenter;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageConfirmEvent;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageDateChoosingEvent;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageDetailEvent;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageListEvent;
import com.nextzy.myais.promotion.helper.roaming.event.GotoRoamingPackageRequest;

public class RoamingApplyPackagePresenter extends NextzyActivityPresenter<RoamingApplyPackageContractor.View> implements RoamingApplyPackageContractor.Presenter {
    public RoamingApplyPackagePresenter(RoamingApplyPackageContractor.View view) {
        super(view);
    }

    public static RoamingApplyPackagePresenter createPresenter(RoamingApplyPackageContractor.View view) {
        return new RoamingApplyPackagePresenter(view);
    }

    @Override
    public void onViewCreate() {
    }

    @Override
    public void onViewDestroy() {
    }

    @Override
    public void onViewStart() {
        super.onViewStart();
        RxBus.get().register(this);
    }

    @Override
    public void onViewStop() {
        super.onViewStop();
        RxBus.get().unregister(this);
    }

    @Subscribe
    public void onGotoRoamingPackageRequest(GotoRoamingPackageRequest event) {
        getView().gotoPackageRequest();
    }
    @Subscribe
    public void onGotoRoamingPackageList(GotoRoamingPackageListEvent event) {
        getView().gotoRoamingPackageList();
    }


    @Subscribe
    public void onGotoRoamingPackageDetail(GotoRoamingPackageDetailEvent event) {
        getView().gotoRoamingPackageDetail(event.getUpPackCDATA(), event.getUpSupportResult());
    }

    @Subscribe
    public void onGotoRoamingPackageDateChoosing(GotoRoamingPackageDateChoosingEvent event) {
        getView().gotoPackageDateChoosing(event.getUpPackCDATA());
    }

    @Subscribe
    public void onGotoRoamingPackageConfirm(GotoRoamingPackageConfirmEvent event) {
        getView().gotoPackageConfirm(event.getStartDateTime(), event.getEndDateTime(), event.getDt(), event.getUpPackCDATA());
    }
}
