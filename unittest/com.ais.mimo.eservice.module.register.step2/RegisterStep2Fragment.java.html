package com.ais.mimo.eservice.module.register.step2;

import android.os.Bundle;
import android.text.InputType;

import android.widget.Button;

import com.ais.mimo.eservice.module.register.base.RegisterBaseFragment;
import com.ais.mimo.eservice.module.register.widget.MyAisRegister2ButtonView;
import com.ais.mimo.eservice.module.register.widget.MyAisRegisterEditText;
import com.ais.mimo.eservice.module.register.widget.MyAisRegisterPasswordEditText;
import com.nextzy.myais.common.mvp.NextzyMvpActivity;
import com.nextzy.myais.common.widget.AISTextView;
import com.nextzy.myais.common.widget.dialog.AisDialog;
import com.nextzy.myais.myais.BuildConfig;
import com.nextzy.myais.myais.R;

import static android.view.View.GONE;
import static android.view.View.VISIBLE;

public class RegisterStep2Fragment extends RegisterBaseFragment<RegisterStep2Contractor.Presenter> implements RegisterStep2Contractor.View {
    public final static String TAG = RegisterStep2Fragment.class.getSimpleName();
    public static final String KEY_POSTPAID_GROUP = "key_postpaid_group";
    public static final String KEY_FBB_GROUP = "key_fbb_group";
    private MyAisRegister2ButtonView vTwoBtn;
    private MyAisRegisterEditText vSmsOtp;
    private MyAisRegisterPasswordEditText vPassword;
    private AISTextView tvMobileNumber;
    private AISTextView tvIdCard;
    private AISTextView tvEmail;
    private LinearLayout baseLayout;
    private LinearLayout subLayout;
    private LinearLayout containerSmsOtp;


    public RegisterStep2Fragment() {
        super();
    }

    public static RegisterStep2Fragment newInstance(Bundle data) {
        RegisterStep2Fragment fragment = new RegisterStep2Fragment();
        fragment.setArguments(data);
        return fragment;
    }

    /*************************/
    /** Fragment life cycle **/
    /*************************/
    //<editor-fold desc="Life cycle folding">

    /**
     * onCreate()
     **/
    @Override
    protected void createPresenter() {
        RegisterStep2Presenter.createPresenter(this);
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    /**
     * onViewCreated()
     **/
    @Override
    public void bindView(View view) {
        baseLayout = (LinearLayout) view.findViewById(R.id.my_ais_register_base_layout);
        subLayout = (LinearLayout) view.findViewById(R.id.my_ais_register_sub_layout);
        vTwoBtn = (MyAisRegister2ButtonView) view.findViewById(R.id.my_ais_register_btn_next_cancel);
        containerSmsOtp = (LinearLayout) view.findViewById(R.id.my_ais_register_container_otp);
        vSmsOtp = (MyAisRegisterEditText) view.findViewById(R.id.my_ais_register_edt_sms_otp);

        vPassword = (MyAisRegisterPasswordEditText) view.findViewById(R.id.my_ais_register_edt_password_repassword);
        tvMobileNumber = (AISTextView) view.findViewById(R.id.my_ais_register_tv_mobile_number);
        tvIdCard = (AISTextView) view.findViewById(R.id.my_ais_register_tv_id_card);
        tvEmail = (AISTextView) view.findViewById(R.id.my_ais_register_tv_email);

        // TODO1: 10/13/2016 delete fill mock to Edt
        mockFillEditText(view);
    }

    public void mockFillEditText(View view) {
        if (BuildConfig.DEBUG) {
            Button btnMock = (Button) view.findViewById(R.id.mock_number);
            btnMock.setVisibility(VISIBLE);
            btnMock.setOnClickListener(view1 -> {
                vSmsOtp.setText("1234");
                vPassword.setPassText("@Test12346");
                vPassword.setRePassText("@Test12346");
            });
        }
    }

    @Override
    public void setupView() {
        setupClickHideKeyboard(baseLayout);
        setBackgroundWhite();
        setIndexIndicator(1);
        tvMobileNumber.setText(getArgumentsStringMobileNumber(), FORMAT_MOBILE_NUMBER);
        tvIdCard.setText(getArgumentsStringIdCard(), FORMAT_ID_CARD);
        tvEmail.setText(getArgumentsStringEmail());
        vSmsOtp.setInputType(InputType.TYPE_CLASS_NUMBER | InputType.TYPE_NUMBER_VARIATION_PASSWORD);
        if( getPresenter().isActiveFibreNumber()){
            vSmsOtp.setHeaderText(getString(R.string.my_ais_register_receive_email_otp));
        }else{
            vSmsOtp.setHeaderText(getString(R.string.my_ais_register_receive_sms_otp));
        }

        vTwoBtn.setBtnFirstText(getContext().getString(R.string.my_ais_register_back));
        vTwoBtn.setBtnSecondText(getContext().getString(R.string.my_ais_register_next));
        vTwoBtn.setOnFirstButtonClickListener(v -> {
            clickCancel();
        });
        vTwoBtn.setOnSecondButtonClickListener(v -> clickNext());

        vPassword.setOnClickTermListener(v -> showPasswordTerm());

        if (getPresenter().isShowOtpEditText()) {
            containerSmsOtp.setVisibility(VISIBLE);
        }else{
            containerSmsOtp.setVisibility(GONE);
        }
    }

    }

    }

        // Save Instance State here
    }

    public void restoreInstanceState(Bundle savedInstanceState) {
        // Restore instance state here

    @Override
    public void showDefaultOtpEdt() {
        vSmsOtp.showDefault();
    }

    @Override
    public void showDefaultPasswordEdt() {
        vPassword.showDefault();
    }

    @Override
    public void showFormatErrorOtpEdt() {
        vSmsOtp.showError();
        String error;
        if (vSmsOtp.getStringFromEditText().isEmpty()) {
            error = getString(R.string.my_ais_register_dialog_otp_none);
        } else {
            error = getString(R.string.my_ais_register_dialog_otp_fault);
        }
        ((NextzyMvpActivity) getActivity())
                .showAlertDialogDismissOnClick(error);
    }

    @Override
    public void showFormatErrorPasswordEdt() {
        vPassword.showError();
        String error;
        if (vPassword.getPasswordStringFromEditText().isEmpty()
                || vPassword.getRePasswordStringFromEditText().isEmpty()) {
            error = getString(R.string.my_ais_register_dialog_password_none);
        } else {
            error = getString(R.string.my_ais_register_dialog_password_fault);
        }
        ((NextzyMvpActivity) getActivity())
                .showAlertDialogDismissOnClick(error);
    }

    @Override
    public void showNotMatchErrorPasswordEdt() {
        vPassword.showError();
        String error = getString(R.string.my_ais_register_dialog_password_not_match);
        ((NextzyMvpActivity) getActivity())
                .showAlertDialogDismissOnClick(error);
    }

    @Override
    public String getPasswordStringFromEditText() {
        return vPassword.getPasswordStringFromEditText();
    }

    @Override
    public String getRePasswordStringFromEditText() {
        return vPassword.getRePasswordStringFromEditText();
    }

    @Override
    public void showResponseException() {
        ((NextzyMvpActivity) getActivity())
                .showAlertDialogDismissOnClick(R.string.my_ais_server_unavailable);
    }


    /********************/
    /** Private method **/
    /********************/
    //<editor-fold desc="Private function folding">
    private void showPasswordTerm() {
        AisDialog.getInstance().showHtmlDialog(
                getFragmentManager(),
                getString(R.string.my_ais_register_dialog_password_title),
                getString(R.string.my_ais_register_dialog_password_info),
                getContext()
        );
    }

    private void clickNext() {
        getPresenter().validateRequestVerifyOtp(
                vSmsOtp.getStringFromEditText(),
                vPassword.getPasswordStringFromEditText(),
                vPassword.getRePasswordStringFromEditText()
        );

    }
    //</editor-fold>
}
