package com.ais.mimo.eservice.module.login.fragment;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import com.ais.mimo.eservice.module.login.LoginKey;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.myais.R;

/**
 * Created by trusttanapruk on 8/17/2016.
 */
public class LoginSecondaryFragment extends LoginBaseFragment {
    private static final String KEY_SECONDARY_TYPE = "SECONDARY_TYPE";
    private EditText etMobileNumber;
    private Button btnPrimary;
    private Button btnSecondary;
    private String secondaryType;

    public static LoginSecondaryFragment newInstance(String secondaryType) {
        Bundle args = new Bundle();
        args.putString(KEY_SECONDARY_TYPE, secondaryType);
        LoginSecondaryFragment fragment = new LoginSecondaryFragment();
        fragment.setArguments(args);
        return fragment;
    }

    public void onSecondaryOtpExists(String secondaryType) {
        getPresenter().onSecondaryTypeUpdate(secondaryType);
    }

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_login_fragment_secondary;
    }

    @Override
    public void bindView(View view) {
        etMobileNumber = (EditText) view.findViewById(R.id.my_ais_login_secondary_et_mobile_number);
        btnPrimary = (Button) view.findViewById(R.id.my_ais_login_btn_primary);
        btnSecondary = (Button) view.findViewById(R.id.my_ais_login_btn_secondary);
        bindTermForgotPasswordLanguageBar(view);
    }

    @Override
    protected int idTermAndConditionView() {
        return R.id.my_ais_login_tv_secondary_term_and_condition;
    }

    @Override
    protected int idTvLanguageBar() {
        return R.id.my_ais_login_secondary_view_change_language;
    }

    @Override
    public void setupView() {
        etMobileNumber.setText(Mobile.getInstance().getMobileNumber());
        btnPrimary.setOnClickListener(v -> getPresenter().onRequestOtpButtonClick(LoginKey.TYPE_DEVICE_PRIMARY));
        btnSecondary.setOnClickListener(v -> getPresenter().onRequestOtpButtonClick(secondaryType));
    }

    @Override
    public void setSecondaryButtonTextEmail() {
        btnSecondary.setText(getString(R.string.my_ais_login_secondary_email));
    }

    @Override
    public void setSecondaryButtonTextSecondary() {
        btnSecondary.setText(getString(R.string.my_ais_login_secondary_secondary));
    }

    @Override
    public void restoreArgument(Bundle bundle) {
        secondaryType = bundle.getString(KEY_SECONDARY_TYPE);
    }

    @Override
    public void initialize() {
        onSecondaryOtpExists(secondaryType);
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {
        secondaryType = savedInstanceState.getString(KEY_SECONDARY_TYPE);
    }

    @Override
    public void restoreView() {
        onSecondaryOtpExists(secondaryType);
    }

    @Override
    public void saveInstanceState(Bundle outState) {
        outState.putString(KEY_SECONDARY_TYPE, secondaryType);

    }

}
