package com.ais.mimo.eservice.module.login.fragment;

import android.os.Bundle;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;
import android.widget.EditText;

import com.ais.mimo.eservice.helper.devtool.NextzyInjector;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.myais.R;

/**
 * Created by trusttanapruk on 8/17/2016.
 */
public class LoginWelcomeFragment extends LoginBaseFragment implements NextzyInjector {
    private EditText etAccount;
    private Button btnLoginNext;
    private ViewGroup viewGroupContainer;

    public static LoginWelcomeFragment newInstance() {
        return new LoginWelcomeFragment();
    }

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_login_fragment_welcome;
    }

    @Override
    public void bindView(View view) {
        viewGroupContainer = (ViewGroup) view.findViewById(R.id.my_ais_login_layout_mobile_number);
        etAccount = (EditText) view.findViewById(R.id.my_ais_login_welcome_et_account);
        btnLoginNext = (Button) view.findViewById(R.id.my_ais_login_welcome_btn_ok);
        bindTermForgotPasswordLanguageBar(view);
    }

    @Override
    protected int idTermAndConditionView() {
        return R.id.my_ais_login_tv_login_term_and_condition;
    }

    @Override
    protected int idTvLanguageBar() {
        return R.id.my_ais_login_welcome_view_change_language;
    }

    @Override
    public void setupView() {
        btnLoginNext.setOnClickListener(v -> getPresenter().onLoginWelcomeOkButtonClick(etAccount.getText().toString()));
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        if (Mobile.getInstance().getMobileNumber() != null) {
            etAccount.setText(Mobile.getInstance().getMobileNumber());
        }

    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    @Override
    public void showEmptyMobileNumberInvalid() {
        showAlertDialogDismissOnClick(getString(R.string.my_ais_login_empty_phone_number));
    }

    @Override
    public void showMobileNumberInvalid() {
        showAlertDialogDismissOnClick(getString(R.string.my_ais_login_incorrect_phone_number));
    }

    @Override
    public void showNotAisMobileNumber(String message) {
        showAlertDialogDismissOnClick(message);
    }


    @Override
    public EditText getInjectEditText() {
        return etAccount;
    }
}
