package com.ais.mimo.eservice.module.login.fragment;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import com.ais.mimo.eservice.module.forgot.ForgotPasswordActivity;
import com.nextzy.myais.common.config.mobile.Mobile;
import com.nextzy.myais.myais.R;

/**
 * Created by trusttanapruk on 8/17/2016.
 */
public class LoginVerifyChoiceFragment extends LoginBaseFragment {
    private EditText etAccount;
    private Button btnGetSmsOtp;
    private Button btnPassword;

    public static LoginVerifyChoiceFragment newInstance() {
        return new LoginVerifyChoiceFragment();
    }

    @Override
    public int getLayoutView() {
        return R.layout.my_ais_login_fragment_verify_choice;
    }

    @Override
    public void bindView(View view) {
        etAccount = (EditText) view.findViewById(R.id.my_ais_login_verify_choice_et_mobile_number);
        btnGetSmsOtp = (Button) view.findViewById(R.id.my_ais_login_verify_choice_btn_get_otp);
        btnPassword = (Button) view.findViewById(R.id.my_ais_login_verify_choice_btn_password);
        bindTermForgotPasswordLanguageBar(view);
    }

    @Override
    protected int idTvLanguageBar() {
        return R.id.my_ais_login_verify_choice_view_change_language;
    }

    @Override
    protected int idTvForgetPassword() {
        return R.id.my_ais_login_verify_choice_tv_forgot_password;
    }

    @Override
    protected int idTermAndConditionView() {
        return R.id.my_ais_login_tv_login_term_and_condition;
    }

    @Override
    public void setupView() {
        btnGetSmsOtp.setOnClickListener(onSmsOtpClickListener());
        btnPassword.setOnClickListener(onPasswordClickListener());
    }

    @Override
    public void restoreArgument(Bundle bundle) {

    }

    @Override
    public void initialize() {
        setMobileNumber(Mobile.getInstance().getMobileNumber());
    }

    @Override
    public void restoreInstanceState(Bundle savedInstanceState) {

    }

    @Override
    public void restoreView() {

    }

    @Override
    public void saveInstanceState(Bundle outState) {

    }

    @Override
    public void onForgetPasswordClick() {
        String accountNumber = etAccount.getText().toString();
        Bundle bundle = new Bundle();
        bundle.putString(ForgotPasswordActivity.KEY_ACCOUNT_NUMBER, accountNumber);
        openActivity(ForgotPasswordActivity.class, bundle);
    }

    public View.OnClickListener onSmsOtpClickListener() {
        return v -> getPresenter().onVerifyChoiceChooseGetOtp();
    }

    public View.OnClickListener onPasswordClickListener() {
        return v -> getPresenter().onVerifyChoiceChoosePassword();
    }

    public void setMobileNumber(String mobileNumber) {
        etAccount.setText(mobileNumber);
    }

}
