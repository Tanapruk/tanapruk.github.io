    private boolean throwIfNull(View view) {
        if (view == null) {
            throw new NullPointerException("Your haven't binded this button");
        } else {
            return true;
        }
    }

    public void bindTermForgotPasswordLanguageBar(View view) {
        if (idTermAndConditionView() != 0) {
            tvTerm = (TermAndConditionView) view.findViewById(idTermAndConditionView());
            setupTermOnClick();
        }

        if (idTvForgetPassword() != 0) {
            tvForgetPassword = (TextView) view.findViewById(idTvForgetPassword());
            setupForgotPasswordOnClick();
        }

        if (idTvLanguageBar() != 0) {
            btnLanguageBar = (LoginLanguageBar) view.findViewById(idTvLanguageBar());
            setupLoginLanguageBar();
        }
    }

    protected int idTvLanguageBar() {
        return 0;
    }

    protected int idTvForgetPassword() {
        return 0;
    }

    protected int idTermAndConditionView() {
        return 0;
    }

    @Override
    public void setLanguage(String language) {
        ((LoginActivity) getActivity()).setLanguage(language);
    }

    private void setupTermOnClick() {
        throwIfNull(tvTerm);
        tvTerm.setOnClickListener(v -> onTermAndConditionClick());
    }

    private void setupForgotPasswordOnClick() {
        throwIfNull(tvForgetPassword);
        tvForgetPassword.setOnClickListener(v -> onForgetPasswordClick());
    }

    public void onTermAndConditionClick() {
        openActivity(TermActivity.class);
    }

    public void onForgetPasswordClick() {
        openActivity(ForgotPasswordActivity.class);
    }

    private void setupLoginLanguageBar() {
        throwIfNull(btnLanguageBar);
        btnLanguageBar.setLoginLanguageBarListener(this);
    }

    protected void hideForgotPasswordButton() {
        tvForgetPassword.setVisibility(View.GONE);
    }

    protected void showForgotPasswordButton() {
        tvForgetPassword.setVisibility(View.VISIBLE);
    }

    @Override
    public void removePasscodeAndMasterNumberAndPrivateIdAndToken() {
        preference.removeAllPasscodeData();
    }

    @Override
    public void showIsFibreButCannotLogin() {
        showAlertDialogDismissOnClick(getString(R.string.my_ais_login_dialog_alert_fibre));
    }

    @Override
    public void showResponseException(String message) {
        if (getActivity() instanceof NextzyMvpActivity) {
            ((NextzyMvpActivity) getActivity())
                    .showAlertDialogDismissOnClick(message);
        }
    }

}
