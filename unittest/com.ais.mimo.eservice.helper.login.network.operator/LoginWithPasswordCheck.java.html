package com.ais.mimo.eservice.helper.login.network.operator;

import com.ais.mimo.eservice.helper.login.network.exception.LoginWithPasswordCaptchaException;
import com.ais.mimo.eservice.helper.login.network.exception.LoginWithPasswordFailedException;
import com.ais.mimo.eservice.helper.login.network.response.LoginWithPasswordResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public class LoginWithPasswordCheck extends ResultHasDataCheck<LoginWithPasswordResult, LoginWithPasswordFailedException> {
    @Override
    protected LoginWithPasswordFailedException getThrowable(String message) {
        return new LoginWithPasswordFailedException(message);
    }

    @Override
    public Single<LoginWithPasswordResult> call(LoginWithPasswordResult result) {
        if (result.getResultCode().equalsIgnoreCase("SSBE002")) {
            LoginWithPasswordCaptchaException exception = new LoginWithPasswordCaptchaException(result.getResultDescription());
            exception.setCaptcha(result.getData().getCaptcha());
            exception.setReferenceId(result.getData().getReferenceId());
            return Single.error(exception);
        }

        if (isNot20000HasNoData(result)) {
            return Single.error(new LoginWithPasswordFailedException(result.getResultDescription()));
        }
        return Single.just(result);
    }
}
