package com.ais.mimo.eservice.helper.login.network.operator;

import com.ais.mimo.eservice.helper.login.network.exception.RequestOtpException;
import com.ais.mimo.eservice.helper.login.network.exception.ShouldRequestCaptchaException;
import com.ais.mimo.eservice.helper.login.network.response.RequestOtpResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public class RequestOtpResultCheck extends ResultHasDataCheck<RequestOtpResult, RequestOtpException> {
    @Override
    protected RequestOtpException getThrowable(String message) {
        return new RequestOtpException(message);
    }

    @Override
    public Single<RequestOtpResult> call(RequestOtpResult result) {
        if (result.getResultCode().equalsIgnoreCase("SSBE309")) {
            ShouldRequestCaptchaException exception = new ShouldRequestCaptchaException(result.getResultDescription());
            exception.setReferenceId(result.getData().getReferenceId());
            return Single.error(exception);
        }

        if (isNot20000HasNoData(result)) {
            return Single.error(new RequestOtpException(result.getResultDescription()));
        }
        return Single.just(result);
    }
}
