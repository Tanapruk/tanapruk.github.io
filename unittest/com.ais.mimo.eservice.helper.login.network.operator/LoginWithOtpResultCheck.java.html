package com.ais.mimo.eservice.helper.login.network.operator;

import com.ais.mimo.eservice.helper.login.network.exception.LoginWithOtpWrong3Exception;
import com.ais.mimo.eservice.helper.login.network.exception.LoginWithOtpWrongException;
import com.ais.mimo.eservice.helper.login.network.response.LoginWithOtpSecretResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public class LoginWithOtpResultCheck extends ResultHasDataCheck<LoginWithOtpSecretResult, LoginWithOtpWrongException> {
    @Override
    protected LoginWithOtpWrongException getThrowable(String message) {
        return new LoginWithOtpWrongException(message);
    }

    @Override
    public Single<LoginWithOtpSecretResult> call(LoginWithOtpSecretResult result) {

        if (result.getDeveloperMessage().equalsIgnoreCase("HACK_TIME value more than 3")) {
            return Single.error(new LoginWithOtpWrong3Exception(result.getResultDescription()));
        }

        if (isNot20000HasNoData(result)) {
            return Single.error(new LoginWithOtpWrongException(result.getResultDescription()));
        }
        return Single.just(result);
    }
}
