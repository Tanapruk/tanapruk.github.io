package com.ais.mimo.eservice.helper.login.network.operator;

import com.ais.mimo.eservice.helper.login.network.response.LoginWithTokensResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;
import com.nextzy.myais.common.passcode.network.exception.LoginWithTokensException;

import rx.Single;

import static com.nextzy.myais.common.passcode.network.PasscodeManager.ERROR_TOUCH_ID_EXPIRED;
import static com.nextzy.myais.common.passcode.network.PasscodeManager.ERROR_TOUCH_ID_FAIL;

/**
 * Created by TheKhaeng on 9/30/2016.
 */

public class LoginWithTokensCheck extends ResultHasDataCheck<LoginWithTokensResult, LoginWithTokensException>{
    @Override
    protected LoginWithTokensException getThrowable(String message) {
        return new LoginWithTokensException(message);
    }

    @Override
    public Single<LoginWithTokensResult> call(LoginWithTokensResult result) {
        if (ERROR_TOUCH_ID_EXPIRED.equals(result.getResultCode())
                || ERROR_TOUCH_ID_FAIL.equals(result.getResultCode())) {
            LoginWithTokensException exception = new LoginWithTokensException(result.getResultDescription());
            exception.setDuplicateOrExpire(true);
            return Single.error(exception);
        }

        if (isNot20000HasNoData(result)) {
            return Single.error(getThrowable(result.getResultDescription()));
        }
        return Single.just(result);
    }

}
