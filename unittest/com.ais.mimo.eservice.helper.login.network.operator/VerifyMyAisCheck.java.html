package com.ais.mimo.eservice.helper.login.network.operator;

import com.ais.mimo.eservice.helper.login.network.exception.InvalidMyAisAccountException;
import com.ais.mimo.eservice.helper.login.network.response.VerifyMyAISResult;
import com.nextzy.myais.common.network.tool.ResultHasDataCheck;

import rx.Single;

/**
 * Created by trusttanapruk on 8/18/2016.
 */
public class VerifyMyAisCheck extends ResultHasDataCheck<VerifyMyAISResult, InvalidMyAisAccountException> {
    @Override
    protected InvalidMyAisAccountException getThrowable(String message) {
        return new InvalidMyAisAccountException(message);
    }

    @Override
    public Single<VerifyMyAISResult> call(VerifyMyAISResult result) {
        if (result.getResultCode().equalsIgnoreCase("SSBE332")) {
            InvalidMyAisAccountException exception = new InvalidMyAisAccountException(result.getResultDescription());
            return Single.error(exception);
        }
        return Single.just(result);
    }
}
