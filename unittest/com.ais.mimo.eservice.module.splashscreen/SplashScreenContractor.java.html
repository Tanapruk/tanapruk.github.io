package com.ais.mimo.eservice.module.splashscreen;

import com.nextzy.myais.common.mvp.NextzyActivityContractorPresenter;
import com.nextzy.myais.common.mvp.NextzyActivityContractorView;

/**
 * Created by Akexorcist on 8/9/16 AD.
 */
public class SplashScreenContractor {
    public interface Presenter extends NextzyActivityContractorPresenter {
        void checkAppUpdate();

        void waitThenGoToLoginActivity();
    }

    public interface View extends NextzyActivityContractorView<Presenter> {
        void checkSmsPermissionBeforeGoToLoginActivity();

        void showAlertDialogAndGotoMainActivity(String message);

        void showAlertDialogAndGotoAppUpdateUrl(String message, String url);

        boolean isActivityVisible();

        String getAppVersion();

        void showAlertDialogUseRoaming();
    }
}
