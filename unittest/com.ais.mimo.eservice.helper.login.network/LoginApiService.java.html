package com.ais.mimo.eservice.helper.login.network;


import com.nextzy.myais.common.network.worker.GenericApiService;

/**
 * Created by Akexorcist on 8/9/2016.
 */
public class LoginApiService extends GenericApiService<LoginApi> {

    public static LoginApiService newInstance(String baseUrl) {
        LoginApiService service = new LoginApiService();
        service.setBaseUrl(baseUrl);
        return service;
    }

    @Override
    protected Class<LoginApi> getApiClassType() {
        return LoginApi.class;
    }
}
