package com.nextzy.myais.common.utility;

import java.util.HashMap;

import rx.Observable;

/**
 * Created by Akexorcist on 9/4/2016 AD.
 */

public class RxObservableManager {
    private static RxObservableManager rxObservableManager;

    public static RxObservableManager getInstance() {
        if (rxObservableManager == null) {
            rxObservableManager = new RxObservableManager();
        }
        return rxObservableManager;
    }

    private HashMap<String, Observable<?>> observableHashMap;

    private RxObservableManager() {
        observableHashMap = new HashMap<>();
    }

    public void add(String key, Observable<?> observable) {
        if (key != null && !key.isEmpty() && observable != null) {
            observableHashMap.put(key, observable);
        }
    }

    public Observable<?> get(String key) {
        Observable<?> observable = observableHashMap.get(key);
        remove(key);
        return observable;
    }

    public void remove(String key) {
        observableHashMap.remove(key);
    }

    public void clear() {
        observableHashMap.clear();
    }
}
