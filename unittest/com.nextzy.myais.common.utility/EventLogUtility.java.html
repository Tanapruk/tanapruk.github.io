package com.nextzy.myais.common.utility;

import com.nextzy.myais.common.config.mobile.Mobile;

import org.json.JSONException;
import org.json.JSONObject;

import java.util.Random;

/**
 * Created by Akexorcist on 7/21/15 AD.
 */
public class EventLogUtility {
    private static EventLogUtility instance;

    public static EventLogUtility getInstance() {
        if (instance == null) {
            instance = new EventLogUtility();
        }
        return instance;
    }

    public static void setInstance(EventLogUtility instance) {
        EventLogUtility.instance = instance;
    }

    public String getEventLog(String eventName, int menuId) {
        String eventLogId = getEventLogId();
        long eventDate = getEventDate();
        JSONObject jsonObject = new JSONObject();
        try {
            jsonObject.put("eventLogId", eventLogId);
            jsonObject.put("eventName", eventName);
            jsonObject.put("eventDate", eventDate);
            jsonObject.put("menuId", menuId);
        } catch (JSONException e) {
            e.printStackTrace();
        }
        return jsonObject.toString();
    }

    private String randomNumber6Digits() {
        Random r = new Random();
        int randomNumber = r.nextInt(1000000);
        return ("000000" + randomNumber).substring(("" + randomNumber).length());
    }

    public long getEventDate() {
        return System.currentTimeMillis();
    }

    public String getEventLogId() {
//        if (Mobile.getAuditLogId().isEmpty()) {
//            NLog.v("getEventLogId", "You haven't setAuditLogId. Do it with Mobile.setAuditLogId(id)");
//        }
        return Mobile.getInstance().getAuditLogId() + "-" + System.currentTimeMillis() + "-" + randomNumber6Digits();
    }
}
