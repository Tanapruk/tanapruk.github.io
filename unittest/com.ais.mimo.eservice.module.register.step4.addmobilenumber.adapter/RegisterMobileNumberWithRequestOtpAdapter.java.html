package com.ais.mimo.eservice.module.register.step4.addmobilenumber.adapter;

import android.support.v7.widget.RecyclerView;
import android.view.LayoutInflater;
import android.view.ViewGroup;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.base.RegisterAdapter;
import com.ais.mimo.eservice.module.register.step4.addmobilenumber.adapter.holder.MobileRequestWithRequestOtpItemHolder;
import com.nextzy.myais.myais.R;

import java.util.ArrayList;


/**
 * Created by Nonthawit on 7/30/2016.
 */

public class RegisterMobileNumberWithRequestOtpAdapter extends RegisterAdapter<RegisterMobileNumberWithRequestOtpContractor.Presenter> implements RegisterMobileNumberWithRequestOtpContractor.View {
    private final static String TAG = RegisterMobileNumberWithRequestOtpAdapter.class.getSimpleName();

    private MobileRequestWithRequestOtpItemHolder.OnClickItemListener listener;

    @Override
    public void createPresenter() {
        RegisterMobileNumberWithRequestOtpPresenter.createPresenter(this);
    }

    public RegisterMobileNumberWithRequestOtpAdapter(){
    }

    public RegisterMobileNumberWithRequestOtpAdapter(ArrayList<MobileData> markConfirmMobileNumberList,
                                                     ArrayList<MobileData> noConfirmMobileNumberList) {
        getPresenter().setMarkConfirmModelList(markConfirmMobileNumberList);
        getPresenter().setNoConfirmModelList(noConfirmMobileNumberList);
    }

    public void addNoConfirmMobileNumber(MobileData mobileNumber) {
        if (mobileNumber != null) {
            getPresenter().addNoConfirmMobileModel(mobileNumber);
            notifyDataSetChanged();
        }
    }

    public void addAllNoConfirmMobileNumber(ArrayList<MobileData> mobileList) {
        if (mobileList != null) {
            getPresenter().addAllNoConfirmModelList(mobileList);
            notifyDataSetChanged();
        }
    }

    public void addMarkConfirmMobileNumber(MobileData mobileNumber) {
        if (mobileNumber != null) {
            getPresenter().addMarkConfirmMobileModel(mobileNumber);
            notifyDataSetChanged();
        }
    }

    public void addAllMarkConfirmMobileNumber(ArrayList<MobileData> mobileList) {
        if (mobileList != null) {
            getPresenter().addAllMarkConfirmModelList(mobileList);
            notifyDataSetChanged();
        }
    }

    public void markConfirmMobileNumberByList(ArrayList<MobileData> mobileList) {
        if (mobileList != null) {
            getPresenter().addAllNoConfirmModelList(mobileList);
            getPresenter().addAllMarkConfirmModelList(mobileList);
            notifyDataSetChanged();
        }
    }


    @Override
    public int getItemCount() {
        return getPresenter().getNoConfirmCount();
    }

    @Override
    public RecyclerView.ViewHolder onCreateViewHolder(ViewGroup viewGroup, int viewType) {
        LayoutInflater mInflater = LayoutInflater.from(viewGroup.getContext());

        ViewGroup vItem = (ViewGroup) mInflater.inflate(R.layout.my_ais_register_view_item_mobile_number_with_request_otp, viewGroup, false);
        return new MobileRequestWithRequestOtpItemHolder(vItem);
    }

    @Override
    public void onBindViewHolder(RecyclerView.ViewHolder viewHolder, int pos) {
        MobileRequestWithRequestOtpItemHolder item = (MobileRequestWithRequestOtpItemHolder) viewHolder;
        item.setPosition(pos);
        item.setText(getPresenter().getNumber(pos));
        if (getPresenter().isPositionMarkConfirm(pos)) {
            item.markConfirm();
            item.hideRequestOtp();
        } else {
            item.markNoConfirm();
        }
        if (listener != null) {
            item.setListener(listener);
        }
        if (!getPresenter().isClickRequestOtpAt(pos)) {
            item.hideRequestOtp();
        } else {
            item.showRequestOtp();
        }
    }

    public void markConfirmCurrentMobileNumber() {
        getPresenter().markConfirmCurrentMobileNumber();
        notifyDataSetChanged();
    }

    public void setListener(MobileRequestWithRequestOtpItemHolder.OnClickItemListener listener) {
        this.listener = listener;
    }


    public void setNoClickAll() {
        getPresenter().noClickAll();
        notifyDataSetChanged();
    }


    public void clickRequestOtpAt(int pos) {
        getPresenter().setClickRequestOtpAt(pos);
        notifyDataSetChanged();
    }

    public boolean isContains(String mobileNumber) {
        return getPresenter().isMobileNumberListContains(mobileNumber);
    }


    /*******************/
    /** Getter Setter **/
    /*******************/
    //<editor-fold desc="Getter Setter folding">
    public ArrayList<MobileData> getConfirmMobileList() {
        return getPresenter().getMarkConfirmMobileDataList();
    }

    public ArrayList<MobileData> getNoConfirmMobileList() {
        return getPresenter().getNoConfirmMobileDataList();
    }

    public int getMarkConfirmItemCount() {
        return getPresenter().getMarkConfirmCount();
    }
}
