package com.ais.mimo.eservice.module.register.step4.addmobilenumber.adapter;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.step4.addmobilenumber.model.AddMobileModel;

import java.util.ArrayList;

/**
 * Created by TheKhaeng on 12/7/2016.
 */

public class AddMobileConverter {

    private static AddMobileConverter converter;

    public static AddMobileConverter getInstance() {
        if (converter == null) {
            converter = new AddMobileConverter();
        }
        return converter;
    }

    public ArrayList<AddMobileModel> convertMobileDataListToAddMobileModelList(ArrayList<MobileData> mobileList) {
        ArrayList<AddMobileModel> modelList = new ArrayList<>();
        for (MobileData mobile : mobileList) {
            modelList.add(convertMobileDataToAddMobileModel(mobile));
        }
        return modelList;
    }

    public AddMobileModel convertMobileDataToAddMobileModel(MobileData mobile) {
        return new AddMobileModel(mobile, false);
    }

    public ArrayList<MobileData> convertAddMobileListToMobileDataList(ArrayList<AddMobileModel> modelList) {
        ArrayList<MobileData> mobileDataList = new ArrayList<>();
        for (AddMobileModel model : modelList) {
            mobileDataList.add(convertAddMobileModelToMobileData(model));
        }
        return mobileDataList;
    }

    public MobileData convertAddMobileModelToMobileData(AddMobileModel model) {
        return model.getMobileData();
    }

}
