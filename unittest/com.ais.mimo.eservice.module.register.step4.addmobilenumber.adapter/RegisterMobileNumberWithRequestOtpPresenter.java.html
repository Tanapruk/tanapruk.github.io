package com.ais.mimo.eservice.module.register.step4.addmobilenumber.adapter;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.ais.mimo.eservice.module.register.step4.addmobilenumber.model.AddMobileModel;
import com.nextzy.myais.common.mvp.NextzyPresenter;

import java.util.ArrayList;

import javax.annotation.Nonnull;
import javax.annotation.Nullable;

/**
 * Created by TheKhaeng on 10/3/2016.
 */

public class RegisterMobileNumberWithRequestOtpPresenter extends NextzyPresenter<RegisterMobileNumberWithRequestOtpContractor.View> implements RegisterMobileNumberWithRequestOtpContractor.Presenter {

    private final AddMobileConverter converter;
    private ArrayList<AddMobileModel> markConfirmModelList;
    private ArrayList<AddMobileModel> noConfirmModelList;
    private int focusPos;

    public static RegisterMobileNumberWithRequestOtpPresenter createPresenter(RegisterMobileNumberWithRequestOtpContractor.View view) {
        return new RegisterMobileNumberWithRequestOtpPresenter(view);
    }

    private RegisterMobileNumberWithRequestOtpPresenter(RegisterMobileNumberWithRequestOtpContractor.View view) {
        super(view);
        converter = AddMobileConverter.getInstance();
        markConfirmModelList = new ArrayList<>();
        noConfirmModelList = new ArrayList<>();
    }

    @Override
    public void onViewCreate() {

    }

    @Override
    public void onViewDestroy() {

    }

    @Override
    public void onViewStart() {

    }

    @Override
    public void onViewStop() {

    }

    @Override
    public void setNoConfirmModelList(@Nullable ArrayList<MobileData> mobileList) {
        if (mobileList != null) this.noConfirmModelList = converter.convertMobileDataListToAddMobileModelList(mobileList);
    }

    @Override
    public void setMarkConfirmModelList(@Nullable ArrayList<MobileData> mobileList) {
        if (mobileList != null) this.markConfirmModelList = converter.convertMobileDataListToAddMobileModelList(mobileList);
    }

    @Override
    public void addNoConfirmMobileModel(@Nonnull MobileData mobileNumber) {
        AddMobileModel model = converter.convertMobileDataToAddMobileModel(mobileNumber);
        if (!getNoConfirmMobileDataList().contains(model.getMobileData())) {
            noConfirmModelList.add(model);
        }
    }

    @Override
    public void addAllNoConfirmModelList(@Nonnull ArrayList<MobileData> mobileNumberList) {
        ArrayList<AddMobileModel> modelList = converter.convertMobileDataListToAddMobileModelList(mobileNumberList);
        for (AddMobileModel model : modelList) {
            addNoConfirmMobileModel(model.getMobileData());
        }
    }

    @Override
    public void addMarkConfirmMobileModel(@Nonnull MobileData mobileNumber) {
        AddMobileModel model = converter.convertMobileDataToAddMobileModel(mobileNumber);
        if (!getMarkConfirmMobileDataList().contains(model.getMobileData())) {
            markConfirmModelList.add(model);
        }
    }

    @Override
    public void addAllMarkConfirmModelList(@Nonnull ArrayList<MobileData> mobileNumberList) {
        ArrayList<AddMobileModel> modelList = converter.convertMobileDataListToAddMobileModelList(mobileNumberList);
        for (AddMobileModel model : modelList) {
            addMarkConfirmMobileModel(model.getMobileData());
        }
    }


    @Override
    public int getNoConfirmCount() {
        if (noConfirmModelList == null) {
            return 0;
        }
        return noConfirmModelList.size();
    }

    @Override
    public int getMarkConfirmCount() {
        if (markConfirmModelList == null) {
            return 0;
        }
        return markConfirmModelList.size();
    }

    @Override
    public String getNumber(int pos) {
        return noConfirmModelList.get(pos).getNumber();
    }

    @Override
    public boolean isPositionMarkConfirm(int pos) {
        return getMarkConfirmMobileDataList().contains(getNoConfirmMobileDataList().get(pos));
    }

    @Override
    public boolean isMobileNumberListContains(String mobileNumber) {
        if (mobileNumber != null) {
            for (AddMobileModel registerMyNumberData : noConfirmModelList) {
                if (mobileNumber.equals(registerMyNumberData.getNumber())) {
                    return true;
                }
            }

            for (AddMobileModel registerMyNumberData : markConfirmModelList) {
                if (mobileNumber.equals(registerMyNumberData.getNumber())) {
                    return true;
                }
            }
        }
        return false;
    }

    @Override
    public boolean isClickRequestOtpAt(int pos) {
        return noConfirmModelList.get(pos).isClickRequestOtp();
    }

    public ArrayList<MobileData> getNoConfirmMobileDataList() {
        return converter.convertAddMobileListToMobileDataList(noConfirmModelList);
    }

    public ArrayList<MobileData> getMarkConfirmMobileDataList() {
        return converter.convertAddMobileListToMobileDataList(markConfirmModelList);
    }


    @Override
    public void setClickRequestOtpAt(int pos) {
        focusPos = pos;
        for (int i = 0; i < noConfirmModelList.size(); i++) {
            if (pos == i) {
                noConfirmModelList.get(i).setClickRequestOtp(true);
            } else {
                noConfirmModelList.get(i).setClickRequestOtp(false);
            }
        }
    }

    @Override
    public void noClickAll() {
        for (AddMobileModel model : noConfirmModelList) {
            model.setClickRequestOtp(false);
        }
    }

    @Override
    public void markConfirmCurrentMobileNumber() {
        addNoConfirmMobileModel(noConfirmModelList.get(focusPos).getMobileData());
        addMarkConfirmMobileModel(noConfirmModelList.get(focusPos).getMobileData());
        setNoClickAll();
    }


    /********************/
    /** Private method **/
    /********************/
    //<editor-fold desc="Private function folding">
    private void setNoClickAll() {
        for (AddMobileModel mobile : noConfirmModelList) {
            mobile.setClickRequestOtp(false);
        }
    }
    //</editor-fold>
}
