    public void mockFillEditText(View view) {
        if (BuildConfig.DEBUG) {
            Button btnMock = (Button) view.findViewById(R.id.mock_number);
            btnMock.setVisibility(View.VISIBLE);
            btnMock.setOnClickListener(view1 -> {
                if (countMockNumber >= numbers.length) {
                    countMockNumber = 0;
                }
                vMobileNumber.setText(numbers[countMockNumber]);
                vIdCard.setText(idCards[countMockNumber]);
                vEmail.setText("nonthawit@gmail.com");
                countMockNumber += 1;

    @Override
    public void setupView() {
        dismissDialog();
        setupClickHideKeyboard(baseLayout);
        setIndexIndicator(0);
        setBackgroundWhite();
        vEmail.setInputType(InputType.TYPE_CLASS_TEXT | InputType.TYPE_TEXT_VARIATION_EMAIL_ADDRESS);
        vBtnCancelNext.setBtnFirstText(getContext().getString(R.string.my_ais_register_back));
        vBtnCancelNext.setBtnSecondText(getContext().getString(R.string.my_ais_register_next_request_otp));
        vBtnCancelNext.setOnFirstButtonClickListener(v -> clickCancel());
        vBtnCancelNext.setOnSecondButtonClickListener(v -> clickNext());

        if(LoginBaseFragment.TAG.equals(getArguments().getString(KEY_COME_FROM_TAG))){
            vMobileNumber.setText(getArguments().getString(KEY_MOBILE_NUMBER));
        } else if(getPresenter().isLogin()){
            vMobileNumber.setText(Mobile.getInstance().getMobileNumber());
            vMobileNumber.setEditable(false);
        }
    }

    @Override
    public void showDialogInvalidateMobileNumber() {
        vMobileNumber.showError();
        String error;
        if (vMobileNumber.getStringFromEditText().isEmpty()) {
            error = getString(R.string.my_ais_register_dialog_mobile_number_none);
        } else {
            error = getString(R.string.my_ais_register_dialog_mobile_number_format_fault);
        }
        ((NextzyMvpActivity) getActivity())
                .showAlertDialogDismissOnClick(error);
    }


    @Override
    public void showDialogInvalidateIdCard() {
