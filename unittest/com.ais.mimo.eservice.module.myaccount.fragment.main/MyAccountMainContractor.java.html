package com.ais.mimo.eservice.module.myaccount.fragment.main;

import com.ais.mimo.eservice.helper.login.network.response.MobileData;
import com.nextzy.myais.common.mvp.NextzyContractorPresenter;
import com.nextzy.myais.common.mvp.NextzyContractorView;

import java.util.ArrayList;

/**
 * Created by TheKhaeng on 11/16/2016.
 */

public class MyAccountMainContractor {
    public interface Presenter extends NextzyContractorPresenter {
        void registerRx();

        void unregisterRx();

        void updateMyAccountUi();

        void requestVerifyMyAis();


        void goToEditAccountFragment();

        void goToEditEmailFragment();

        void goToEditPasswordFragment();

        void goToAddMobileActivity(
                ArrayList<MobileData> markConfirmPostpaidList,
                ArrayList<MobileData> markConfirmPrepaidList,
                ArrayList<MobileData> markConfirmFbbList,
                ArrayList<MobileData> noConfirmPostpaidList,
                ArrayList<MobileData> noConfirmPrepaidList,
                ArrayList<MobileData> noConfirmFbbList
        );
    }

    public interface View extends NextzyContractorView<Presenter> {

        void setMobileAccountsToAdapter(ArrayList<MobileData> post,
                                        ArrayList<MobileData> pre,
                                        ArrayList<MobileData> fbb);

        void openAddMobileActivity(
                String mobileNumber,
                String email, ArrayList<MobileData> confirmPreList, ArrayList<MobileData> confirmFbbList, ArrayList<MobileData> noConfirmPostList, ArrayList<MobileData> noConfirmPreList, ArrayList<MobileData> noConfirmFbbList, ArrayList<MobileData> confirmPostList);

        void showResponseException();

        void showResponseException(String resultDescription);

        void updateUi(String mobileNumber, String email);

        void showLoadingDialog();

        void dismissDialog(int delay);
    }
}
