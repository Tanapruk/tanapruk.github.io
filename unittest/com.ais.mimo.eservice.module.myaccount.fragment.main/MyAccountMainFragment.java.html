    private AISTextView tvEditPassword;

    @Override
    public void bindView(View view) {
        tvAccountNumber = (AISTextView) view.findViewById(R.id.my_ais_account_tv_account_number);
        tvEmail = (AISTextView) view.findViewById(R.id.my_ais_account_tv_email);
        tvEditMobileNumber = (AISTextView) view.findViewById(R.id.my_ais_account_tv_edit_mobile_number);
        tvEditEmail = (AISTextView) view.findViewById(R.id.my_ais_account_tv_edit_email);
//        btnChangePassword = (AISButton) view.findViewById(R.id.my_ais_account_btn_change_password);
        tvEditPassword = (AISTextView) view.findViewById(R.id.my_ais_account_tv_edit_password);
        vMobileAccounts = (MyAisRegisterMobileAccountView) view.findViewById(R.id.my_ais_account_mobile_accounts);
    }

    @Override
    public void setupView() {
        vMobileAccounts.setOnClickAddListener(v -> clickAdd());
        vMobileAccounts.setHeader(getString(R.string.my_ais_account_header_add_mobile));
        vMobileAccounts.setDescAddMobile(getString(R.string.my_ais_account_desc_add_mobile));

        setUnderScroll(tvEditMobileNumber);
        setUnderScroll(tvEditEmail);
        setUnderScroll(tvEditPassword);

        tvEditMobileNumber.setOnClickListener(view -> clickEditMobileNumber());

        tvEditEmail.setOnClickListener(view -> clickEditEmail());
        tvEditPassword.setOnClickListener(view -> clickEditPassword());

//        btnChangePassword.setOnClickListener(view -> clickChangePassword());
    }

    @Override
    public void updateUi(String accountNumber, String email) {
        setAccountNumber(accountNumber);
        setEmail(email);
    }

    @Override
    public void setMobileAccountsToAdapter(ArrayList<MobileData> post,
                                           ArrayList<MobileData> pre,
                                           ArrayList<MobileData> fbb) {
        postAdapter = new MyAccountMobileNumberAdapter(post);
        preAdapter = new MyAccountMobileNumberAdapter(pre);
        fbbAdapter = new MyAccountMobileNumberAdapter(fbb);
        vMobileAccounts.setPostAdapter(postAdapter);
        vMobileAccounts.setPreAdapter(preAdapter);
        vMobileAccounts.setFbbAdapter(fbbAdapter);
    }

    @Override
    public void openAddMobileActivity(String mobileNumber,
                                      String email,
                                      ArrayList<MobileData> confirmPreList,
                                      ArrayList<MobileData> confirmFbbList,
                                      ArrayList<MobileData> noConfirmPostList,
                                      ArrayList<MobileData> noConfirmPreList,
                                      ArrayList<MobileData> noConfirmFbbList,
                                      ArrayList<MobileData> confirmPostList) {
        Intent i = new Intent(getActivity(), MyAccountAddNumberActivity.class);
        Bundle data = new Bundle();
        data.putString(KEY_MOBILE_NUMBER, mobileNumber);
        data.putString(KEY_EMAIL, email);
        data.putParcelableArrayList(KEY_CONFIRM_POSTPAID_LIST_FOR_CHECK_USED, confirmPostList);
        data.putParcelableArrayList(KEY_NO_CONFIRM_POSTPAID_LIST_FOR_CHECK_USED, noConfirmPostList);
        data.putParcelableArrayList(KEY_CONFIRM_PREPAID_LIST_FOR_CHECK_USED, confirmPreList);
        data.putParcelableArrayList(KEY_NO_CONFIRM_PREPAID_LIST_FOR_CHECK_USED, noConfirmPreList);
        data.putParcelableArrayList(KEY_CONFIRM_FBB_LIST_FOR_CHECK_USED, confirmFbbList);
        data.putParcelableArrayList(KEY_NO_CONFIRM_FBB_LIST_FOR_CHECK_USED, noConfirmFbbList);
        i.putExtra(EXTRA_DATA, data);
        startActivityForResult(i, START_ADD_MOBILE);
    }

    @Override
    public void showResponseException() {
        showAlertDialogDismissOnClick(R.string.my_ais_server_unavailable);
    }

    @Override
    public void showResponseException(String resultDescription) {
        showAlertDialogDismissOnClick(resultDescription);
    }

    //<editor-fold desc="Private function folding">
    private void clickEditMobileNumber() {
        getPresenter().goToEditAccountFragment();
    }

    private void clickEditEmail() {
        getPresenter().goToEditEmailFragment();
    }

    private void clickEditPassword() {
        getPresenter().goToEditPasswordFragment();
    }


    private void setAccountNumber(String accountNumber) {
        tvAccountNumber.setText(accountNumber, FORMAT_MOBILE_NUMBER);
    }

    private void clickAdd() {
        if (postAdapter != null && preAdapter != null && fbbAdapter != null) {
            getPresenter().goToAddMobileActivity(
                    postAdapter.getConfirmMobileList(),
                    preAdapter.getConfirmMobileList(),
                    fbbAdapter.getConfirmMobileList(),
                    postAdapter.getNoConfirmMobileList(),
                    preAdapter.getNoConfirmMobileList(),
                    fbbAdapter.getNoConfirmMobileList()
            );
        }
    }

    private void setEmail(String email) {
        tvEmail.setText(email);
    }


    private void setUnderScroll(TextView view) {
        view.setClickable(true);
        view.setPaintFlags(view.getPaintFlags() | Paint.UNDERLINE_TEXT_FLAG);
    }
    //</editor-fold>
}
