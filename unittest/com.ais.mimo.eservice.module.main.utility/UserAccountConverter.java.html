package com.ais.mimo.eservice.module.main.utility;

import com.nextzy.myais.common.config.mobile.MobileContact;
import com.nextzy.myais.common.config.mobile.MobileType;
import com.nextzy.myais.common.utility.NLog;
import com.nextzy.myais.common.widget.toolbar.userdetail.MyAisToolbarUserDetailSwitchAccountMenu;
import com.nextzy.myais.myais.R;

import java.util.ArrayList;
import java.util.List;

/**
 * Created by Akexorcist on 8/24/16 AD.
 */
public class UserAccountConverter {
    private static UserAccountConverter converter;

    public static UserAccountConverter getInstance() {
        if (converter == null) {
            converter = new UserAccountConverter();
        }
        return converter;
    }

    public List<MyAisToolbarUserDetailSwitchAccountMenu.ContentItem> convertUserAccountToSwitchAccountMenuItem(List<MobileContact> mobileContactList) {
        List<MyAisToolbarUserDetailSwitchAccountMenu.ContentItem> contentItemList = new ArrayList<>();
        if (mobileContactList != null) {
            for (MobileContact mobileContact : mobileContactList) {
                String accountNumber = mobileContact.getMobileNo();
                int iconResId = getBusinessTypeImageResource(mobileContact.getMobileType());
                MyAisToolbarUserDetailSwitchAccountMenu.ContentItem contentItem = new MyAisToolbarUserDetailSwitchAccountMenu.ContentItem(iconResId, accountNumber);
                contentItemList.add(contentItem);
            }
            if (contentItemList.size() > 0) {
                contentItemList.get(0).setSelected(true);
            }
        }
        return contentItemList;
    }

    private int getBusinessTypeImageResource(MobileType mobileType) {
        if (MobileType.FBB.equals(mobileType)) {
            return R.drawable.my_ais_user_account_selector_mobile_type_fibre;
        } else if (MobileType.PREPAID.equals(mobileType)) {
            return R.drawable.my_ais_user_account_selector_mobile_type_prepaid;
        } else if (MobileType.POSTPAID.equals(mobileType)) {
            return R.drawable.my_ais_user_account_selector_mobile_type_postpaid;
        }
        NLog.e(this.getClass().getSimpleName(), "No Image Resource for " + mobileType + "Postpaid image is used instead");
        return R.drawable.my_ais_user_account_selector_mobile_type_postpaid;
    }
}
