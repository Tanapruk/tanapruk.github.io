package com.ais.mimo.eservice.helper.register.network.realm;

import com.google.gson.Gson;
import com.nextzy.myais.common.database.model.GenericSaveRealm;
import com.nextzy.myais.common.network.model.GenericResult;
import com.nextzy.myais.common.utility.NLog;

import io.realm.exceptions.RealmException;
import rx.functions.Func1;

import static com.ais.mimo.eservice.helper.register.network.RegisterManager.SSBE309;

/**
 * Created by TheKhaeng on 10/26/2016.
 */

public class RegisterSaveRealmResult<T> extends GenericSaveRealm<T> implements Func1<T, T> {
    private String serviceName;

    public RegisterSaveRealmResult() {
    }

    public RegisterSaveRealmResult(String serviceName) {
        this.serviceName = serviceName;
    }

    @Override
    public T call(T t) {
        if (t == null) {
            deleteDataByServiceName(serviceName);
            throw new RealmException("No Realm Object");
        }
        if (t instanceof GenericResult) {
            GenericResult result = (GenericResult) t;
            if (!"20000".equalsIgnoreCase(result.getResultCode())
                    && !SSBE309.equals(result.getResultCode())) {
                NLog.d(getClass().getSimpleName(), "this data won't be kept because we received a failure response from server. " + new Gson().toJson(result));
                deleteDataByServiceName(serviceName);
                return t;
            }
        }
        return saveToRealm(t, serviceName);
    }
}
