package com.ais.mimo.eservice.helper.splashscreen.network.operator;

import com.nextzy.myais.common.network.exception.NoInternetConnectionException;
import com.nextzy.myais.common.network.exception.ServerNotFoundException;
import com.nextzy.myais.common.network.tool.SubscribingState;
import com.nextzy.myais.common.utility.NLog;

import java.net.ConnectException;
import java.net.UnknownHostException;

import retrofit2.adapter.rxjava.HttpException;
import rx.Single;
import rx.android.schedulers.AndroidSchedulers;
import rx.schedulers.Schedulers;

/**
 * Created by trusttanapruk on 8/22/2016.
 */
public class InternationalSchedulersAndErrorHandling<T> implements Single.Transformer<T, T> {
    protected final String TAG = this.getClass().getSimpleName();
    Class<T> typeOfResultClass;
    String serviceName;

    public InternationalSchedulersAndErrorHandling(Class<T> typeOfResultClass, String serviceName) {
        this.typeOfResultClass = typeOfResultClass;
        this.serviceName = serviceName;
    }

    @Override
    public Single<T> call(Single<T> tSingle) {
        SubscribingState.getInstance().start(serviceName);
        return tSingle
                .subscribeOn(Schedulers.io())
                .onErrorResumeNext(throwable -> {
                    return handleException(throwable, typeOfResultClass);
                })
                .observeOn(AndroidSchedulers.mainThread());
    }


    private void printLog(String errorName) {
        NLog.e(TAG, String.format("\n%s\nservice:\t %s", errorName, serviceName));
    }

    private Single<T> handleException(Throwable throwable, Class<T> type) {
        if (throwable instanceof UnknownHostException || throwable instanceof ConnectException) {
            printLog(throwable.getClass().getName());
            return Single.error(new NoInternetConnectionException(throwable.getMessage()));
        }
        if (!(throwable instanceof HttpException)) {
            printLog(throwable.getClass().getName());
            return Single.error(throwable);
        }

        HttpException httpException = (HttpException) throwable;
        if (httpException.response() == null || httpException.response().errorBody() == null) {
            printLog(httpException.getClass().getName());
            return Single.error(new NullPointerException("Null response() or response().errorBody()"));
        }

        if (httpException.response().code() == 404) {
            // TODO: 8/2/2016 response().errorBody() cached the previous repeated request
            //if we don't do like this then httpException
            // may get response from its cache and successfully convert to Json
            // cannot send out other object, too.
            printLog(httpException.getClass().getName());
            return Single.error(new ServerNotFoundException());
        }
        return Single.error(throwable);
    }
}
