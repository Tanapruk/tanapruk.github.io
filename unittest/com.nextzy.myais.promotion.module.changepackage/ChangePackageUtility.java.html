    public String getBillingCycleText(BillingCycleResult billingCycleResult) {
        if (billingCycleResult != null && billingCycleResult.getData() != null) {
            return billingCycleResult.getData().getBillCycleText();
        }
        return "-";
    }

    public String getPackageGroupDescription(PackageGroupData packageGroupData) {
        if (packageGroupData != null && packageGroupData.getDescription() != null) {
            return packageGroupData.getDescription();
        }
        return "-";
    }

    public PackageGroupData getCategoryByTabCode(String tabCode, PackageGroupResult packageGroupResult) {
        if (packageGroupResult != null && packageGroupResult.getDataList() != null) {
            for (PackageGroupData packageGroupData : packageGroupResult.getDataList()) {
                if (packageGroupData != null) {
                    List<PackageGroupData.ChildGroup> childGroupList = packageGroupData.getChildGroupList();
                    if (childGroupList != null) {
                        for (PackageGroupData.ChildGroup childGroup : childGroupList) {
                            if (tabCode.equals(childGroup.getTabCode())) {
                                return createPackageGroupDataFromChildGroup(childGroup);
                            }
                        }
                    }
                    if (tabCode.equals(packageGroupData.getTabCode())) {
                        return packageGroupData;
                    }
                }
            }
        }
        return null;
    }

    private PackageGroupData createPackageGroupDataFromChildGroup(PackageGroupData.ChildGroup childGroup) {
        PackageGroupData childPackageGroupData = new PackageGroupData();
        childPackageGroupData.setTitle(childGroup.getTitle());
        childPackageGroupData.setName(childGroup.getName());
        childPackageGroupData.setActionCode(childGroup.getActionCode());
        childPackageGroupData.setDescription(childGroup.getDescription());
        childPackageGroupData.setTabCode(childGroup.getTabCode());
        childPackageGroupData.setEffectiveOptionList(childGroup.getEffectiveOptionList());
        return childPackageGroupData;
    }

    public PackageData.PackageDetail getPackageDetailById(String packageId, PackageResult packageResult) {
        if (packageResult != null && packageResult.getData() != null) {
            if (packageResult.getData().getPackageDetailList() != null) {
                for (PackageData.PackageDetail packageDetail : packageResult.getData().getPackageDetailList()) {
                    if (packageId.equals(packageDetail.getId())) {
                        return packageDetail;
                    }
                }
            }
            if (packageResult.getData().getGroupList() != null) {
                for (PackageData.Group packageGroup : packageResult.getData().getGroupList()) {
                    if (packageGroup.getPackageDetailList() != null) {
                        for (PackageData.PackageDetail packageDetail : packageGroup.getPackageDetailList()) {
                            if (packageId.equals(packageDetail.getId())) {
                                return packageDetail;
                            }
                        }
                    }
                }
            }
        }
        return null;
    }

    private Calendar convertStringDateToCalendar(String strDate) {
        Calendar cal = null;
        try {
            DateFormat formatter = new SimpleDateFormat("d/MM/yyyy", Locale.US);
            Date date = formatter.parse(strDate);
            cal = Calendar.getInstance();
            cal.setTime(date);
        } catch (ParseException e) {
            e.printStackTrace();
        }
        return cal;
    }

    private String getBillingCycleFinishDate(String billingCycleStartDate, String language) {
        Calendar calendar = convertStringDateToCalendar(billingCycleStartDate);
        calendar.add(Calendar.MONTH, 1);
        String finishDate = new SimpleDateFormat("dd/MM/yyyy", Locale.getDefault()).format(calendar.getTime());
        return NextzyUtility.getInstance().convertYearAD2BE(finishDate, "/", language);
    }

    public String getNextDayDate(String language) {
        Calendar calendar = Calendar.getInstance();
        calendar.add(Calendar.DAY_OF_MONTH, 1);
        String nextDate = new SimpleDateFormat("dd/MM/yyyy", Locale.getDefault()).format(calendar.getTime());
        return NextzyUtility.getInstance().convertYearAD2BE(nextDate, "/", language);
    }

    public String getNextBillDate(BillingCycleResult billingCycleResult, String language) {
        if (billingCycleResult != null && billingCycleResult.getData() != null) {
            return getBillingCycleFinishDate(billingCycleResult.getData().getStartDate(), language);
        }
        return "-";
    }

    public PackageGroupData getCategoryByTabCode2(String tabCode, PackageGroupResult packageGroupResult) {
        if (packageGroupResult != null && packageGroupResult.getDataList() != null) {
            for (PackageGroupData data : packageGroupResult.getDataList()) {
                if (data.getTabCode().equals(tabCode)) {
                    return data;
                }
            }
        }
        return null;
    }

    public PackageGroupData getChildGroupFromActionCode(String categoryTabCode, PackageGroupResult packageGroupResult) {
        if (packageGroupResult != null) {
            for (PackageGroupData packageGroupData : packageGroupResult.getDataList()) {
                if (packageGroupData.getTabCode().equals(categoryTabCode)) {
                    return packageGroupData;
                }
            }
        }
        return null;
    }
