package com.ais.mimo.eservice.helper.login.event;

/**
 * Created by trusttanapruk on 10/23/2016.
 */
public class GotoLoginRegisterEvent {

    private final String comeFromTag;


    private final String mobileNumber;

    public GotoLoginRegisterEvent(String mobileNumber, String tag) {
        this.mobileNumber = mobileNumber;
        this.comeFromTag = tag;
    }


    public String getComeFromTag() {
        return comeFromTag;
    }

    public String getMobileNumber() {
        return mobileNumber;
    }
}
