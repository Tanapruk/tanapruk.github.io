package com.ais.mimo.eservice.helper.login.event;

import com.ais.mimo.eservice.module.login.LoginKey;

/**
 * Created by trusttanapruk on 10/13/2016.
 */
public class GotoLoginPasswordEvent {
    String otpOrFragment;
    String referenceId;
    String captcha;
    String otpMethod;

    public GotoLoginPasswordEvent(String referenceId, String otpMethod) {
        this.referenceId = referenceId;
        this.otpMethod = otpMethod;
    }

    public GotoLoginPasswordEvent() {
    }

    public String getOtpOrPasswordFragment() {
        return otpOrFragment;
    }

    public GotoLoginPasswordEvent setGotoSms() {
        this.otpOrFragment = LoginKey.VERIFICATION_TYPE_SMS;
        return this;
    }

    public GotoLoginPasswordEvent setGotoEmail() {
        this.otpOrFragment = LoginKey.VERIFICATION_TYPE_EMAIL;
        return this;
    }

    public GotoLoginPasswordEvent setGotoPassword() {
        this.otpOrFragment = LoginKey.VERIFICATION_TYPE_PASSWORD;
        return this;
    }

    public String getReferenceId() {
        return referenceId;
    }

    public String getOtpMethod() {
        return otpMethod;
    }

    public String getCaptcha() {
        return captcha;
    }

    public void setCaptcha(String captcha) {
        this.captcha = captcha;
    }
}
